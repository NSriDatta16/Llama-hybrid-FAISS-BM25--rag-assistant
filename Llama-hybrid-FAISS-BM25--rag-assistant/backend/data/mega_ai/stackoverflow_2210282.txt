How about this? SELECT d.mac, r.name, a.name FROM device as d, global_user as g, agent as a, reseller as r WHERE d.global_user_id = g.id AND g.id = a.global_user_id AND (g.id = r.global_user_id OR a.reseller_id = r.id) AND (r.id = '200' OR a.reseller_id = '200');
