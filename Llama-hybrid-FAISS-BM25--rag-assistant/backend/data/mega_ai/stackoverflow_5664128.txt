Add accept: "#selector" as an option to the droppable plugin: $(function(){ $("#piece1").draggable({ revert: "invalid" }); $("#piece1drop").droppable({ drop: function() { $('#piece1').hide(); $('#piece1drop').css('background-image', 'url("piece1placed.png")'); }, accept: "#piece1" }); $("#piece2").draggable({ revert: "invalid" }); $("#piece2drop").droppable({ drop: function() { $('#piece2').hide(); $('#pin2drop').css('background-image', 'url("piece2placed.png")'); }, accept: "#piece2" }); }); Here's the jQuery Demo
