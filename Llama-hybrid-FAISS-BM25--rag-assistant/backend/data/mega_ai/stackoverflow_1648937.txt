complete example (FF only) var angle = 10; window.onload = function() { var d = document.getElementById("drag"); d.onmousedown = function() { document.onmouseup = function() { document.onmousemove = null; } document.onmousemove = function(e) { d.style.left = e.clientX; d.style.top = e.clientX * Math.tan(angle * Math.PI / 180); } } }
