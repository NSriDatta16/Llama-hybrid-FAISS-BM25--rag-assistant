To fix this problem, try one of below. Use #pragma GCC visibility push() statement like this. #pragma GCC visibility push(hidden) struct MainLockDetector : Action { bool wait(Uint32 time) { /* ... */ } int handle() { /* ... */ } }; #pragma GCC visibility pop Use __attribute__ ((visibility("hidden"))) like this. struct __attribute__ ((visibility("hidden"))) MainLockDetector : Action { bool wait(Uint32 time) { /* ... */ } int handle() { /* ... */ } }; Add the command line option -fvisibility=default. For more details, refer http://gcc.gnu.org/wiki/Visibility .
