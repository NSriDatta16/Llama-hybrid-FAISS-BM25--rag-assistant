It is marginally more clear to build your statement using an array: $params = array(); $fragments = array(); foreach($_POST as $col => $val) { $fragments[] = "{$col} = ?"; $params[] = $val; } $sql = sprintf("UPDATE sometable SET %s", implode(", ", $fragments)); $stmt = $mysqli->prepare($sql); $stmt->bind_param($params);
