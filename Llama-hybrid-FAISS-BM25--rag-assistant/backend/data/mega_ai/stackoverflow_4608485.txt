from BeautifulSoup import BeautifulSoup def removeTags(html, *tags): soup = BeautifulSoup(html) for tag in tags: for tag in soup.findAll(tag): tag.replaceWith("") return soup testhtml = ''' Page title text here This is paragraph one . This is paragraph two . ''' print removeTags(testhtml, 'b', 'p')
