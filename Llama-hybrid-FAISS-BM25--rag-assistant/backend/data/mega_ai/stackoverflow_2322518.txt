It turns out that it's easy enough to get the location for the Show command from the UltraGrid by querying the UIElement associated with it. Here's what I'm doing: private void ultraGrid1_BeforeCellUpdate(object sender, BeforeCellUpdateEventArgs e) { if (!DataFormat.CanEdit(e.Cell.Row.ListObject, e.Cell.Column.PropertyDescriptor)) { var tip = new System.Windows.Forms.ToolTip(); tip.BackColor = Color.Orange; tip.Show("unable to edit", this, e.Cell.GetUIElement().Rect.Left, e.Cell.GetUIElement().Rect.Top, 500); e.Cancel = true; } }
