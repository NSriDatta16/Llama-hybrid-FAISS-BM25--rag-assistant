navigator.sayswho= (function(){ var N= navigator.appName, ua= navigator.userAgent, tem; var M= ua.match(/(opera|chrome|safari|firefox|msie)\/?\s*(\.?\d+(\.\d+)*)/i); if(M && (tem= ua.match(/version\/([\.\d]+)/i))!= null) M[2]= tem[1]; M= M? [M[1], M[2]]: [N, navigator.appVersion,'-?']; return M.join(' '); })(); as the name suggests, this is who the browser says it is- but use object detection before asking it to actually do anything... I use it for logging errors from users and in testing code in multiple browsers- where I know the userAgent strings.
