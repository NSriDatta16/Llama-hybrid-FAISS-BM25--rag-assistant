Just for show the way how, I strongly advise you not to use this "elegant" ugly way: >>> data [0, 0, 1, 8, 8, 8, 4, 4, 1, 0] >>> def fromDataToSplitRLE(dat,n): RLE=[(k,len(tuple(g))) for k,g in itertools.groupby(dat)] tmp=tuple(zip(*RLE)) return [list(g) for k,g in itertools.groupby((zip((sum(tmp[1][:i]) for i in range(len(tmp[1]))) ,(zip(*tmp)))),lambda x:x[1][0]!=n) if k] >>> fromDataToSplitRLE(data,1) [[(0, (0, 2))], [(3, (8, 3)), (6, (4, 2))], [(9, (0, 1))]]
