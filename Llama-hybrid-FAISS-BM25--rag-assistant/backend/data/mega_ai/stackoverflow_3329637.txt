One option for loading text into RichTextBox is to use XamlReader. Dependent on the text which you are planning to load, you might need to add tag around it public class TextToXamlConverter { private const String ParaHead = " "; private const String ParaTail = " "; static public Paragraph Convert(string text) { String formattedText = ParaHead + text + ParaTail; Paragraph p = (Paragraph)XamlReader.Load(formattedText); return p; } }
