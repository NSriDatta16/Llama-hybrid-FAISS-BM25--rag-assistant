I found a non-elegant solution that works var result = list.GroupBy(i=>i.first); foreach (IGrouping grp in result) { grp.Aggregate ((f1, f2) => { return new Foo() { b = f1.b | f2.b, i = f1.i | f2.i, first = f1.first, last = f1.last }; }); }
