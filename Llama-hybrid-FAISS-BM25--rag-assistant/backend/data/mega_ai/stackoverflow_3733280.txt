Found the solution to my problem myself. This code myself will set the Cursor position to the last Paragraph of my editor: var frame = window.content.document.getElementsByTagName('iframe')[2]; var win = frame.contentWindow; var editingSession = Components.classes["@mozilla.org/editor/editingsession;1"].createInstance(Components.interfaces.nsIEditingSession); var editor = editingSession.getEditorForWindow(win); selection = window.getSelection(); var body = frame.contentDocument.body; text = frame.contentDocument.createTextNode("."); body.lastChild.appendChild(text); // add textnode to be selected var range = editor.document.createRange(); range.setStartBefore(text); range.setEndAfter(text); editor.selection.removeAllRanges(); editor.selection.addRange(range); body.lastChild.removeChild(text); // remove Child
