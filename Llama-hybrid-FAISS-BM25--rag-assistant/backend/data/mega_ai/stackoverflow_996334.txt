By starting the sync agent in a separate thread on the client, the callback works just fine: private int kickOffSyncInSeparateThread() { SyncRunner syncRunner = new SyncRunner(); Thread syncThread = new Thread( new ThreadStart(syncRunner.RunSyncInThread)); try { syncThread.Start(); } catch (ThreadStateException ex) { Console.WriteLine(ex); return 1; } catch (ThreadInterruptedException ex) { Console.WriteLine(ex); return 2; } return 0; } And this is my SyncRunner: class SyncRunner { public void RunSyncInThread() { MysyncAgent = new MySyncAgent(); syncAgent.addUserIdParameter("56623239-d855-de11-8e97-0016cfe25fa3"); Microsoft.Synchronization.Data.SyncStatistics syncStats = syncAgent.Synchronize(); } }
