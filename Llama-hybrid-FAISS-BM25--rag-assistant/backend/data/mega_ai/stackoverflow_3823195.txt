You need to subclass UIView to implement the method touchesBegan. @interface YourCustomView : UIView @implementation YourCustomView // Override this function to get the touch - (void)touchesBegan:(NSSet *)touches withEvent:(UIEvent *)event { NSLog(@"TOUCH!"); } And now set the view of your VC as a "YourCustomView" @interface YourViewController : UIViewController { YourCustomView* view }
