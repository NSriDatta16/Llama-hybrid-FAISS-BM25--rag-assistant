You can use $_SERVER variables to check if the source is from the IP you're limiting to. Here are some useful function snippets ($s is just $_SERVER): function gethostname($ip){ return gethostbyaddr($ip);} function gethostnamepretty($ip){ $host=gethostbyaddr($ip); $host=explode('.',$host); $max=count($host); return $host[$max - 2];} function getRawRealIP($s){ if (!empty($s['HTTP_CLIENT_IP'])) //check ip from share internet { $ip=$s['HTTP_CLIENT_IP']; } elseif (!empty($s['HTTP_X_FORWARDED_FOR'])) //to check ip is pass from proxy { $ip=$s['HTTP_X_FORWARDED_FOR']; } else { $ip=$s['REMOTE_ADDR']; } return $ip;} function checkIP($ip){ return long2ip(ip2long($ip));} function useragent($s){ return $s['HTTP_USER_AGENT'];}
