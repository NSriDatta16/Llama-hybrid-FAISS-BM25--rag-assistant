This should also work, taking into account, for example, page anchors (#) and at least some of those "weird characters" you mention but don't seem worried about: function remove_query_part($url, $term) { $query_str = parse_url($url, PHP_URL_QUERY); if ($frag = parse_url($url, PHP_URL_FRAGMENT)) { $frag = '#' . $frag; } parse_str($query_str, $query_arr); unset($query_arr[$term]); $new = '?' . http_build_query($query_arr) . $frag; return str_replace(strstr($url, '?'), $new, $url); } Demo: $string[] = 'http://domain.com/php/doc.php?arg1=0&arg2=1&arg3=0'; $string[] = 'http://domain.com/php/doc.php?arg1=0&arg2=1'; $string[] = 'http://domain.com/php/doc.php?arg1=0&arg2=1&arg3=0#frag'; $string[] = 'http://domain.com/php/doc.php?arg1=0&arg2=1&arg3=0&arg4=4'; $string[] = 'http://domain.com/php/doc.php'; $string[] = 'http://domain.com/php/doc.php#frag'; $string[] = 'http://example.com?arg1=question?mark&arg2=equal=sign&arg3=hello'; foreach ($string as $str) { echo remove_query_part($str, 'arg3') . "\n"; } Output: http://domain.com/php/doc.php?arg1=0&arg2=1 http://domain.com/php/doc.php?arg1=0&arg2=1 http://domain.com/php/doc.php?arg1=0&arg2=1#frag http://domain.com/php/doc.php?arg1=0&arg2=1&arg4=4 http://domain.com/php/doc.php http://domain.com/php/doc.php#frag http://example.com?arg1=question%3Fmark&arg2=equal%3Dsign Tested only as shown.
