I got the answer to your question. Put an UIImageView on your view through Interface Builder and make it an IBOutlet in .h and name it "img". and write this code in the .m - (void)touchesBegan:(NSSet *)touches withEvent:(UIEvent *)event { UITouch *touch = [[event allTouches] anyObject]; if (CGRectContainsPoint([img frame], [touch locationInView:self.view])) { img.alpha = 0.5; [UIView beginAnimations:@"frame" context:nil]; [UIView setAnimationDuration:1.0]; [UIView setAnimationDelegate:self]; [UIView setAnimationCurve:UIViewAnimationCurveEaseInOut]; img.alpha = 1.0; [UIView commitAnimations]; NSLog(@"Image Touched"); } }
