>>> import itertools >>> def sl(lst, i, j): il, iu = max(0, i-1), min(len(lst)-1, i+1) jl, ju = max(0, j-1), min(len(lst[0])-1, j+1) return (il, iu), (jl, ju) >>> lst = [[1, 2, 3], [4, 5, 6], [7, 8, 9]] >>> tup = 0, 2 >>> [lst[i][j] for i, j in itertools.product(*sl(lst, *tup)) if (i, j) != tup] [2, 5, 6] I don't know how elegant it seems to you, but it seems to work w/o any hard-coding.
