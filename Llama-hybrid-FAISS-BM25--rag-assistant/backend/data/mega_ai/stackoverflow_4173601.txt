Here's what i use var cookie = { "create": function(name, value, days) { if (typeof days !== 'number' || typeof name !== 'string' || typeof value !== 'string') { return false; } var date = new Date(); date.setTime(date.getTime() + (days*86400000)); document.cookie = name + '=' + value + '; expires=' + date.toGMTString() + '; path=/'; }, "read": function(name) { var cookie = document.cookie, i, val = false; cookie = cookie.split(';'); for (i = 0; i 0 && cookie[i].length > name.length) { cookie[i] = cookie[i].substr(1); } val = cookie[i].substr(name.length + 1); } } return val; }, "erase": function(name) { this.create(name, '', -1); } }; You can then use: cookie.create("userName", "Bill", 7); // store userName "Bill" for 7 days. cookie.read("userName"); // "Bill" cookie.erase("userName"); Here's a fiddle to see how it works. http://jsfiddle.net/robert/4vLT6/
