This is basically the same solution as the one Anurag Uniyal provided, only a little more compact: import re def parse_name(name): first_name, last_name = name.split(' ', 1) first_initial = re.search("^[A-Z.]+", first_name).group() if not first_initial.endswith("."): first_initial += "." return {"FirstName": first_name, "FirstInitial": first_initial, "LastName": last_name}
