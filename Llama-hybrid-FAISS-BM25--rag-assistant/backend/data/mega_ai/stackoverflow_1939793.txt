Something along these lines:- Dim cnn As New OleDbConnection("Provider=Microsoft.Jet.OLEDB.4.0;Data Source=MyDatabase.mdb;User Id=admin;Password=;") Dim cmd As New OleDbCommand Dim _ID as Integer=1 cmd.Connection = cnn cmd.CommandType = CommandType.Text cmd.CommandText = "DELETE FROM blah WHERE primarykey=" & _ID Using cnn cnn.open() cmd.ExecuteNonQuery() End Using
