pass in this in the inline click handler link or use event.target in the function (according to the W3C DOM Level 2 Event model ) function click123(event) { var a = event.target; } But of course, IE is different , so the vanilla JavaScript way of handling this is function doSomething(e) { var targ; if (!e) var e = window.event; if (e.target) targ = e.target; else if (e.srcElement) targ = e.srcElement; if (targ.nodeType == 3) // defeat Safari bug targ = targ.parentNode; } or less verbose function doSomething(e) { e = e || window.event; var targ = e.target || e.srcElement || e; if (targ.nodeType == 3) targ = targ.parentNode; // defeat Safari bug } where e is the event object that is passed to the function in browsers other than IE. If you're using jQuery though, I would strongly encourage unobtrusive JavaScript and use jQuery to bind event handlers to elements.
