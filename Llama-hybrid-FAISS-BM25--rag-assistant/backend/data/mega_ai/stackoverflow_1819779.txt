$("p > a").each(function(el) { var parent = $(this).parent(); var text = parent.text(); parent.replaceWith(" "+text+" "); }); What this does, is go through all your anchor tags inside paragraphs, get the only the text part and replace the paragraphs (the parent var) with a new anchor tag containing the same href attribute and the extracted text.
