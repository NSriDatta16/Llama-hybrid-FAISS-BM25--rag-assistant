Now my solution is: To use a converter to convert the image path into BitmapImage. in the converter, load the image using a FileStream and copy the data into a MemoryStream and finally close the FileStream. BitmapImage bmp = new BitmapImage(); bmp.CacheOption = BitmapCacheOption.OnLoad; bmp.BeginInit(); var fileStream = new FileStream(path, FileMode.Open, FileAccess.Read); var memStream = new MemoryStream(); memStream.SetLength(fileStream.Length); fileStream.Read(memStream.GetBuffer(), 0, (int)fileStream.Length); memStream.Flush(); fileStream.Close(); bmp.StreamSource = memStream; bmp.EndInit(); return bmp;
