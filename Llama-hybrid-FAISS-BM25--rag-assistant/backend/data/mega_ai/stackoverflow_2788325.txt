UPDATE organisaties o SET o.trefwoorden_flat = ( SELECT GROUP_CONCAT(t.naam SEPARATOR ',') FROM trefwoorden t INNER JOIN organisaties_has_trefwoorden AS o ON t.id_trefwoorden = o.id_trefwoorden WHERE o.id_organisaties = NEW.id_organisaties ) WHERE o.id_organisaties = NEW.id_organisaties
