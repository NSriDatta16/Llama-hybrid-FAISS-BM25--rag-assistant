SELECT wind_direction, AVG(windspeed) FROM weather WHERE time >= curdate() and time 0 GROUP BY wind_direction ORDER BY wind_direction ASC If you want a single wind_direction , say, that with the greatest average speed, use this: SELECT wind_direction, AVG(windspeed) AS avg_speed FROM weather WHERE time >= curdate() and time 0 GROUP BY wind_direction ORDER BY avg_speed DESC LIMIT 1
