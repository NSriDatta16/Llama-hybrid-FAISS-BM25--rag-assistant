See if you can find and delete them this way: NSHTTPCookieStorage * sharedCookieStorage = [NSHTTPCookieStorage sharedHTTPCookieStorage]; NSArray * cookies = [sharedCookieStorage cookies]; for (NSHTTPCookie * cookie in cookies){ NSLog(@"%@",cookie.domain); if ([cookie.domain rangeOfString:@"yourdomain"].location != NSNotFound){ NSLog(@"deleting"); [sharedCookieStorage deleteCookie:cookie]; } }
