scrollBar.addSelectionListener(new SelectionListener() { public void widgetDefaultSelected(SelectionEvent e) { } public void widgetSelected(SelectionEvent e) { if (e.detail == SWT.NONE) { // end of drag System.out.println("Drag end"); } else if (e.detail == SWT.DRAG) { // drag System.out.println("Currently dragging"); } } }); Hope this will help you... But I can see a problem with mousewheel use that throws multiple drag end events...
