As somebody mentioned, you probably need to convert the title string to utf-8 using iconv : require 'rubygems' require 'mechanize' require 'iconv' agent = Mechanize.new # If source encoding is indeed windows-1251, 'from' is CP1251 title = Iconv.conv('UTF8', 'CP1251', title) agent.get(url).save_as("#{mp3_dir}/#{title}.mp3")
