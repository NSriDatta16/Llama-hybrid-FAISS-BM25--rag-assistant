Something like this? class Content( object ): def __init__( self, content, maps_to ): self.content= content.split() self.maps_to = maps_to def matches( self, words ): return all( c in words for c in self.content ) def __str__( self ): return "%s -> %r" % ( " ".join(self.content), self.maps_to ) rules = [ Content('cats',("tim", "1 infinite loop")), Content('three',("sally", "123 fake st")), Content('three',("tim", "1 infinite loop")), Content('three cats',("john", "123 fake st")), Content('four cats',("albert", "345 real road")), Content('dogs',("tim", "1 infinite loop")), Content('cats hat', ("janet", None)), ] blah = "There are three cats in the hat" for r in rules: if r.matches(blah.split()): print r Output cats -> ('tim', '1 infinite loop') three -> ('sally', '123 fake st') three -> ('tim', '1 infinite loop') three cats -> ('john', '123 fake st') cats hat -> ('janet', None)
