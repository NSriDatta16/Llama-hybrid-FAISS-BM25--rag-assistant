List claims = ( from d in query.Descendants(xmlns + "claim") let t = d.Attribute("type").Value select new Claim { id = d.Attribute("id").value, type = t == "Subject Section" ? typesectionEnum.subjectSection : (t == "Vehicle Section" ? typesectionEnum.vehicleSection : (t == "Possible Related Section" ? typesectionEnum.possibleRelatedSection : typesectionenum.AgentSection)) } ).ToList ();
