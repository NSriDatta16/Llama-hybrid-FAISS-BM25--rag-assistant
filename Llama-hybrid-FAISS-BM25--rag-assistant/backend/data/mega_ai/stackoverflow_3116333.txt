You can include MKMapView both by code or by Interface builder. For Interface builder just drag it & drop it to your xib.(Tools->Library->MapView) By code In your .h file MKMapView * mapView; In your .m file -(void)viewWillAppear:(BOOL)animated { self.mapView = [[[MKMapView alloc] initWithFrame:self.view.frame] autorelease]; [self.view addSubview:self.mapView]; }
