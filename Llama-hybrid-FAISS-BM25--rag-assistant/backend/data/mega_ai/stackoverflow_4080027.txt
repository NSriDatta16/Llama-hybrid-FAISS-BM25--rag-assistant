I solved this very problem by putting some html(divs and anchors) over the telephone fields. The below code puts a phone icon on the end of the textbox for the given field. crmForm.ApplyClickToDial= function(field, href){ var phoneField = field; phoneField.style.position = "relative"; var imgAnchor = document.createElement("a"); phoneField.appendChild(imgAnchor); imgAnchor.href=href; imgAnchor.style.position = "absolute"; imgAnchor.style.right = ".5em"; imgAnchor.style.top=".5em"; var image = document.createElement("img"); image.src ="/_imgs/ico_16_4210.gif"; imgAnchor.appendChild(image); } var mobileNumber = crmForm.all.mobilephone.DataValue; crmForm.ApplyClickToDial(crmForm.all.mobilephone_d, "http://callphone/" + mobileNumber ); /* the "_d" represents the control's encompassing td element*/
