I am in a hurry now so this might not be the most simple or elegant way, but it is possible: #include using namespace std; class Container {}; template class TreeNode { Container mChildren; }; class TreeNodeWrapper; typedef TreeNode > recursiveTreeNode; class TreeNodeWrapper : public recursiveTreeNode { }; recursiveTreeNode myTreeNode;
