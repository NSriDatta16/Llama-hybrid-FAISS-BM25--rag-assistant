So we came up with a solution to do the following on the tilesloaded event of Map : var styleZoomControl = function () { var imgs = mapCanvas.find('img[src$="mapfiles/mapcontrols3d6.png"]'); if (imgs.length > 0 || mapCanvas.find('img[src$="mapfiles/szc3d.png"]').length > 0) { if (imgs.length > 0) imgs.attr('src', '/design/consumer/images/zoom-control.png'); } else { setTimeout(styleZoomControl, 200); } }
