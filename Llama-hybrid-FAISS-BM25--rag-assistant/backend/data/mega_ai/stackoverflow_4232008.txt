I believe your code is correct, and would translate in MSVC by replacing __thread by __declspec(thread) (see Thread Local Storage on MSDN) : template class my_class { struct some_type { }; static __declspec(thread) some_type * ptr; }; template __declspec(thread) typename my_class ::some_type * my_class ::ptr = 0;
