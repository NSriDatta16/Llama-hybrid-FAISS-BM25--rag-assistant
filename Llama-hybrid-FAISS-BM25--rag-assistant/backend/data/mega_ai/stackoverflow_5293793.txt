Use Mechanize#auth require 'rubygems' require 'mechanize' require 'logger' agent = WWW::Mechanize.new { |a| a.log = Logger.new("mech.log") } agent.user_agent_alias = 'Windows Mozilla' agent.auth('username', 'password') agent.get('http://example.com') do |page| puts page.title end
