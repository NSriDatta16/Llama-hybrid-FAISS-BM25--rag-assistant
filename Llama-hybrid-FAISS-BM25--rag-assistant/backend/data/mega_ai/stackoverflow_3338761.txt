On iPhone you make a subclass of the UIView and implement the funktion -(void)touchesBegan:(NSSet *)touches withEvent:(NSEvent *)anEvent: - (void)touchesBegan:(NSSet *)touches withEvent:(UIEvent *)event { UITouch *myTouch = [[touches allObjects] objectAtIndex: 0]; CGPoint currentPos = [myTouch locationInView: self]; NSLog(@"Point in myView: (%f,%f)", currentPos.x, currentPos.y); }
