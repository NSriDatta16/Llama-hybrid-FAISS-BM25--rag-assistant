Elegant? Why not Linq it. public static IEnumerable > SubSetsOf (IEnumerable source) { if (!source.Any()) return Enumerable.Repeat(Enumerable.Empty (), 1); var element = source.Take(1); var haveNots = SubSetsOf(source.Skip(1)); var haves = haveNots.Select(set => element.Concat(set)); return haves.Concat(haveNots); }
