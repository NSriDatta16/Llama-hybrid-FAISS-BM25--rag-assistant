I found a solution for SQLServer, but looks big and much less elegant than Quassnoi's WITH SortedList (Id, ParentId, SomeData, Level) AS ( SELECT Id, ParentId, SomeData, 0 as Level FROM LinkedList WHERE ParentId IS NULL UNION ALL SELECT ll.Id, ll.ParentId, ll.SomeData, Level+1 as Level FROM LinkedList ll INNER JOIN SortedList as s ON ll.ParentId = s.Id ) SELECT Id, ParentId, SomeData FROM SortedList ORDER BY Level
