Here's how to use the .tapCount property inside touchesBegan: -(void)touchesBegan:(NSSet *)touches withEvent:(UIEvent *)event; { NSUInteger numTaps = [[touches anyObject] tapCount]; UITouch *touch = [[event allTouches] anyObject]; if ([touch view] == yourThing) { NSLog(@"%i taps", numTaps); } }
