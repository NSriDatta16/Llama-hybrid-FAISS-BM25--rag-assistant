per line Public Sub ParseDoc() Dim doc As Document Set doc = ActiveDocument Dim paras As Paragraphs Set paras = doc.Paragraphs Dim para As Paragraph Dim sents As Sentences Dim sent As Range For Each para In paras Set sents = para.Range.Sentences For Each sent In sents Debug.Print sent.Text Next Next End Sub
