You need to create a custom UIView. The best way I have found to do this is in Interface Builder. .h @interface MyViewController : UITableViewController { IBOutlet UIView *headerView; } @property (nonatomic, retain) UIView *headerView; @end .m #import “MyViewController.h” @implementation MyViewController @synthesize headerView; #pragma mark Table Stuff - (UIView *)tableView:(UITableView *)tableView viewForHeaderInSection:(NSInteger)section { /* create a UIView here */ return headerView; } #pragma mark Memory Management - (void)dealloc { [headerView release]; [super dealloc]; }
