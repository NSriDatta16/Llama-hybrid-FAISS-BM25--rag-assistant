The code you've included is valid. Is the drag initiator configured correctly? Is the drop target configured to accept the drag-drop? Here's some code that will add a panel to canvas1 when canvas2 is dragged into canvas1: protected function canvas2_dragStartHandler(event:MouseEvent):void { var dragInitiator:Canvas=Canvas(event.currentTarget); var ds:DragSource = new DragSource(); DragManager.doDrag(dragInitiator, ds, event); } protected function canvas1_dragEnterHandler(event:DragEvent):void { DragManager.acceptDragDrop(Canvas(event.currentTarget)); } protected function canvas1_dragDropHandler(event:DragEvent):void { var userPanel:Panel = new Panel(); userPanel.width = 100; userPanel.height = 100; userPanel.x = 10; userPanel.y = 10; userPanel.visible = true; Canvas(event.currentTarget).addChild(userPanel); }
