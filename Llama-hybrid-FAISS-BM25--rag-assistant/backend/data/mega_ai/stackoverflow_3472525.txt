This is not as elegant as katrielalex's recursion, but it illustrates a basic class. class guesser: def __init__(self, l_bound, u_bound): self.u_bound = u_bound self.l_bound = l_bound self.nextguess() def nextguess(self): self.guess = int((self.u_bound + self.l_bound)/2) print 'Higher or lower than %i?' % self.guess def mynumberishigher(self): self.l_bound = self.guess self.nextguess() def mynumberislower(self): self.u_bound = self.guess self.nextguess()
