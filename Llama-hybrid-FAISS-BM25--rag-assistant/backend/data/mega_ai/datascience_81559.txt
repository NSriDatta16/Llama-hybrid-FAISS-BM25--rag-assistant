When you want to do sequential transformations, you should use Pipeline . imp_std = Pipeline( steps=[ ('impute', SimpleImputer(strategy='median')), ('scale', StandardScaler()), ] ) ColumnTransformer( remainder='passthrough', transformers=[ ('imp_std', imp_std, ['feat_1', 'feat_2']), ('std', StandardScaler(), ['feat_3']), ] ) or imp = ColumnTransformer( remainder='passthrough', transformers=[ ('imp', SimpleImputer(strategy='median'), ['feat_1', 'feat_2']), ] ) Pipeline( steps=[ ('imp', imp), ('std', StandardScaler()), ] )
