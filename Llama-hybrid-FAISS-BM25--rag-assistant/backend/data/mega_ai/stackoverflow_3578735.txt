Instead of defining method using def you can try define_method method: PeopleOrganisation::ROLES.each do |role| has_many role.to_s.pluralize.to_sym, :through => :people_organisations, :source => :person, :conditions => "people_organisations.role = '#{role.to_s}'" do define_method(: {:role => role}) { self.concat object } end end end
