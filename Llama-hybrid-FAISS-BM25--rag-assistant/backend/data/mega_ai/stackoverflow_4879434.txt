I used this with additionally jQuery tmpl plugin. var confirmTemplate = jQuery.template(" ${text} "); function showDialog(options) { if (options && options.data && options.dialog) { var dialogOptions = jQuery.extend({}, { modal: true, resizable: false, draggable: false }, options.dialog); return jQuery.tmpl(confirmTemplate, options.data).dialog(dialogOptions); } } function hideDialog (item) { if (!item.jQuery) item = $(item); item.dialog("close").dialog("destroy").remove(); } usage: showDialog({ data: { title: "My Title", text: "my Text" } dialog: { myDialog: "options" } });
