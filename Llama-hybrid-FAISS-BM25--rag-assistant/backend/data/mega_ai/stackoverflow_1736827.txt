See Recursive Queries Using Common Table Expressions : with cteAnchor as ( select Name, id from DP_ORG_OrganizationUnit where ManagedBy = 0) , cteRecursive as ( select id, Name from cteAnchor union all select t.id, r.Name + '/' + t.Name from DP_ORG_OrganizationUnit t join cteRecursive r on t.ManagedBy = r.id) select * from cteRecursive;
