The "center" would be (I believe) the average of the highest & lowest latitute and the highest and lowest longitute. var hilat = list.Max(loc=>loc.Latitute); var lolat = list.Min(loc=>loc.Latitute); var hilng = list.Max(loc=>loc.Longitute); var lolng = list.Min(loc=>loc.Longitute); var center = new Location() { Latitute = (hilat-lolat)/2 + lolat, Logitute = (hilng-lolng)/2 + lolng};
