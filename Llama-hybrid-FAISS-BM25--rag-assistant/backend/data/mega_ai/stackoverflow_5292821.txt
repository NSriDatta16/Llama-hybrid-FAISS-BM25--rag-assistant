You would need to do this manually, by adding a a static map in the class that maps Integers to enums, such as private static final Map intToTypeMap = new HashMap (); static { for (PcapLinkType type : PcapLinkType.values()) { intToTypeMap.put(type.value, type); } } public static PcapLinkType fromInt(int i) { PcapLinkType type = intToTypeMap.get(Integer.valueOf(i)); if (type == null) return PcapLinkType.DLT_UNKNOWN; return type; }
