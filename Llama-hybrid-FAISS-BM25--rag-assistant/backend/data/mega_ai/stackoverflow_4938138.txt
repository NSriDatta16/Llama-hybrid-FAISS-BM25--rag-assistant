As an alternative, file_get_contents and stream_context_create can also be used: $opts = array('http' => array( 'header' => 'User-agent: Mozilla/5.0 (iPhone; U; CPU like Mac OS X; en) AppleWebKit/420.1 (KHTML, like Gecko) Version/3.0 Mobile/3B48b Safari/419.3', ) ); $context = stream_context_create($opts); $result = file_get_contents($url, false, $context);
