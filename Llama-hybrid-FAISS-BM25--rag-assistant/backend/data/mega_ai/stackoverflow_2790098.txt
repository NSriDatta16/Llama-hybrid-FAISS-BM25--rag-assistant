i thought this solved my issue: drop: function(ev, ui) { var $unid = $(ui.draggable).attr('id'); recycleImage(ui.draggable,$unid); } if i use this: var $unid = ui.draggable.find('id'); then i get [object object] also, i had to modify: $('ul.gallery > li').click(function(ev) { var $item = $(this); var $unid = $(this).attr('id'); var $target = $(ev.target); if ($target.is('a.ui-icon-suitcase')) { deleteImage($item,$unid); } else if ($target.is('a.ui-icon-arrowreturnthick-1-w')) { recycleImage($item,$unid); } return false; }); works like a charm :)
