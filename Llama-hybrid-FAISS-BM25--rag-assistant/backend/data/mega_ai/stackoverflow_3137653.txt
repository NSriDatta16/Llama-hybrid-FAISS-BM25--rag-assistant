Hi I assume you would want to load the page completely before starting the timer to load the next URL, otherwise you would end up showing the next URL before the existing page even shows up (depending on your internet speed). Secondly you said you want to rotate the URLS. Below is the tested code for this: var urls = [ "http://www.google.com" ,"http://www.yahoo.com" ,"http://www.ajaxian.com" ,"http://www.ebay.com" ]; function showUrl(idx) { alert(idx + " Showing " + urls[idx]); var f = document.getElementById("f"); // call the next load after 5 seconds only after // this iframe loads f.onload = function() { var next = ++idx % urls.length; setTimeout(function(){ showUrl(next); }, 5000); } // set the src f.src = urls[idx]; } Adding a more compact version: (Works in IE6, FF, Opera, Chromium) window.onload = (function(urls, interval) { var idx = 0; var bAttached = false; return function showUrl() { var f = document.getElementById("f"); var onLoad = function() { // loading only after previous page loads idx = ++idx % urls.length; // rotation setTimeout(showUrl, interval); } if(! bAttached) { if(navigator.userAgent.indexOf("MSIE") !== -1) { f.attachEvent("onload", onLoad); bAttached = true; }else { f.onload = onLoad; } bAttached = true; } f.src = urls[idx]; }; })([ "http://www.google.com" ,"http://www.yahoo.com" , "http://www.sun.com" ,"http://www.ebay.com" ], 5000 );
