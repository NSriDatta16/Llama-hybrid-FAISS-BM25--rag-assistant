This is the storage format of a CGRect / NSRect . You can easily read it using CGRectFromString / NSRectFromString and then get the values like this: NSString *string = @"{{2, 2}, {103, 166}}"; CGRect rect = CGRectFromString(string); CGFloat x = rect.origin.x; // 2 CGFloat y = rect.origin.y; // 2 CGFloat width = rect.size.width; // 103 CGFloat height = rect.size.height; // 166
