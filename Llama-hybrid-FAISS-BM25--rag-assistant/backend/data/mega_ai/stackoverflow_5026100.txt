If you're trying to add to the file, you need to use the System.IO.FileMode.Append property. var multipleStorage = IsolatedStorageFile.GetUserStoreForApplication(); string multipleFile = "multipleFile.txt"; using (var file = multipleStorage.OpenFile(multipleFile, System.IO.FileMode.Append, System.IO.FileAccess.Write)) { using (var writer = new StreamWriter(file)) { writer.Write(nameTextBox.Text + ", " + dunsTextBox.Text + ", " + typeCheck + ", " + resellerCheck + System.Environment.NewLine); } }
