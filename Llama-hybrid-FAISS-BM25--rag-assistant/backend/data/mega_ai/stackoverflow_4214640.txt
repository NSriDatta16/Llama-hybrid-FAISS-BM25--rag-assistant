Define regexps: scala> val link = """ (.+) """.r link: scala.util.matching.Regex = (.+) scala> val paragraph = """ (.+) """.r paragraph: scala.util.matching.Regex = (.+) scala> val text = """ here we have a link example. """ text: java.lang.String = here we have a link example. Apply them to the input: scala> val modifiedText = paragraph.replaceAllIn(text, {matched => val paragraph(content) = matched; content}) modifiedText: String = here we have a link example. scala> link.replaceAllIn(modifiedText, {matched => val link(href, title) = matched; "\\\\url{%s}{%s}" format(href, title)}) res11: String = here we have a \url{http://www.scala-lang.org/api/current/index.html}{link} example.
