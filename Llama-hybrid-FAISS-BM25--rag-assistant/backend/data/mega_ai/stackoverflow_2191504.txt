Not so elegant solution- but it works so who cares: - (IBAction)changeTextProgrammaticaly{ myTextView.text = @"Some text"; [NSTimer scheduledTimerWithTimeInterval:0.1 target:self selector:@selector(rewindOffset) userInfo:nil repeats:NO]; } - (void)rewindOffset{ [myTextView setContentOffset:CGPointMake(0,0) animated: NO]; }
