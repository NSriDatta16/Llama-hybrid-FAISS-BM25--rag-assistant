Two versions in Common Lisp: (defun average (items) (destructuring-bind (l . s) (reduce (lambda (c a) (incf (car c)) (incf (cdr c) a) c) items :initial-value (cons 0 0)) (/ s l))) (defun average (items &aux (s 0) (l 0)) (dolist (i items (/ s l)) (incf s i) (incf l)))
