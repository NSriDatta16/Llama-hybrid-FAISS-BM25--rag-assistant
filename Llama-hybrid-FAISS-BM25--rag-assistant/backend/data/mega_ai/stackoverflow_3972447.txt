I think your mapping needs to be like this: public void Override(AutoMapping mapping) { mapping.Table("MasterCategories"); mapping.Id(x => x.Id).Column("masterCategoryId").GeneratedBy.Guid(); mapping.Map(x => x.Number).Column("categoryNumber"); mapping.References(x => x.Organization) .Column("organizationId") .Cascade.All(); }
