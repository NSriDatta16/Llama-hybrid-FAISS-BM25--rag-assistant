public double GetAvgResult() { // Assumes GetWeeklyValues() never returns null. return GetWeeklyValues().DefaultIfEmpty().Average(); } Do note that this returns a double , which I assume is what you really want (the average of a bunch of integers is logically not an integer). You can cast it to int if necessary, or if you want to stick with integer math all the way: var seq = GetWeeklyValues().DefaultIfEmpty(); return seq.Sum() / seq.Count();
