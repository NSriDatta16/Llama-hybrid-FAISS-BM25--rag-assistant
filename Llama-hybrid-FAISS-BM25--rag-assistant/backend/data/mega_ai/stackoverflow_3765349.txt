The only event that contains information about the character typed is keypress . Anything character-related you may infer from the keyCode property of keydown or keyup events is unreliable and dependent on a particular keyboard mapping. The following will prevent non-numeric keyboard input all major browsers by using the character obtained from the keypress event. It won't prevent the user from pasting or dragging non-numeric text in. var input = document.getElementById("your_input"); input.onkeypress = function(evt) { evt = evt || window.event; if (!evt.ctrlKey && !evt.metaKey && !evt.altKey) { var charCode = (typeof evt.which == "undefined") ? evt.keyCode : evt.which; if (charCode && !/\d/.test(String.fromCharCode(charCode))) { return false; } } };
