Not quite as pretty as the JQuery example by Marco but with prototype (i may be missing a more elegant solution) it would be: function sort_select(select) { var options = $A(select.options).sortBy(function(o) { return o.innerHTML }); select.innerHTML = ""; options.each(function(o) { select.insert(o); } ); } And then just pass it a select element: sort_select( $('category-select') );
