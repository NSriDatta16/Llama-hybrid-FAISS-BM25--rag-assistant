UITouch *touch = [[UITouch alloc] initInView:view]; UIEvent *eventDown = [[UIEvent alloc] initWithTouch:touch]; [touch.view touchesBegan:[eventDown allTouches] withEvent:eventDown]; [touch setPhase:UITouchPhaseEnded]; UIEvent *eventUp = [[UIEvent alloc] initWithTouch:touch]; [touch.view touchesEnded:[eventUp allTouches] withEvent:eventUp]; [eventDown release]; [eventUp release]; [touch release]; from here http://cocoawithlove.com/2008/10/synthesizing-touch-event-on-iphone.html
