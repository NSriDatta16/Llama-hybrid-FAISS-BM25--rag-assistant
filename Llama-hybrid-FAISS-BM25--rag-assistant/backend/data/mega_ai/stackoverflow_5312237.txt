Try to save the original position before starting to drag and restore it if drops fail. You can save the original position like this: var dragposition = ''; $('#divdrag').draggable({ // options... start: function(event,ui){ dragposition = ui.position; } }); $("#dropHere").droppable({ drop: function(){ $.ajax({ url: 'myurl.php', data: 'html', async: true, error: function(){ $('#divdrag').css({ 'left': dragposition.left, 'top': dragposition.top }); } }); } });
