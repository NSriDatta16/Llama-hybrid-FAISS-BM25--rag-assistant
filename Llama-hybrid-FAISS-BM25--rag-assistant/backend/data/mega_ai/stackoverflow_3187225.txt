It's not an elegant one-liner, but you could do it with perl. #!/usr/bin/perl open A, $ARGV[0]; while(split/,/, ) { $k{$_[0]} = [@_]; } close A; open B, $ARGV[1]; while(split/,/, ) { print join(',',@{$k{$_[0]}}) if defined($k{$_[0]}) && $k{$_[0]}->[2] == $_[1] && $k{$_[0]}->[5] == $_[2] && $k{$_[0]}->[6] == $_[3]; } close B;
