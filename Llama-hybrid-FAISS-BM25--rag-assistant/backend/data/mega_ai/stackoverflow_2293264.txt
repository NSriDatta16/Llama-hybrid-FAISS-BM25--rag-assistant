Edited to take account of main post edits Html.RouteLink( "Looga", new { Controller = "Cooga", Action = "Aooga" }, new { Fragment = "Fooga" }) This code will current produce Looga Not really what you want. Instead, you could write your own extension method for RouteLink, something like this... public static class RouteLinkExtensions { public static string RouteLink( this HtmlHelper htmlHelper, string linkText, object routeValues, string fragment) { // There's probably better ways to do the implementation, but you get the idea var url = new UrlHelper(htmlHelper.ViewContext.RequestContext); return string.Format(" {2} ", url.RouteUrl(routeValues), fragment, linkText); } } This will allow you to use a clean call to RouteLink in your page :-) Html.RouteLink( "Looga", new { Controller = "Cooga", Action = "Aooga" }, "Fooga")
