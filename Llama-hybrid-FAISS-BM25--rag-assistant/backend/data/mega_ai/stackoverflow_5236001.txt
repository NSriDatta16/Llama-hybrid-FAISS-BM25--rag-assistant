Here's some sample code for mechanize implementing cookies and a fake user-agent -- might help out... import mechanize cookies = mechanize.CookieJar() opener = mechanize.build_opener(mechanize.HTTPCookieProcessor(cookies)) opener.addheaders = [("User-agent", "Mozilla/5.0 (Macintosh; U; Intel Mac OS X 10_6; en-us) AppleWebKit/533.4 (KHTML, like Gecko) Version/4.1 Safari/533.4"),] mechanize.install_opener(opener) url = "http://your_home_router_address" content = mechanize.urlopen(url) return content.read()
