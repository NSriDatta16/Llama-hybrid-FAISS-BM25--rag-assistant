Turns out that this older type of animation doesn't seem to block new touchesBegan or touchesEnded: - (void)touchesBegan:(NSSet*)touches withEvent:(UIEvent*)event { [UIView beginAnimations:@"touchesBegan" context:nil]; self.backgroundColor = altColor; [UIView commitAnimations]; }
