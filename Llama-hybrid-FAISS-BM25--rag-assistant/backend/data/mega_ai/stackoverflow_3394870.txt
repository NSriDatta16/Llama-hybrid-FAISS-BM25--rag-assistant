Yes, it is possible (UITableView has a method allowing to what row specific point belongs to): -(void) touchBegan:(NSSet*) touches withEven:(UIEvent*)event { CGPoint pt=[[touches anyObject] locationInView:self]; CGPoint pointInTable = [[touches anyObject] locationInView:yourTableView]; NSIndexPath* path = [yourTableView indexPathForRowAtPoint:pointInTable]; }
