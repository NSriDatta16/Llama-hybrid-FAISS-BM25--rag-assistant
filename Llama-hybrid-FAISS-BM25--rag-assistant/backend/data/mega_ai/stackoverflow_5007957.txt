you make ALL images draggable and resizable in "assignProperty" function. You just need to do this for new element: function assignProperty(property, instanceID) { switch(property) { case "draggable": $("#Image_"+instanceID).draggable({ grid: [ 16, 16 ], snap: false, containment: "#mainCanvas", scroll: false }); break; case "resizableImage": $("#Image_"+instanceID).resizable({ grid: [ 16, 16 ], minWidth: 32, maxWidth: 208, minHeight: 32, maxHeight: 208, aspectRatio: 1/1 }); break; } }
