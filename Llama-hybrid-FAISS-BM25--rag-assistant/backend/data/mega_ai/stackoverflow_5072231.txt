This should do it: function cbfunc(data) { var s = ''; $.each(data.query.results.result, function(i, item) { s += ' ' + item.title + ' '; }); $('#searchResults').html(s); } $(function() { $('#search').live('submit', function(e) { e.preventDefault(); $.ajax({ url: 'http://query.yahooapis.com/v1/public/yql?q=' + encodeURIComponent('SELECT title FROM search.web WHERE query="' + this.searchBar.value + '"') + '&format=json&diagnostics=true&callback=cbfunc', dataType: 'jsonp', jsonp: 'callback', jsonpCallback: 'cbfunc' }); }); }); Live demo: http://jsfiddle.net/JAS4H/59/
