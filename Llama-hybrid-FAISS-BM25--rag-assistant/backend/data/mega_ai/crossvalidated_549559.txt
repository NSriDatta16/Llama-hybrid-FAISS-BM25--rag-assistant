To simulate from the end of the series, pass the argument anchor='end' to the simulate method. For example: import pandas as pd import statsmodels.api as sm import matplotlib.pyplot as plt # Load data y = sm.datasets.macrodata.load_pandas().data['cpi'] y.index = pd.period_range('1959Q1', '2009Q3', freq='Q') # Create and fit the model mod = sm.tsa.SARIMAX(y, order=(1, 1, 0), trend='c') res = mod.fit() # Simulate data starting at the end of the time series sim = res.simulate(100, anchor='end', repetitions=100) # Plot the data and simulations fig, ax = plt.subplots(figsize=(15, 4)) y.plot(color='k', ax=ax) sim.plot(ax=ax, color='C0', alpha=0.2, legend=False); Gives:
