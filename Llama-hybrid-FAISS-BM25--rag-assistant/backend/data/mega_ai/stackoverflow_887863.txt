That would be something like this: Dim bitmap As New BitmapImage() Dim buffer As Byte() = GetHugeByteArray() Using stream As New MemoryStream(buffer, False) bitmap.BeginInit() bitmap.CacheOption = BitmapCacheOption.OnLoad bitmap.StreamSource = stream bitmap.EndInit() bitmap.Freeze() End Using
