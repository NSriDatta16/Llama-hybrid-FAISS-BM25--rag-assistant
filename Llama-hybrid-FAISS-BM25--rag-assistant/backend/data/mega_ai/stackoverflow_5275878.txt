- (void)setupCaptureSession { NSError *error = nil; [self setCaptureSession: [[AVCaptureSession alloc] init]]; ... That leaks the capture-session, which will keep all inputs and outputs and all their little internal helpers alive. Two options: AVCaptureSession *session = [[AVCaptureSession alloc] init]; self.captureSession = session; [session release], session = nil; // or: self.captureSession = [[[AVCaptureSession alloc] init] autorelease];
