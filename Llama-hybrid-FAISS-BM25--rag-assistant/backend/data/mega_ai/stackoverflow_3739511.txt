This is the best I could come up with quickly float calcWeightedAverage(float v1,float v2,float w1,float w2) { float a1 = 0.0; float a2 = 0.0; if (v1 != 0) { a1 = v1/(v1+v2) * w1; } if (v2 != 0) { a2 = v2/(v1+v2) * w2; } return a1 + a2; }
