google.maps.event.addListener(map, 'click', function(event) { addMarker(event.latLng); }); function addMarker(location) { marker = new google.maps.Marker({ position: location, map: map, draggable: true, visible: true }); google.maps.event.addListener(marker, 'click', function(event) { this.setMap(null); }); }
