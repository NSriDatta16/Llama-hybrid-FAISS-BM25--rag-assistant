Are you looking for an average rating per user id? If so, then you need to use both GroupBy and Average . var rates = ctx.Rates .Where( r => r.Id == Id ) .GroupBy( g => g.UserId, r => r.Rating ) .Select( g => new { UserId = g.Key, Rating = g.Average() });
