I think this is what you wanted: http://jsfiddle.net/maniator/brCX5/33/ $(function() { $(".draggable").draggable({ helper: 'clone', snap: '.ui-droppable', snapMode: 'outer' }); $(".droppable").droppable({ accept: '.draggable', drop: function(event, ui) { var clone = $(ui.draggable).clone(); $(this).append(clone); clone.removeClass('draggable') clone.draggable({ snap: '.ui-droppable', snapMode: 'outer' }); } }); }); Works like a charm :-), your welcome
