I think this is the closest to an elegant solution: public MyWebBrowser() { InitializeComponent(); this.DocumentCompleted += new WebBrowserDocumentCompletedEventHandler(SuperWebBrowser_DocumentCompleted); this.DocumentCompleted += new WebBrowserDocumentCompletedEventHandler(SuperWebBrowser_DocumentFirstCompleted); } private void SuperWebBrowser_DocumentFirstCompleted(object sender, WebBrowserDocumentCompletedEventArgs e) { this.DocumentCompleted -= new WebBrowserDocumentCompletedEventHandler(SuperWebBrowser_DocumentFirstCompleted); this.Document.MouseDown += new HtmlElementEventHandler(Document_MouseDown); } private void SuperWebBrowser_DocumentCompleted(object sender, WebBrowserDocumentCompletedEventArgs e) { } That way I don't need to check a variable every time a document completes. Guess that's kinda what Microsoft had in mind.
