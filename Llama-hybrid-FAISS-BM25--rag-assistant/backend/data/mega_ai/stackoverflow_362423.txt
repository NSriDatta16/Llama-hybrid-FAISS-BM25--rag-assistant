I had same issue public class FriViewPage : ViewPage { public override string MasterPageFile { get { return "~/Views/Shared/Site.Master"; // base.MasterPageFile; } set { if (ViewData["agent"].ToString() == "steve") base.MasterPageFile = "~/Views/Shared/Site.Master"; else base.MasterPageFile = "~/Views/Shared/Site2.Master"; } } } Then just ensure all the views inherit from FriViewPage instead of ViewPage
