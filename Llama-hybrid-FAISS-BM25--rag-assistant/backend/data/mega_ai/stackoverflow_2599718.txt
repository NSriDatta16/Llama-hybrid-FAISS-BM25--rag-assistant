Here is how I would do it. Use 2 images, one colour and one grayscale. Use a clip on the grayscale. Below is a sample code var color:ImageView = ImageView { image: Image { url: "{__DIR__}color.jpg" } } var x:Number = 100; var y:Number = 100; var grayscale:ImageView = ImageView { image: Image { url: "{__DIR__}grayscale.jpg" } clip: Circle { centerX: bind x centerY: bind y radius: 40 } onMouseDragged: function (e: MouseEvent): Void { x = e.sceneX; y = e.sceneY; } } Stage { title: "Application title" scene: Scene { width: 500 height: 500 content: [ color, grayscale ] } }
