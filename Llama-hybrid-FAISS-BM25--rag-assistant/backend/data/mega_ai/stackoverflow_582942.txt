I used LWP::Parallel::UserAgent for something similar. An example from the POD: require LWP::Parallel::UserAgent; $ua = LWP::Parallel::UserAgent->new(); ... $ua->redirect (0); # prevents automatic following of redirects $ua->max_hosts(5); # sets maximum number of locations accessed in parallel $ua->max_req (5); # sets maximum number of parallel requests per host ... $ua->register ($request); # or $ua->register ($request, '/tmp/sss'); # or $ua->register ($request, \&callback, 4096); ... $ua->wait ( $timeout ); ... sub callback { my($data, $response, $protocol) = @_; .... }
