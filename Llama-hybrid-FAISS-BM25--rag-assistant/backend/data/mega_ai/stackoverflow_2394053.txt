I used the jquery cycle option 'after' to call the sIFR function once each transition is completed: $(document).ready(function(){ $('#div').cycle({ fx: 'fade', after: function(){ sIFR.replace(alberta, { selector: '.flash-text', wmode: 'transparent' }); } }); });
