If the MouseDragElementBehavior doesn't work the way you need it to, you could always descend from the class and customize it to your needs. For example: public class DragBehvaior : MouseDragElementBehavior { public DragBehvaior() { this.DragBegun += new MouseEventHandler(DragBehvaior_DragBegun); } void DragBehvaior_DragBegun(object sender, MouseEventArgs e) { var element = this.AssociatedObject as UIElement; // Calculate the correct ZIndex here. Canvas.SetZIndex(element, 100); } }
