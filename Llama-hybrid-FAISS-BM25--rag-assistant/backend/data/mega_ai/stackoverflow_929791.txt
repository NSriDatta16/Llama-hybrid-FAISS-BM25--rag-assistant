This is how Prototype does it: Object.extend = function(destination, source) { for (var property in source) { if (source.hasOwnProperty(property)) { destination[property] = source[property]; } } return destination; }; called as, for example: var arr1 = { robert: "bobby", john: "jack" }; var arr2 = { elizabeth: "liz", jennifer: "jen" }; var shortnames = Object.extend(arr1,arr2); EDIT : added hasOwnProperty() check as correctly pointed out by bucabay in comments
