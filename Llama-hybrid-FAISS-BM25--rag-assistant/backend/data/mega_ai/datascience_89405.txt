Scikit-learn has ColumnTransformer for heterogeneous data . Here is a rough code snippet to get you started: from sklearn.compose import ColumnTransformer from sklearn.feature_extraction.text import TfidfVectorizer from sklearn.naive_bayes import MultinomialNB from sklearn.pipeline import Pipeline from sklearn.preprocessing import OneHotEncoder pipeline = Pipeline([('union', ColumnTransformer([('tfidf', TfidfVectorizer(), 'text-column'), ('onehot', OneHotEncoder(), ['gender-column'])])), ('clf', MultinomialNB())]) pipeline.fit(X_train, y_train)
