Use this as a global action filter. public class NoCacheForAuthenticatedUsersAttribute: ActionFilterAttribute { public override void OnResultExecuted(ResultExecutedContext filterContext) { if(filterContext.HttpContext.User.Identity.IsAuthenticated) { filterContext.HttpContext.Response.Cache.SetCacheability(HttpCacheability.NoCache); } } }
