Brief example which may help... lua_State* L=lua_open(); // create a Lua state luaL_openlibs(L); // load standard libs lua_pushstring(L, "nick"); // push a string on the stack lua_setglobal(L, "name"); // set the string to the global 'name' luaL_loadstring(L, "print(name)"); // load a script lua_pcall(L, 0, 0, 0); // call the script
