I'm not really an expert in jparsec, but the following alternative definition of NUMBER appears to work: import org.codehaus.jparsec.Parser; import org.codehaus.jparsec.Terminals; import org.codehaus.jparsec.Tokens.Fragment; import org.codehaus.jparsec.functors.Map; public class Test { static final Parser NUMBER = Terminals.DecimalLiteral.TOKENIZER.map( new Map () { public Double map(Fragment arg0) { return Double.valueOf(arg0.text()); } }); public static void main(String[] args) { System.out.println(NUMBER.parse("234234234")); } }
