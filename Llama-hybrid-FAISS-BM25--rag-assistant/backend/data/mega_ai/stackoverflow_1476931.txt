You may use 'enum' or 'define' to declare the size: #define XX 1024 static int const YY = 1024; enum{ ZZ = 1024 }; extern int main(void){ static char x[XX]; // no error *(int*)&XX = 123; // error: lvalue required as unary ‘&’ operand static char y[YY]; // error: storage size of ‘y’ isn’t constant *(int*)&YY = 123; // no error, the value of a const may change static char z[ZZ]; // no error *(int*)&ZZ = 123; // error: lvalue required as unary ‘&’ operand }
