This code will create a new paragraph style with 1/2 inch tab intervals and set it as both the default and current paragraph style on the variable textView. NSMutableParagraphStyle *style = [[NSMutableParagraphStyle alloc] init]; [style setDefaultTabInterval:36.]; [style setTabStops:[NSArray array]]; [textView setDefaultParagraphStyle:style]; [textView setTypingAttributes:[NSDictionary dictionaryWithObject:style forKey:NSParagraphStyleAttributeName]]; [style release];
