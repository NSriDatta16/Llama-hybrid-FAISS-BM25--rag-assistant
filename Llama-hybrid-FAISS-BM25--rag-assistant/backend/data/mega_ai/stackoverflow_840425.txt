There's a pure C# way to do this without native calls. Taken from here . //codesnippet:06EE3DE0-D469-44DD-A15F-D8AF629E4E03 public string GetStorageCardFolder() { string storageCardFolder = string.Empty; foreach (string directory in Directory.GetDirectories("\\")) { DirectoryInfo dirInfo = new DirectoryInfo(directory); //Storage cards have temporary attributes do a bitwise check. //http://forums.microsoft.com/MSDN/ShowPost.aspx?PostID=612136&SiteID=1 if ((dirInfo.Attributes & FileAttributes.Temporary) == FileAttributes.Temporary) storageCardFolder = directory; } return storageCardFolder; }
