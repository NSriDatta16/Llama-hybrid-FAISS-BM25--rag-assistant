import BeautifulSoup doc = ''' Page title This is paragraph one . This is paragraph two . ''' soup = BeautifulSoup.BeautifulSoup(doc) for tag in soup.recursiveChildGenerator(): if isinstance(tag,BeautifulSoup.Tag) and tag.name in ('a','b','i'): print(tag) yields paragraph one paragraph two If you just want the text contents, you could change print(tag) to print(tag.string) . If you want to remove an attribute like onclick="" from the a tag, you could do this: if isinstance(tag,BeautifulSoup.Tag) and tag.name in ('a','b','i'): if tag.name=='a': del tag['onclick'] print(tag)
