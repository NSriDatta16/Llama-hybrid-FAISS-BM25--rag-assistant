You'll need to parse it as a date first, try this: var dateFromLocalStorage = function(s) { var m = (""+s).match(/^(\d+),(\d+),(\d+)$/); return (m) ? new Date(m[1], m[2], m[3]) : null; }; dateFromLocalStorage("2011,3,20"); // => Wed Apr 20, 2011 ... Note that the date constructor takes the month as a zero based integer, so Jan=0, Feb=1, Mar=2, etc.
