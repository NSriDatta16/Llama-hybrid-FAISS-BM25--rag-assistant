Use a Common Table Expression to calculate the value. WITH AddressWithCalc AS ( SELECT AddressID, Street, City, State, Zip, ((@ZIP - 1000) * 10) AS ZipCalculatedValue FROM dbo.tbl_Addresses ) INSERT INTO #Listings (AddressID, Street, City, State, Zip, ZipCalculatedValue) SELECT * FROM AddressWithCalc WHERE ZipCalculatedValue
