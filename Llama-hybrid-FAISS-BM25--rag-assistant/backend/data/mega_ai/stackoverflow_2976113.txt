Perhaps what you are looking for is: int (*var1)[10][10] = malloc(sizeof *var1); // 400 bytes i(10, 10, *var1); printf("var1[0][0]=%i\n", (*var1)[0][0]); Added: A complete code fragment for gcc might be: #include #include void i(int n, int m, int var1[][m]) { // n not needed var1[0][0] = 5; var1[1][2] = 6; } int main(void) { int n = 10, m = 10; int (*var1)[n][m] = malloc(sizeof *var1); // 400 bytes i(n, m, *var1); printf("var1[0][0]=%i\n",(*var1)[0][0]); printf("var1[1][2]=%i\n",(*var1)[1][2]); return 0; } For msvc , you'll need to make n and m constants , as in: enum {n = 10, m = 10}; void i(int var1[][m]) { var1[0][0] = 5; var1[1][2] = 6; } int main(void) { int (*var1)[n][m] = malloc(sizeof *var1); i(*var1); // ... }
