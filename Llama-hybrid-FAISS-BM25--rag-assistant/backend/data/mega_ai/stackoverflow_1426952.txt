Here's my take - it meets all the requirements in your question. You'll note that I've chosen to reject illegal assignments with an IllegalArgumentException . package com.pragmaticsoftwaredevelopment.stackoverflow; public class FixedLengthString { private String string=null; private int length=-1; public FixedLengthString(int length) { this.length = length; } public void setValue (String str) { if (str.length() > length) throw new IllegalArgumentException("ERROR: Assigned string length (" + str.length() + ") exceeded given limit (" + length + ")"); else this.string = str; } public String toString() { return String.format("%1$-" + length + "s", string); } public String getString() { return toString(); } }
