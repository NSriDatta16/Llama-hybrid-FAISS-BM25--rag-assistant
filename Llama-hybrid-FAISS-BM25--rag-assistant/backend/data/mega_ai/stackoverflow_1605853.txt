If executing a task periodically is the problem here (I'm not quite sure) I suggest using ScheduledExecutorService, which is part of JDK 1.5 and later: import java.util.concurrent.Executors; import java.util.concurrent.ScheduledExecutorService; import java.util.concurrent.TimeUnit; public class SchedulerExample { private ScheduledExecutorService scheduler = Executors.newScheduledThreadPool(1); public void activateHourlyDataRetrieval() { Runnable dataRetriever = new Runnable() { MngtImpl pers = new MngtImpl(); public void run() { pers.findItemByPIdEndDate(); } }; scheduler.scheduleAtFixedRate(dataRetriever, 0, 1, TimeUnit.HOURS); } }
