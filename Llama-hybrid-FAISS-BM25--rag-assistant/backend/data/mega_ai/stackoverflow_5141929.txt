Why don't you try to read file using a simple StreamReader ? Below a part of a method I have created to readfile from store. Have a try, check your content, and then try loading xml from String (XDocument.Parse etc ...) String fileContent = String.Empty; using (_store = IsolatedStorageFile.GetUserStoreForApplication()) { if (_store.FileExists(file)) { _storeStream = new IsolatedStorageFileStream(file, FileMode.Open, _store); using (StreamReader sr = new StreamReader(_storeStream)) { fileContent = sr.ReadToEnd(); } __storeStream.Close(); return fileContent; } else { return null; } }
