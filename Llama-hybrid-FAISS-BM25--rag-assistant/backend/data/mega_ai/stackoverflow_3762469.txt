You can do this in code: // Create a paragraph with two coloured runs Paragraph para = new Paragraph(); Run run1 = new Run("Red "); run1.Foreground = Brushes.Red; Run run2 = new Run("Green"); run2.Foreground = Brushes.Green; para.Inlines.Add(run1); para.Inlines.Add(run2); // Get the document FlowDocument doc = richTextBox1.Document; // Clear existing content doc.Blocks.Clear(); // Add new content doc.Blocks.Add(para); Or in XAML: Red Green
