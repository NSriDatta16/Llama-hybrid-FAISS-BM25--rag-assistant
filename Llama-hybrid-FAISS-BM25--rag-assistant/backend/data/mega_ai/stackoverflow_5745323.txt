Try below updated code of yours - (IBAction)digitPressed:(UIButton *)sender { NSString *digit = [[sender titleLabel] text]; if (userIsInTheMiddleofTypingANumber) { // if the user enters a decimal point, check if the existing display text contains a "." if not, then allow it, but if it does, do not allow it. NSString *decimal = @"."; NSRange range = [digit rangeOfString:decimal]; if (range.location == NSNotFound) { [display setText:[[display text] stringByAppendingString:digit]]; [display setText:digit]; } } else { [display setText:digit]; userIsInTheMiddleofTypingANumber = YES; } }
