You can use StartCoroutine("DelayDeath"); where DelayDeath is the name of a method. See below. if(other.tag == "enemy"){ other.transform.parent.gameObject.name = ("enemy" + currentEnemy); print(other.name); StartCoroutine("DelayDeath"); BloodParticle.emit = true; Stabbed = true; Character.GetComponent("MouseLook").enabled = false; Character.GetComponent("CharacterMotor").enabled = false; } private IEnumerator DelayDeath() { //this will return...and one second later re-enter and finish the method yield return new WaitForSeconds(1.0f); gameObject.Find("enemy" + currentEnemy).GetComponent("RagdollOrNot").MakeKinematicFalse(); }
