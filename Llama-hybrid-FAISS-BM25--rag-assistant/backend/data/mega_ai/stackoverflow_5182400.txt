This is the easiest way to me... /* Load Image */ var myLoader:Loader = new Loader(); imageContainer.addChild(myLoader); var url:URLRequest = new URLRequest("photo.jpg"); myLoader.load(url); /* Drag and Drop */ imageContainer.addEventListener(MouseEvent.MOUSE_DOWN, pickUp); function pickUp(event:MouseEvent):void { imageContainer.startDrag(); } stage.addEventListener(MouseEvent.MOUSE_UP, dopIt); function dopIt(event:MouseEvent):void { imageContainer.stopDrag(); }
