Please Try this Code it is working both version, you need to add the audio control dynamically in page. it is working. var isiPad = false; if (navigator.userAgent.match(/iPad/i) || navigator.userAgent.match(/iPhone/i) || navigator.userAgent.match(/Android/i) || navigator.userAgent.match(/webOS/i)) { isiPad = true; } $(function() { if (isiPad) { var audioElement2 = document.createElement('audio'); audioElement2.setAttribute('src', 'img/bgsound.mp3'); audioElement2.controls = true; audioElement2.loop = true; audioElement2.load(); audioElement2.play(); } });
