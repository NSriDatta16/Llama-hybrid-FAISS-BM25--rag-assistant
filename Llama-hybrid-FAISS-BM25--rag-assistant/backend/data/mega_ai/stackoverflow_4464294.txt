This is my solution: var foo = true; $( ".draggable" ).draggable({ revert: function(){if(foo == true){return true;}else{foo = true;}}, }); $("#droppable").droppable({ activeClass: "ui-state-hover", hoverClass: "ui-state-active", drop: function( event, ui ) { if($this == $that){ foo = true; alert("this will revert the draggable back to original position"); }else{ foo = false; alert("this will NOT revert the draggable back to original position"); } } });
