"Common properties can be kept in a single class. Then you can derive two different classes from the common class and add specific properties" - Kirtan Gor class BaseClassWithSharedProperties { public Int32 SharedId { get; set; } public String SharedName { get; get; } } class UniqueClassNumberOne : BaseClassWithSharedProperties { public UniqueClassNumberOneProperty { get; set; } } class UniqueClassNumberTwo : BaseClassWithSharedProperties { public UniqueClassNumberTwoProperty { get; set; } }
