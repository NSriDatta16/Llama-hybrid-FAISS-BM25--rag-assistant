you can track that afeter scrolling, what is the coordinate of the origin, using the following two methods. This are two delegate methods conforms to your UIScrollView class. - (void)scrollViewDidEndScrollingAnimation:(UIScrollView *)newScrollView { NSLog(@"scrollViewDidEndScrollingAnimation"); CGPoint p = scrollView.contentOffset; int curr= floor(p.x/320); NSLog(@"x=%f,y=%f,curr=%d",p.x,p.y,curr); } - (void)scrollViewDidEndDecelerating:(UIScrollView *)newScrollView { [self scrollViewDidEndScrollingAnimation:newScrollView]; }
