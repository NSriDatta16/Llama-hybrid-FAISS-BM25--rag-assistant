the panel name is protected. you can derive your own class from CMFCRibbon and add a "SetName" method. class MyRibbonPanel : public CMFCRibbonPanel { public: MyRibbonPanel(LPCTSTR lpszName = NULL, HICON hIcon = NULL ) : CMFCRibbonPanel(lpszName, hIcon) {}; void SetName(CString& name ) { m_strName = name; }; }; for example ( after creating a dummy SDI application in VS2010 ) CMFCRibbonCategory* pCategory = m_wndRibbonBar.AddCategory(_T("&Legume"), IDR_PROPERTIES, IDB_PROPERTIES_HC ); MyRibbonPanel* pMyPanel = (MyRibbonPanel*)pCategory->AddPanel(_T("Patate"), m_PanelImages.ExtractIcon(1)); pMyPanel->SetKeys(_T("zc")); pMyPanel->SetCenterColumnVert(); pMyPanel->SetJustifyColumns(); CString s(_T("sdcasdc")); pMyPanel->SetName(s);
