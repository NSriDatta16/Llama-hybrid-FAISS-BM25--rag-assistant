A variation of the function provided by Paolo Bergantino that works directly on String: String.prototype.addSlashes = function() { //no need to do (str+'') anymore because 'this' can only be a string return this.replace(/[\\"']/g, '\\$&').replace(/\u0000/g, '\\0'); } By adding the code above in your library you will be able to do: var test = "hello single ' double \" and slash \\ yippie"; alert(test.addSlashes()); EDIT: Following suggestions in the comments, whoever is concerned about conflicts amongst JavaScript libraries can add the following code: if(!String.prototype.addSlashes) { String.prototype.addSlashes = function()... } else alert("Warning: String.addSlashes has already been declared elsewhere.");
