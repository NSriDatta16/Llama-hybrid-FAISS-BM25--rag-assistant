You need to use the invalid option, and disable and enable the target when you hide: $('#toggle').click(function(){ if($('#content').is(":visible")) { $( "#target" ).sortable( "option", "disabled", true ); $("#content").hide(); } else { $( "#target" ).sortable( "option", "disabled", false ); $("#content").show(); } }); $('#target').sortable(); $('#source div').draggable({ connectToSortable: '#target', helper: 'clone', revert: 'invalid' }); A modified JSFiddle showing this working .
