Is that what you mean? NSArray *frontCardArray = [NSArray arrayWithObject:[myDict valueForKey:@"frontCard"]]; NSArray *flipCardArray = [NSArray arrayWithObject:[myDict valueForKey:@"flipCard"]]; //EDIT: and just to make sure. If you set a value and then another: [myDic setValue:@"1" forKey:@"frontCard"]; [myDic setValue:@"2" forKey:@"frontCard"]; the key "frontCard" would NOT contain both! the "2" would overwrite the "1". You can only have one value in one key. OR you can save an array into a dictionary: NSArray *myArray = [NSArray arrayWithObjects:@"1", @"2", nil]; [myDic setValue:myArray forKey:@"frontCard"];
