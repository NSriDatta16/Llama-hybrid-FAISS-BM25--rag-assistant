For the more general case, where not only IDs are used, as suggested by @Anurag, you can do the following: // Save the selector var selector = ".someClass, #someId, tr.someTrClass"; $(selector).change(function () { var selectors = selector.split(","), matching = []; // Remember that each element can // match more than one selector for (var i = 0, s; s = selectors[i]; i++) { if ($(this).is(s)) matching.push(s); } str = matching.join(","); // Your list of all matching selectors });
