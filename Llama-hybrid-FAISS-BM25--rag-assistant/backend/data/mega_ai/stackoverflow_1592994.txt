use MouseDown, MouseMove and MouseUp. You can set a variable flag for that. I have a sample, but I think you need to revise. I am coding the mouse action to a panel. Once you click the panel, your form will move with it. //Global variables; private bool _dragging = false; private Point _offset; private Point _start_point=new Point(0,0); private void panel1_MouseDown(object sender, MouseEventArgs e) { _dragging = true; // _dragging is your variable flag _start_point = new Point(e.X, e.Y); } private void panel1_MouseUp(object sender, MouseEventArgs e) { _dragging = false; } private void panel1_MouseMove(object sender, MouseEventArgs e) { if(_dragging) { Point p = PointToScreen(e.Location); Location = new Point(p.X - this._start_point.X,p.Y - this._start_point.Y); } }
