How about writing a Wrapper class to the List you are using, this would avoid to have inconsistent storage of the elements. public class CaseInsensitiveStringList extends ArrayList { @Override public void add(final int index, final String element) { super.add(index, element.toLowerCase()); } @Override public boolean add(final String o) { return super.add(o.toLowerCase()); } @Override public boolean addAll(final Collection c) { final ArrayList temp = new ArrayList (c.size()); for (final String s : c) { temp.add(s.toLowerCase()); } return super.addAll(temp); } @Override public boolean addAll(final int index, final Collection c) { final ArrayList temp = new ArrayList (c.size()); for (final String s : c) { temp.add(s.toLowerCase()); } return super.addAll(index, temp); } }
