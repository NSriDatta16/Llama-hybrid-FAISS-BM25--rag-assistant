In this case, you could grab the contents of #phones during the ready event: $(document).ready(function() { var phnFragment = $("#phones").html(); $('#addmorephones').click(function() { $("#phones").append(phnFragment); }); }); If you need to copy more complex information (like events), you can use the .clone() method : $(document).ready(function() { var $phnFragment = $("#phones").children().clone(); $('#addmorephones').click(function() { $("#phones").append($phnFragment.clone()); }); });
