A less elegant and simpler way to do it: $password = new Zend_Form_Element_Password('password'); $password->setLabel('Password:') ->addValidator('StringLength', false, array(6,24)) ->setLabel('Choose your password:') ->setRequired(true); $password2 = new Zend_Form_Element_Password('password-confirm'); $password2->setLabel('Confirm:') ->addValidator('StringLength', false, array(6,24)) ->setLabel('Confirm your password:') ->addValidator(new Zend_Validate_Identical($_POST['password'])) ->setRequired(true);
