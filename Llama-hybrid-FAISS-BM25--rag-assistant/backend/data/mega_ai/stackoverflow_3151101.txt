I had problems with your solution because all browsers except for IE were firing the "onEachRequest" function BEFORE the jQuery validator's submitHandler() function. I came up with a more elegant solution that simply cancels the postback if validation fails, rather than relying on a preset variable. Sys.Application.add_init(function() { var prm = Sys.WebForms.PageRequestManager.getInstance(); prm.add_initializeRequest(onEachRequest); }); function onEachRequest(sender, args) { if (args.get_postBackElement().id == ) { args.set_cancel(!$('#aspnetForm').valid()); } }
