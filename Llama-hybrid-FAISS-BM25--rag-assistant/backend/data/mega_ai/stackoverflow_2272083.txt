This does the job but are there more elegant solutions? Private Sub Grid1ButtonClick(ByVal sender As System.Object, ByVal e As System.Windows.Forms.DataGridViewCellEventArgs) Handles Grid1.CellClick //allow checking, but don't allow unchecking If Grid1.Columns("myColumn").Index = e.ColumnIndex Then Dim chkbox As DataGridViewCheckBoxCell = Grid1.CurrentCell chkbox.ReadOnly = True chkbox.Value = True End If End Sub
