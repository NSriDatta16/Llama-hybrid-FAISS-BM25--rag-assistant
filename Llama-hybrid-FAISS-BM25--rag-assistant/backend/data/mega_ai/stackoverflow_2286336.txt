You can use HTTP::Request use HTTP::Request::Common qw(POST); use LWP::UserAgent; $ua = LWP::UserAgent->new; $ua->timeout(3); my $req = (POST 'http://stackoverflow.com', ["param1" => $var1, "param2" => $var2]); $response = $ua->request($req); $content = $response->content; exit; For response elements, see more here: http://kobesearch.cpan.org/htdocs/libwww-perl/HTTP/Request.html
