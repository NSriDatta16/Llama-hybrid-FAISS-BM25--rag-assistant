Here is a more realistic example: function Average() { $result = 0; $arguments = func_get_args(); foreach ($arguments as $argument) { $result += $argument; } return ($result / max(1, func_num_args())); } Average(1, 2, 3, 4, 5); // 3 This is called a variadic function .
