I don't know much about VB6, but in VBA... Dim objHttp As Object, strURL as string, strText as string Set objHttp = CreateObject("MSXML2.ServerXMLHTTP") strURL = "http://www.yoursite.com/" objHttp.Open "GET", strURL, False objHttp.setRequestHeader "User-Agent", _ "Mozilla/4.0 (compatible; MSIE 6.0; Windows NT 5.0)" objHttp.Send ("") strText = objHttp.responseText Set objHttp = Nothing
