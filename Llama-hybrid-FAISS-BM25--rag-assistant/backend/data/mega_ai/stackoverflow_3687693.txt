Edit: to address Timwi's concerns about 3 enumerators: bool same = ; var first = items.FirstOrDefault(); if (first != null) // assuming it's a class { same = items.Skip(1).All(i => i.Template.Frequency == first.Template.Frequency); } Which still uses 2 enumerators. Not an issue for the average List<> , but for a DB query it might pay to use the less readable: bool same = ; Item first = null; foreach(var item in items) { if (first == null) { first = item; same = true; } else { if (item.Template.Frequency != first.Template.Frequency) { same = false; break; } } }
