this is how I did it, you want the FieldType which actually returns a Type instance. var members = typeof(TestMe).GetFields().Select(m => new { Name = m.Name, MemType = m.MemberType, RtField = m.GetType(), Type = m.FieldType }); foreach (var item in members) Console.WriteLine(" {3} {2}", item.MemType, item.RtField, item.Name, item.Type, item.Type, item.Name); public class TestMe { public string A; public int B; public byte C; public decimal D; } This is the output: System.String A System.Int32 B System.Byte C System.Decimal D
