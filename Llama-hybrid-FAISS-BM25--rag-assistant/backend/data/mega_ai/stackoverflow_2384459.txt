In addition to my other answer, you can do this manually in a Control like this: Point dragOffset; protected override void OnMouseDown(MouseEventArgs e) { base.OnMouseDown(e); if (e.Button == MouseButtons.Left) { dragOffset = this.PointToScreen(e.Location); var formLocation = FindForm().Location; dragOffset.X -= formLocation.X; dragOffset.Y -= formLocation.Y; } } protected override void OnMouseMove(MouseEventArgs e) { base.OnMouseMove(e); if (e.Button == MouseButtons.Left) { Point newLocation = this.PointToScreen(e.Location); newLocation.X -= dragOffset.X; newLocation.Y -= dragOffset.Y; FindForm().Location = newLocation; } } EDIT : Tested and fixed - this now actually works.
