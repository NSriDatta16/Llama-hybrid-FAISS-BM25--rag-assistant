Maybe sed and awk have more elegant solutions, but I have a "poor man's" approach with grep, cut, head, and tail. #!/bin/bash dataFile="/path/to/some/data.txt" startToken="token 1" stopToken="token 2" startTokenLine=$( grep -n "${startToken}" "${dataFile}" | cut -f 1 -d':' ) stopTokenLine=$( grep -n "${stopToken}" "${dataFile}" | cut -f 1 -d':' ) let stopTokenLine=stopTokenLine-1 let tailLines=stopTokenLine-startTokenLine head -n ${stopTokenLine} ${dataFile} | tail -n ${tailLines}
