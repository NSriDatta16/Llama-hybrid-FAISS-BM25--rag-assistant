You need to handle DRAG_ENTER too: Handlers: private function input_dragEnterHandler(event:DragEvent):void { var data:Array = event.dragSource.dataForFormat("items") as Array; if (data && data.length > 0) DragManager.acceptDragDrop(input); } private function input_dragDropHandler(event:DragEvent):void { var data:Array = event.dragSource.dataForFormat("items") as Array; input.text = data[0].name; }
