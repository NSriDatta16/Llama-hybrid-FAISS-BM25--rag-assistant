typedef struct User { struct sockaddr_storage address; // User's address socklen_t address_length; // Length of users address int fs_id; // ID to the socket they belong to char ip_address[INET6_ADDRSTRLEN]; // User's IP address char *name; // Pointer to the user's name struct User *nextUser; // Next user in the list } User; Add the "struct User" at the top so that you can reference it within the struct. Also, you need to swap the order of your declarations so that the Server struct knows about the User struct.
