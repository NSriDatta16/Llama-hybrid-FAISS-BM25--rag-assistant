To get the system time Dim timestamp As Date: timestamp = Now timestamp = timestamp - Int(timestamp) To add or subtract a time interval (e.g. add one hour) timestamp = DateAdd("h", 1, timestamp) To store it in a database, something like Dim cnn As New Connection cnn.ConnectionString = "YourConnectionString" Dim cmd As New Command cmd.CommandType = adCmdStoredProc cmd.CommandText = "YourStoredProcName" cmd.ActiveConnection = cnn Dim prm As Parameter: Set prm = cmd.CreateParameter("YourParameterName", adDBTime, adParamInput) prm.Value = timestamp Call cmd.Parameters.Append(prm) Call cnn.Open Call cmd.Execute Call cnn.Close
