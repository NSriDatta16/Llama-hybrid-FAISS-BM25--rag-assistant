If you are targeting iOS 3.2+ I recommend using UIGestureRecognizer . For your use case UIPanGestureRecognizer is the right choice. The gesture handling code should be look something like this: - (void)handlePanGesture:(UIGestureRecognizer *)gestureRecognizer; { if (gestureRecognizer.state == UIGestureRecognizerStateBegan) { panOffset = [gestureRecognizer locationOfTouch:0 inView:topView]; } if (gestureRecognizer.state == UIGestureRecognizerStateChanged) { CGPoint delta = self.scrollView.contentOffset; delta += ([gestureRecognizer locationOfTouch:0 inView:scrollView].x - panOffset.x); self.scrollView.contentOffset = delta; } if (gestureRecognizer.state == UIGestureRecognizerStateEnded) { if(self.scrollView.contentOffset >= (self.scrollView.bounds.size.width / 2)) { // animate to the next multiple of self.scrollView.bounds.size.width } else { // animate to previous multiple } } }
