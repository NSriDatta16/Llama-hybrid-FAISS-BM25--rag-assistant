I think found solution here : var result = ( from role in db.Roles join user in db.Users on role.RoleID equals user.RoleID join deal in db.Deals on user.UserID equals deal.SalesAgentID into deals where user.CreatedByUserID == userID orderby user.FirstName ascending select new UserViewModel { UserID = user.UserID, PhotoID = user.PhotoID.ToString(), FirstName = user.FirstName, LastName = user.LastName, FullName = user.FirstName + " " + user.LastName, Email = user.Email, PhoneNumber = user.Phone, AccessLevel = role.Name, DealsThisYear = deals.Where(deal => deal.DateCreated.Year == date.Year).Count() });
