I'm not sure, if this is the most elegant solution, but it should work (tested on Postgres): SELECT (CASE WHEN max(sub.jan)=1 THEN true ELSE false END) AS jan, (CASE WHEN max(sub.feb)=1 THEN true ELSE false END) AS feb, ... FROM ( SELECT (CASE WHEN (EXTRACT(MONTH FROM logdate))=1 THEN 1 ELSE 0 END) AS jan, (CASE WHEN (EXTRACT(MONTH FROM logdate))=2 THEN 1 ELSE 0 END) AS feb, ... FROM mytable WHERE EXTRACT(YEAR FROM logdate) = '2009' ) as sub
