This can be achieved by passing a supporting function as an argument to the Array.sort method call. Something like this: function Comparator(a, b) { if (a[1] b[1]) return 1; return 0; } var myArray = [ [1, 'alfred', '...'], [23, 'berta', '...'], [2, 'zimmermann', '...'], [4, 'albert', '...'], ]; myArray = myArray.sort(Comparator); console.log(myArray);
