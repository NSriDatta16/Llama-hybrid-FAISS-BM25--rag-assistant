Is there an elegant way in LINQ The elegant way is not very performant. Here is a more performant way... public static List > Chunk ( this List theList, int chunkSize ) { if (!theList.Any()) { return new List >(); } List > result = new List >(); List currentList = new List (); result.Add(currentList); int i = 0; foreach(T item in theList) { if (i >= chunkSize) { i = 0; currentList = new List (); result.Add(currentList); } i += 1; currentList.Add(item); } return result; }
