You can create a custom log level by extending the Level class; just give it a unique ID. import java.util.logging.*; public class CustomLogLevel extends Level { public static void main(String[] args) { Logger log = Logger.getLogger("robertgrant.org"); Level templevel = Level.WARNING; Level level = new CustomLogLevel("Rob Level", templevel.intValue()); Level customlevel = level.parse("Rob Level"); log.log(customlevel, "This is from a custom level"); } public CustomLogLevel(String name, int value){ super(name, value); } }
