Assuming that space between the fields are not fixed so split string on the basis of two or more spaces so that it will not break the Name like JONH MILLER into two parts . #!/usr/bin/perl use strict; use warning; my $string = "NAME1 NAME2 DEPTNAME POSITION JONH MILLER ROBERT JIM CS ASST GENERAL MANAGER "; my @string_parts = split /\s\s+/, $string; foreach my $test (@string_parts){ print"$test\n"; }
