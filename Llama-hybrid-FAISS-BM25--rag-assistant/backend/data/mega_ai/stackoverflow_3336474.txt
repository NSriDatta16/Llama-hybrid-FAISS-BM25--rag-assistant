Have you tried using height instead? var li_content_height = $('span').height() + $(this).height(); $(this).css('height', li_content_height); If you want the height including padding or margin, then you can use outerHeight : var heightWithPadding = $('span').outerHeight(); var heightWithPaddingAndMargin = $('span').outerHeight(true);
