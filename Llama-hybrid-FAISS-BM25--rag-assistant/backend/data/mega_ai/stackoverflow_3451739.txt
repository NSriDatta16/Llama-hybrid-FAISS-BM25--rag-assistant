Just set the location to the adjusted href value and return false from the handler to avoid taking the original link. $(function(){ $('#cnt-area form table tbody').dragCheck(); // I just split up the .each and .click to see if it mattered, which it doesnt $('.multi_assign_link').each(function(){ $(this).click(function(e){ var href = $(this).attr('href'); $('#cnt-area form input:checkbox').each(function(){ if($(this).attr('checked')){ if(href.search(','+$(this).val()) == -1){ href += ','+$(this).val(); } }else{ var s = ','+$(this).val(); s.toString(); href = href.replace(s, ''); } }); location.href = href; return false; }); }); });
