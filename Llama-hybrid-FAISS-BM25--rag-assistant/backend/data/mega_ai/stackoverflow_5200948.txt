I solved this by adding a listener to each marker in the for loop after creating them. I added an id attribute to the marker in order to create the listeners. $.getJSON('get_markers.php', function(data) { $.each(data, function(pair) { var id = data[pair]['id']; $.goMap.createMarker({ latitude: data[pair]['lat'], longitude: data[pair]['lng'], draggable: false, id: id /*html: { ajax: 'marker_description.php?q=' + id, content: 'loading...' }*/ }); $.goMap.createListener({type:'marker', marker:id}, 'click', function() { $.ajax({ url: "show_post.php?q="+id, success: function(html){ $("#results").html(html); } }); }); }); }); Would this hurt performance though?
