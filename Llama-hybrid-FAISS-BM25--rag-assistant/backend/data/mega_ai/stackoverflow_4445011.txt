You can drill down to the object type by using the following method: private void OldFaithful_ItemDragStarting(object sender, ItemDragEventArgs e) { SelectionCollection selections = e.Data as SelectionCollection; if (selections != null) { IEnumerable draggedItems = selections.Select(s => s.Item as YOUREXCPECTEDOBJECTTYPE); foreach (YOUREXCPECTEDOBJECTTYPE x in draggedItems) { MessageBox.Show(x.GetType().ToString()); } } }
