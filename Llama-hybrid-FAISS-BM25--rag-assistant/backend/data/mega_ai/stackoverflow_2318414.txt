Full example (Robert) public abstract class AsyncCall implements AsyncCallback { public AsyncCall() { loadingMessage.show(); } public final void onFailure(Throwable caught) { loadingMessage.hide(); onCustomFailure(caught); } public final void onSuccess(T result) { hideLoadingMessage(); onCustomSuccess(result); } /** the failure method needed to be overwritte */ protected abstract void onCustomFailure(Throwable caught); /** overwritte to do something with result */ protected abstract void onCustomSuccess(T result); }
