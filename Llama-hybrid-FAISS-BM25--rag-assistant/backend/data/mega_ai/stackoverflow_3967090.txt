I know you asked for RegEx, but I won't do it. Instead here's a solution using Html Agility Pack . public static void Parse() { string htmlFragment = @" I want to match the word 'highlight' in a string. But I don't want to match highlight when it is contained in an HTML anchor element. The expression should not match highlight in the following text: highlight more "; HtmlDocument htmlDocument = new HtmlAgilityPack.HtmlDocument(); htmlDocument.LoadHtml(htmlFragment); foreach (HtmlNode node in htmlDocument.DocumentNode.SelectNodes("//.").Where(FilterTextNodes())) { Console.WriteLine(node.OuterHtml); } } private static Func FilterTextNodes() { return node => node.NodeType == HtmlNodeType.Text && node.ParentNode != null && node.ParentNode.Name != "a" && node.OuterHtml.Contains("highlight"); }
