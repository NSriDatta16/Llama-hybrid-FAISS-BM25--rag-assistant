HRESULT CreateCatalogManager(ISearchCatalogManager **ppSearchCatalogManager) { *ppSearchCatalogManager = NULL; ISearchManager *pSearchManager; HRESULT hr = CoCreateInstance(CLSID_CSearchManager, NULL, CLSCTX_SERVER, IID_PPV_ARGS(&pSearchManager)); if (SUCCEEDED(hr)) { hr = pSearchManager->GetCatalog(L"SystemIndex", ppSearchCatalogManager); pSearchManager->Release(); } return hr; } { ISearchCatalogManager *pCatalogManager; HRESULT hr = CreateCatalogManager(&pCatalogManager); if (SUCCEEDED(hr)) { pCatalogManager->Reset(); pCatalogManager->Release(); } }
