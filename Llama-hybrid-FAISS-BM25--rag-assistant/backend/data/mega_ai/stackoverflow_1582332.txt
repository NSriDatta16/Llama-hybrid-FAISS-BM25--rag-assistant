Here's a different approach for you to consider: use strict; use warnings; use File::Slurp qw(read_file); my %para = map { $_ => Read_paragraphs("$_.txt") } qw(FR EN); my $query = 'La petite ville de'; my @matches = map { $para{EN}[$_] } grep { $para{FR}[$_] =~ /$query/ } 0 .. @{$para{FR}} - 1 ; print $_, "\n" for @matches; sub Read_paragraphs { return [split /\n{2,}/, read_file(shift)]; }
