public function RemoveItem($item) { foreach($this->shopcart as $i => $key) { if($key['productID'] == $item) { unset($this->shopcart[$i]); break; } } } That should do the trick. Update There is also an alternative way: if ( false !== $key = array_search($item, $this->shopcart) ) { unset($this->shopcart[$key]; }
