You can also disable this warning in specific headers: #if defined(_MSC_VER) && _MSC_VER >= 1400 #pragma warning(push) #pragma warning(disable:4996) #endif /* your code */ #if defined(_MSC_VER) && _MSC_VER >= 1400 #pragma warning(pop) #endif
