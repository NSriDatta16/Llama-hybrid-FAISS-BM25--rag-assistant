Instead of calling updateLevel() in head's MOUSE_MOVE event, call it in the head's ENTER_FRAME event: head.addEventListener(MouseEvent.MOUSE_DOWN, function():void { head.startDrag(false, new Rectangle(stubDiv,0,width - stubDiv - ((levels-maxLevel)*stubDiv),0)); head.addEventListener(Event.ENTER_FRAME, function():void { updateLevel(); }); }); head.addEventListener(MouseEvent.MOUSE_UP, function():void { head.stopDrag(); head.removeEventListener(Event.ENTER_FRAME, function():void { updateLevel(); }); setHeadPos(); }); head.Stage.addEventListener(Event.MOUSE_LEAVE, function():void { head.stopDrag(); head.removeEventListener(Event.ENTER_FRAME, function():void { updateLevel(); }); setHeadPos(); }); I worked on a similar feature before and I also needed to handle Stage's Mouse_Leave. Depending on your requirement, you might also want to add MouseEvent.MOUSE_OUT to head
