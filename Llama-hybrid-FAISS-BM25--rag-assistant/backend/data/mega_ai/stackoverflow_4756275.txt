In your mgDiv.dialog function, modify the buttons key to have a conditional value. E.g.: function showDivPopUp(title,msg,options){ var mgDiv = $("#msgDiv"); mgDiv.attr("innerHTML", msg); return mgDiv.dialog({ modal: true, buttons: options.buttons || { Ok: function () { $(this).dialog("destroy"); } }, resizable: true, show: "explode", position: "center", closeOnEscape: true, draggable: false, title : titl, open: function (event, ui) { $(".ui-dialog-titlebar-close").hide(); } }); }
