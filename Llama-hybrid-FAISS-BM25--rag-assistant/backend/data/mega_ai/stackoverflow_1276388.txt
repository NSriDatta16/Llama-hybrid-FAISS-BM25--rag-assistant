Applying Singleton to your original class: public class Storage { private static Storage instance; private Storage() {} public static Storage Instance { get { if (instance == null) { instance = new Storage(); } return instance; } } public string FilePath { get; set; } } usage: string filePath = Storage.Instance.FilePath;
