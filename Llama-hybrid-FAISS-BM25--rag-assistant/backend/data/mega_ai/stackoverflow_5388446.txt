You can use traversing methods of jQuery var arrLabel1 = new Array('Not Good', 'OK', 'Average', 'Above Average', 'Excellent', 'Outstanding'); $(function() { $( ".slider_control" ).slider({ value:0, min: 0, max: 5, step: 1, slide: function( event, ui ) { $(this).siblings('input:hidden').val( ui.value ); $(this).siblings('.title').children('span').text( arrLabel1[ ui.value ] ); } }).each(function(index){ $(this).siblings('input:hidden').val( $( this ).slider( "value" ) ); $(this).siblings('.title').children('span').text( arrLabel1[ $(this).slider( "value" ) ] ); }); });
