You can use the following code for IE6 based on jQuery library if (jQuery.browser.msie && navigator.userAgent.toLowerCase().indexOf('msie 6') > -1){ jQuery(document).ready(function() { jQuery('.menu li').hover( function() { jQuery(this).addClass('hover'); }, function() { jQuery(this).removeClass('hover'); } ); }); } Please change the selector ".menu li" to yours and wrote in CSS in the following way .menu li:hover, .menu li.hover { ... }
