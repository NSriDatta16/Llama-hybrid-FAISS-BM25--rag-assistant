(" Tracking Number: " + row["Tracking_No"]) : (""); MailMessage msg = new MailMessage("DoNotReply@optoma.com", to, subject, string.Format("Dear {0}, {1} Your ISO ID is {2}{3} Please examine the loaned items for this transaction: {4}", attention, body, isoId, message, sw)); }
