This code will allow you to click and place two markers, which you can then use to load GDirections, and remove the original markers. Note that you must use this format for the query string: "from: marker@35,-25 to: marker@-20,15". var markerArray = []; var listener = GEvent.addListener(map, "click", function(overlay, latlng) { var marker = new GMarker(latlng, { draggable: true }); map.addOverlay(marker); markerArray.push(marker); if (markerArray.length > 1) { GEvent.removeListener(listener); var marker1 = markerArray[0]; var marker2 = markerArray[1]; gdir.load("from: marker1@" + marker1.getLatLng() + " to: marker2@" + marker2.getLatLng()); map.removeOverlay(marker1); map.removeOverlay(marker2); } });
