Timer pressTimer = new Timer(2000); pressTimer.addEventListener(TimerEvent.TIMER, onTimer); container.addEventListener(MouseEvent.MOUSE_DOWN, onMouseDown); container.addEventListener(MouseEvent.MOUSE_UP,onMouseUp); function onMouseDown(e:MouseEvent):void { pressTimer.start(); } function onMouseUp(e:MouseEvent):void { pressTimer.reset(); } function onTimer(e:TimerEvent):void { pressTimer.reset(); //do the dragging and stuff. }
