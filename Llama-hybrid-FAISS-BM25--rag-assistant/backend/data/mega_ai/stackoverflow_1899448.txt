Dispose RegistryKey after usage. Don't cache unless you need it. Don't afraid to hardcode registry key names - they will most probably be only used in one place. public class Agent { public string Version { get { using (var regkey = Registry.LocalMachine.OpenSubKey(SpecialRegistry.SpecialAgentRoot)) return (string)regkey.GetValue("CurrentVersion"); } } }
