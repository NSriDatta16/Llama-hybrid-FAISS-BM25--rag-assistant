You can use the System.Net.HttpWebRequest object. It looks something like this: // Setup the HTTP request. HttpWebRequest httpWebRequest = (HttpWebRequest)HttpWebRequest.Create("http://www.google.com"); // This is optional, I'm just demoing this because of the comments receaved. httpWebRequest.UserAgent = "My Web Crawler"; // Send the HTTP request and get the response. HttpWebResponse httpWebResponse = (HttpWebResponse)httpWebRequest.GetResponse(); if (httpWebResponse.StatusCode == HttpStatusCode.OK) { // Get the HTML from the httpWebResponse... Stream responseStream = httpWebResponse.GetResponseStream(); StreamReader reader = new StreamReader(responseStream); string html = reader.ReadToEnd(); }
