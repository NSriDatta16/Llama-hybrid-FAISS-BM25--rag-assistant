SELECT charGuild, AVG(charRating) average FROM ( SELECT charRating, charGuild, @num := IF(@guild = charGuild, @num + 1, 1) AS row_number, @guild := charGuild FROM ins_rated a ORDER BY charGuild DESC ) ordered WHERE row_number Output: charGuild average team 1 95.0000 team 2 97.5000
