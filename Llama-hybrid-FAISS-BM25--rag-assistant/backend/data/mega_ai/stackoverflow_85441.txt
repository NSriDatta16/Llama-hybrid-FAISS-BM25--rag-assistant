Microsoft guidelines as checked by FxCop discourage use of List in public APIs - prefer IList . Incidentally, I now almost always declare one-dimensional arrays as IList , which means I can consistently use the IList .Count property rather than Array.Length. For example: public interface IMyApi { IList GetReadOnlyValues(); } public class MyApiImplementation : IMyApi { public IList GetReadOnlyValues() { List myList = new List (); ... populate list return myList.AsReadOnly(); } } public class MyMockApiImplementationForUnitTests : IMyApi { public IList GetReadOnlyValues() { IList testValues = new int[] { 1, 2, 3 }; return testValues; } }
