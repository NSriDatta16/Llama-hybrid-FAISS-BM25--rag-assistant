I used an array that matches my marker data for my HTML and it works well: function createMarker(posn, title, icon, i) { var marker = new GMarker(posn, {title: title, icon: icon, draggable:false}); GEvent.addListener(marker, 'mouseover', function() { map.closeInfoWindow() marker.openInfoWindowHtml(infoText[i]) } ); return marker; }
