Turns out that this seems to work: SELECT * FROM visitor_fields INNER JOIN visitor_inputs ON (visitor_inputs.input_id = visitor_fields.input_id) INNER JOIN visitor_fields as filter_0 ON (filter_0.input_id=visitor_inputs.input_id AND filter_0.field_name = 'province' AND filter_0.field_value != 'Alberta') INNER JOIN visitor_fields as filter_1 ON (filter_1.input_id=visitor_inputs.input_id AND filter_1.field_name = 'country' AND filter_1.field_value = 'Canada') INNER JOIN visitor_fields as filter_2 ON (filter_2.input_id=visitor_inputs.input_id AND filter_2.field_name = 'first_name' AND filter_2.field_value LIKE '%jim%')
