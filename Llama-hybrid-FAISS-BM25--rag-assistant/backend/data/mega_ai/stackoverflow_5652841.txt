This is the easiest way I can think of doing it: create table #tempSales (salesagentid int, salesagentname varchar(50), salesamount money) go insert into #tempSales select salesagentid, salesagentname, sum(salesamount) from salesTable group by salesagentid, salesagentname go select * from #tempSales
