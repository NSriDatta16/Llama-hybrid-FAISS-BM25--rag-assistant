jQuery Drag and drop #dv { position: absolute; cursor: move; } var x1 = y1 = 0; var drag = false; $(document).ready(function(e){ //this is a possible solution jQuery('#dv>img').disableSelection(); $('#dv').mousedown(function(e){ x1 = e.pageX - parseInt($('#dv').css('left')); y1 = e.pageY - parseInt($('#dv').css('top')); drag = true; }) $(document).mouseup(function(e){ drag = false; }) $(document).mousemove(function(e){ if(!drag) return $('#dv').css('left',eval(e.pageX - x1)+'px') $('#dv').css('top',eval(e.pageY - y1)+'px') }) });
