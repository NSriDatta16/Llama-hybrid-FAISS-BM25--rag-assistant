This is what I use for fragments, adapted from something found on the web somewhere. from twisted.words.xish import domish class ElementParser(object): "callable class to parse XML string into Element" def __call__(self, s): self.result = None def onStart(el): self.result = el def onEnd(): pass def onElement(el): self.result.addChild(el) parser = domish.elementStream() parser.DocumentStartEvent = onStart parser.ElementEvent = onElement parser.DocumentEndEvent = onEnd tmp = domish.Element(("", "s")) tmp.addRawXml(s) parser.parse(tmp.toXml()) return self.result.firstChildElement()
