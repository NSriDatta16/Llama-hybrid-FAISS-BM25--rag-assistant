You lack Content-type for your request: "application/x-www-form-urlencoded" . This works: import urllib import httplib params = urllib.urlencode({'method' : 'artist.getsimilar', 'artist' : 'band', 'limit' : '5', 'api_key' : '#API key goes here'}) header = {"user-agent" : "myapp/1.0", "Content-type": "application/x-www-form-urlencoded"} lastfm = httplib.HTTPConnection("ws.audioscrobbler.com") lastfm.request("POST","/2.0/?",params,header) response = lastfm.getresponse() print response.read()
