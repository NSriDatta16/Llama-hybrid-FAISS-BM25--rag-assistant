I got an elegant and fast solution from Maxim Rytin, relying on ConnectedComponents function Module[{gens, verts, edges}, gens = PermutationList /@ GroupGenerators@DihedralGroup[16]; verts = Permutations@{2, 2, 2, 2, 2, 2, 2, 1, 1, 0, 0, 0, 0, 0, 0, 0}; edges = Join @@ (Transpose@{verts, verts[[All, #]]} &) /@ gens; Length@ConnectedComponents@Graph[Rule @@@ Union@edges]] // Timing
