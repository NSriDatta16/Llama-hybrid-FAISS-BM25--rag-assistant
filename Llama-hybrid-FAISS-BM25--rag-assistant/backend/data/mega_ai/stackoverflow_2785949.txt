Add this to your db/seed.rb file: Account.transaction do [ ["Albert", "albert@abc.com", "123456", "ABC Company"], ["Betty", "betty@abc.com", "234567", "ABC Company"] ].each do |record| company = find_or_create_by_company_name(record[3]) company.users.create(:user_name => record[0].downcase, :email => record[1], :password => record[2]) end end Run the following rake task: rake db:seed
