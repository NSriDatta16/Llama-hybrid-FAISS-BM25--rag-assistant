Like this, for example: public static string GetUa(HttpRequest hr) { try { string originalBrowser = hr.ServerVariables["X-OperaMini-Phone-UA"]; string anotherOriginalBrowser = hr.ServerVariables["X-Device-User-Agent"]; return !String.IsNullOrEmpty(originalBrowser) ? "OPERAMINI " + originalBrowser : !String.IsNullOrEmpty(anotherOriginalBrowser) ? "NOVARRA " + anotherOriginalBrowser : hr.UserAgent; } catch { return "No UA Found"; } }
