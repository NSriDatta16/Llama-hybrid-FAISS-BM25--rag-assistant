try out jQuery.simulate plugin Something like this should work. var dropZone = $("#dropZone").offset() //get dropZone's offset object $("#dragableEle").simulate("drag", { dx: dropZone.left, // move to this x dy: dropZone.top, // move to this y speed:5000 // set speed });
