Here's an example that keeps the concepts of delegates out, and just does a raw call back. @interface Foo : NSObject { } - (void)doSomethingAndNotifyObject:(id)object withSelector:(SEL)selector; @end @interface Bar : NSObject { } @end @implementation Foo - (void)doSomethingAndNotifyObject:(id)object withSelector:(SEL)selector { /* do lots of stuff */ [object performSelector:selector withObject:self]; } @end @implementation Bar - (void)aMethod { Foo *foo = [[[Foo alloc] init] autorelease]; [foo doSomethingAndNotifyObject:self withSelector:@selector(fooIsDone:)]; } - (void)fooIsDone:(id)sender { NSLog(@"Foo Is Done!"); } @end Typically the method -[Foo doSomethingAndNotifyObject:withSelector:] would be asynchronous which would make the callback more useful than it is here.
