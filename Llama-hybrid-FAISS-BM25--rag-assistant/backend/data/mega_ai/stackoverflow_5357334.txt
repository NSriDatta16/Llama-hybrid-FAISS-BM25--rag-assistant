This is a little crude and I'm sure there are much more elegant solutions out there but it does the job. Set GRAB_FILE and GRAB_FROM_REV appropriately. $ mkdir versions $ GRAB_FILE="MGSource.h" $ GRAB_FROM_REV=700 $ for i in `svn log -r$GRAB_FROM_REV:HEAD "$GRAB_FILE" | grep -o '^r[0123456789]*'`; do svn up "$GRAB_FILE" -$i; cp $GRAB_FILE versions/"$GRABFILE.$i.h"; done Updated to revision 768.
