I finally ended up getting this to work on my own. The code I ended up using is posted below. Creating the map of objects in the correct order proved to be important. I learned a lot fighting through this thing. I've organized my mappings into a profile, which I won't get into here, suffice to say that if you can use my example outside of a class inheriting from the AutoMapper Profile class, you'll want to use Mapper.CreateMap instead of just Create Map. private void CreateMaps() { CreateMap (); //ContactEntity Mapping CreateMap (); //CustomerEntity Mapping CreateMap (); //AddressEntity Mapping CreateMap (); //ServiceEntity Mapping CreateMap () .ForMember(dto => dto.ServiceTypeCode, opt => opt.MapFrom(source => source.TypeCode)) .ForMember(dto => dto.ServiceDescriptionCode, opt => opt.MapFrom(source => source.DescriptionCode)) .ForMember(dest => dest.SourceSystemName, opt => opt.ResolveUsing ().FromMember(entity => entity.SourceSystemName)); //VehicleEntity Mapping CreateMap () .ForMember(dest => dest.SourceSystemName, opt => opt.ResolveUsing ().FromMember(entity => entity.SourceSystemName)) .ForMember(dto => dto.PortalId, option => option.Ignore()); //TODO: Should PortalID be mapped to anything? It is not in the entity. //ContentEntity Mapping CreateMap () .ForMember(dest => dest.SourceSystemName, opt => opt.ResolveUsing ().FromMember(entity => entity.SourceSystemName)); //OrderForwardingEntity Mapping CreateMap (); //ContainerEntity Mapping CreateMap () .ForMember(dest => dest.SourceSystemName, opt => opt.ResolveUsing ().FromMember(entity => entity.SourceSystemName)); //ShipmentForwardingEntity Mapping CreateMap (); //ShipmentRouting Mapping CreateMap (); //ShipmentEntity Mapping CreateMap () .ForMember(dest => dest.SourceSystemName, opt => opt.ResolveUsing ().FromMember(entity => entity.SourceSystemName)) .ForMember(dto => dto.Services, option => option.MapFrom(source => source.ServiceEntities)); //Forwarder mapping CreateMap (); //TODO: This property doesn't have any properties in the data contract //OrderEntity Mapping CreateMap () .ForMember(dest => dest.SourceSystemName, opt => opt.ResolveUsing ().FromMember(entity => entity.SourceSystemName)); //.ForMember(dto => dto.Forwarder, option => option.MapFrom(entity=>entity.Forwarder) //MoveEntityMapping CreateMap () .ForMember(dto => dto.SourceSystemName, opt => opt.ResolveUsing ().FromMember(entity => entity.SourceSystemName)); }
