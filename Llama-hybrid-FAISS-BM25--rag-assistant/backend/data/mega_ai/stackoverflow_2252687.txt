NHibernate mapping

How to map this class? [Serializable] public class AgentSourceCounter { private int agentId; private IDictionary sourceCounters; protected AgentSourceCounter() { } public AgentSourceCounter(int agentId, Dictionary sourceCounters) { this.agentId = agentId; this.sourceCounters = sourceCounters; } public virtual int AgentId { get { return agentId; } protected set { agentId = value; } } public virtual IDictionary SourceCounters { get { return sourceCounters; } set { sourceCounters = value; } } } To this table: agentId int; sourceId int; hitsCounter int;
