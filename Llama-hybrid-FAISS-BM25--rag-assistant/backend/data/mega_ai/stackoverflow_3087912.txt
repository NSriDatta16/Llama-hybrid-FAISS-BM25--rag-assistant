public function getQuerystringProperty(property:String):String { var bm:IBrowserManager = BrowserManager.getInstance(); var oArgs:Object = {}; bm.init("", ""); oArgs = mx.utils.URLUtil.stringToObject(bm.fragment, “&”); if (oArgs[property]) return oArgs[property].toString(); return ""; } Gets the QueryString from within Flex (no ExternalInterface).
