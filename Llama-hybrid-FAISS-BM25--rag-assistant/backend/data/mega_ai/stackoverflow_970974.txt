I have solved it by editing the addNode funcion. I commented the temp_node.remove(); and added dragNode_destination.after(dragNode_source); Just like that: TREE.addNode = function(id, text, callback) { var temp_node = $(' '+text+' '); TREE.setTreeNodes(temp_node); dragNode_destination = TREE.getSelected(); dragNode_source = $('.doc-last',temp_node); TREE.moveNodeToFolder(dragNode_destination); // temp_node.remove(); dragNode_destination.after(dragNode_source); if(typeof(callback) == 'function') { callback(dragNode_destination, dragNode_source); } };
