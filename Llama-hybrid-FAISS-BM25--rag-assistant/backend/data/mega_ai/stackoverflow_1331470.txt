Its simple really, when you think about it. 0 == 'any text which does not start with a number' Your last digit 0 will cause an endless loop. So you need to change it to if ($day === 'array') EDIT I also took the liberty to fix up your code: /** * @obsolete */ public function getRepeat($day = "array") { if ($day === 'array') { return $this->getAllRepeat(); } return $this->getRepeatByDay($day); } public function __construct() { $this->repeat = array_fill(0, 7, ''); } public function getAllRepeat() { return $this->repeat; } public function __get($value) { switch ($value) { case 'repeat0': case 'repeat1': case 'repeat2': case 'repeat3': case 'repeat4': case 'repeat5': case 'repeat6': return $this->getRepeatByDay(intval(substr($value, -1, 1))); } } public function getRepeatByDay($day) { if (!isset($this->repeat[$day])) { return null; } return $this->repeat[$day]; }
