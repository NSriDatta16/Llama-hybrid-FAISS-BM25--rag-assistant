In continuation from your other post, I added the helper: 'clone' option, and then told the script to remove any elements with the .ui-draggable-dragging class jQuery.fn.swapWith = function(to) { return this.each(function() { var copy_to = $(to).clone(); var copy_from = $(this).clone(); $(to).replaceWith(copy_from); $(this).replaceWith(copy_to); }); }; $(document).ready(function() { options = { revert: true, helper: 'clone' }; $("li").draggable(options); $('#wrapper').droppable({ drop: function(event, ui) { // window.setTimeout(function(){ $('#one').swapWith($('#two')); $(".ui-draggable-dragging").remove(); $("li").draggable(options); //}, 1); } }); }); working example: http://jsfiddle.net/XkUDv/26/ Hope that helps!
