If you want to set an entire div at once without using innerHTML, you can use replaceChild() : function createDetails(...) { var details = document.getElementById("detailsDIV"); var docFragment = document.createDocumentFragment(); // Process docFragment here... var parentNode = details.parentNode; parentNode.replaceChild(docFragment, details); }
