Here is a solution which will work for any depth: def depthGenerator(seq, depth): if depth==0: for x in seq: yield x[:2] #strip subsequences return for x in seq: if len(x)==3: #has subsequence? for y in depthGenerator(x[2], depth-1): yield y Example: >>> list(depthGenerator(PLACES, 1)) [('AB', 'Alberta'), ('BC', 'British Columbia'), ('AL', 'Alabama'), ('AK', 'Alaska')]
