Scikit-learn FeatureUnion is what you are looking for: import numpy as np from sklearn.preprocessing import FunctionTransformer from sklearn.pipeline import FeatureUnion from sklearn.datasets import load_iris X, _ = load_iris(return_X_y = True) def col_add(x): x1 = x[:, 0] + x[:, 1] x2 = x[:, 0] * x[:, 1] x3 = x[:, 0] / x[:, 1] return np.c_[x, x1, x2, x3] transformer = FunctionTransformer(col_add) f = FeatureUnion([("t",transformer)]).fit(X) Xnew = f.transform(X)
