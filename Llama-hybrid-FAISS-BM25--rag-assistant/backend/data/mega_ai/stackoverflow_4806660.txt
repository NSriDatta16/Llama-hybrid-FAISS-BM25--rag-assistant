I realize this question has already been answered, but I was googling for this problem today and came across this link. I used a javascript fix that seems to work very well. I found the original javascript in this Mozilla bug discussion, and then I modified it a bit. https://bugzilla.mozilla.org/show_bug.cgi?id=550246 Here is the javascript that I ended up using (I hope it shows up properly in this post): var isFireFoxMac=false; if (navigator.userAgent.indexOf("Firefox")!=-1) { if (navigator.platform == "MacIntel" || navigator.platform == "MacPPC") { isFireFoxMac = true; } } function isEven(value){ return (value%2 == 0); } function ensureOddWidth() { var width = window.innerWidth; if (isEven(width)) { self.resizeTo(width-1, window.outerHeight); } } if (isFireFoxMac) { window.onresize = ensureOddWidth; window.onload = ensureOddWidth; }
