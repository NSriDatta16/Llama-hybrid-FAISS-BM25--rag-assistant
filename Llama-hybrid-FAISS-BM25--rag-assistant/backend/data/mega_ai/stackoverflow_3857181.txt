This code does it for strings. var users = new [] {"John", "Richard", "Jack", "Roy", "Robert", "Susan" }; var prefixes = new [] { "J", "Ro" }; var filtered = prefixes.Aggregate(Enumerable.Empty (), (acc, pref) => acc.Union(users.Where(u => u.StartsWith(pref)).ToList())); For your User class it would look like var filtered = prefixes.Aggregate( Enumerable.Empty (), (acc, pref) => acc.Union( users.Where( u => u.FistName.StartsWith(pref) || u.LastName.StartsWith(pref) ).ToList()));
