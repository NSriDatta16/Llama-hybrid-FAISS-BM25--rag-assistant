Are you sure both average_r and amt_r are both in descending order? SELECT news_id FROM (SELECT news_id, AVG(rating) AS average_r, COUNT(*) AS amt_r FROM rates GROUP BY news_id ORDER BY average_r DESC, amt_r DESC LIMIT 5 ) AS news_rates
