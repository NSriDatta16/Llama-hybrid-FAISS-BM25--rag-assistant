You can disable the default decorator for 'bio' element by adding a call to setDisableLoadDefaultDecorators() $this->addElement('textarea', 'bio', array( 'disableLoadDefaultDecorators' => true, 'decorators' => array( 'ViewHelper', 'Errors', array('Description', array('tag' => 'p', 'escape' => false)), 'Label', array('HtmlTag', array('tag' => 'p')), new Application_Form_Decorator_WmdPreview, ) ));
