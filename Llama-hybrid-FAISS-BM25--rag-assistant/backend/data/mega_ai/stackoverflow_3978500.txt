To get whatever color you want, you have to replace the selectedBackgroundView with your own UIView. - (UITableViewCell *)tableView:(UITableView *)table cellForRowAtIndexPath:(NSIndexPath *)indexPath { static NSString *CellIdentifier = @"CellID"; UITableViewCell *cell = [table dequeueReusableCellWithIdentifier:CellIdentifier]; if (!cell) { cell = [[[UITableViewCell alloc] initWithStyle:UITableViewCellStyleDefault reuseIdentifier:CellIdentifier] autorelease]; UIView *selectedBackground = [[[UIView alloc] init] autorelease]; selectedBackground.backgroundColor = [UIColor magentaColor]; cell.selectedBackgroundView = selectedBackground; } // configure cell return cell; }
