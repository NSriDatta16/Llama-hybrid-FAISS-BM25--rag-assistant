select cName as couseName, a.tName as masterName, a.tAge as masterAge, a.lDesc as masterLocation, c.tName as assistName, c.tAge as assistAge, d.lDesc as assistLocation from CourseInfo join TeacherInfo a on masterTeacherId = a.tID join LocationInfo b on a.tLocationId = b.lId join TeacherInfo c on assistTeacherId = c.tID join LocationInfo d on c.tLocationId = d.lId
