This is a solution for jquery's jquery.bgiframe.js . if ( $.browser.msie && /9.0/.test(navigator.userAgent) ) { var iframe = document.createElement("iframe"); iframe.setAttribute("class", "bgiframe"); iframe.setAttribute("frameborder", "0"); iframe.setAttribute("style", "display:block;position:absolute;z-index:-1;filter:Alpha(Opacity=\'0\');top:expression(((parseInt(this.parentNode.currentStyle.borderTopWidth)||0)*-1)+\'px\');left:expression(((parseInt(this.parentNode.currentStyle.borderLeftWidth)||0)*-1)+\'px\');width:expression(this.parentNode.offsetWidth+\'px\');height:expression(this.parentNode.offsetHeight+\'px\');"); this.insertBefore( iframe, this.firstChild ); } else { this.insertBefore( document.createElement(html), this.firstChild ); }
