I'm not absolutely sure but I'm always using try-catch-finally block for db actions: using (SqlConnection cnn = new SqlConnection(ConnectionString)) { using (SqlCommand cmmnd = new SqlCommand("SELECT Date();", cnn)) { try { cnn.Open(); using (SqlDataReader rdr = cmmnd.ExecuteReader()) { if (rdr.Read()) { result = rdr[0].ToString(); } } } catch (Exception ex) { LogException(ex); } finally { cnn.Close(); } } }
