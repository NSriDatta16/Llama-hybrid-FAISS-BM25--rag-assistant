Well, you said it - leverage MouseEvent.MOUSE_MOVE to set a conditional in your looping routine. private var _isMoving:Boolean = false; stage.addEventListener(MouseEvent.MOUSE_MOVE, checkMouse); this.addEventListener(Event.ENTER_FRAME, doLoop); private function checkMouse(e:MouseEvent):void { _isMoving = true; } private function doLoop(e:Event):void { trace("moving =" + _isMoving); if(_isMoving) { // loop animation } _isMoving = false; }
