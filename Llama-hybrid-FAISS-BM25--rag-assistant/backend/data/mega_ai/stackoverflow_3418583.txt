Yes! Split structs apart! (just like when there's a huge function.) typedef struct { char *global_type; char *type; char *address; /* Meant to be a boolean which C doesn't have -_- */ unsigned preferred : 1; } vCardEmail; ... typedef struct { /* Custom Fields */ char *field_name; union { /* Single value */ int i; float f; double d; time_t t; vCardEmail email; vCardPhone phone; vCardAddress address; char *s; /* Multiple values */ int *is; float *fs; double *ds; time_t *ts; vCardEmail *emails; vCardPhone *phones; vCardAddress *addresses; char **ss; } field_value; } vCardCustomField; typedef struct { char *version; char **names; char *formatted_name; char *nickname; char *organisation; char *title; vCardEmail *emails; vCardPhone *phones; vCardAddress *addresses; time_t birthday; vCardCustomField *custom_fields; } vCard;
