You need to take the close click event handler out of the open click event handler $(function() { $('#dialog').dialog({bgiframe: true, dialogClass: 'change', resizable: false, draggable: false, modal: true, height: 334, width: 450, autoOpen: false, show: 'fade'}); $('a.open-trigger').click(function(){ $('#dialog').dialog('open'); }); $('a.close-trigger').click(function(){ alert("hello"); var myDialog = $('#dialog'); if (myDialog.dialog('isOpen')) myDialog.dialog('close'); }); });
