here's one way, if you can live with an objc++ translation: #import namespace { // ok, this storage should preferably be in a function/deferred static struct sa { NSString* const s; NSArray* const a; } r = { [[NSString alloc] initWithString:@"hello"], [[NSArray alloc] initWithObjects:@"w", @"o", @"r", @"l", @"d", @"= =", nil] }; } int main(int argc, const char* argv[]) { NSAutoreleasePool* pool = [[NSAutoreleasePool alloc] init]; NSLog(@"\n\n%@...\n\n...\n%@", r.s, r.a); [pool drain]; return 0; }
