Well, just have a function to populate the inputs within the placeMarker function: Example Link I've also noted that you are enabling dragging on your marker so I've added a drag event listener to populate the inputs accordingly (inside the placeMarker function). function placeMarker(location) { var marker = new google.maps.Marker({ position: location, map: map, draggable: true }); map.setCenter(location); var markerPosition = marker.getPosition(); populateInputs(markerPosition); google.maps.event.addListener(marker, "drag", function (mEvent) { populateInputs(mEvent.latLng); }); } function populateInputs(pos) { document.getElementById("t1").value=pos.lat() document.getElementById("t2").value=pos.lng(); }
