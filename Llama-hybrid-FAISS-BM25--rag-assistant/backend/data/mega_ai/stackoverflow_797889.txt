For oracle dialect. If u have problem with mapping database column type to java data type u can set it manually like that: .addScalar("integerFieldName", Hibernate.INTEGER) public class LookupCodeName { private String code; private String name; /*... getter-setters ... */ } public class someBL { public List returnSomeEntity() { SQLQuery sqlQuery = (SQLQuery)((HibernateSession)em).getHibernateSession() .createSQLQuery( "SELECT st.name as name, st.code as code FROM someTable st") .addScalar("code") .addScalar("name") .setResultTransformer(Transformers.aliasToBean(LookupCodeName.class)); } return (List )sqlQuery.list(); }
