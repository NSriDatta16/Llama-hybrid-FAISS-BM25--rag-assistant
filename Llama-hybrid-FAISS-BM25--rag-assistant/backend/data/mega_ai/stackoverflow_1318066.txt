All the answers so far seem to be using uninitialized variables (minX, etc) in comparisons. Here is a functional, elegant solution: var minX = anarray.reduce( function(a,b) { return {x : Math.min(a.x,b.x)};}).x; var minY = anarray.reduce( function(a,b) { return {y : Math.min(a.y,b.y)};}).y; var maxX = anarray.reduce( function(a,b) { return {x : Math.max(a.x,b.x)};}).x; var maxY = anarray.reduce( function(a,b) { return {y : Math.max(a.y,b.y)};}).y; Look here for an explanation of the reduce function and how to enable it in browsers that do not support it natively.
