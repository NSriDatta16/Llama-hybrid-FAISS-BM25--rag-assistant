This will return the average color for the image. static Color AverageColor(string fileName) { using (var bmp = new Bitmap(fileName)) { int width = bmp.Width; int height = bmp.Height; int red = 0; int green = 0; int blue = 0; int alpha = 0; for (int x = 0; x avg = c => c / (width * height); red = avg(red); green = avg(green); blue = avg(blue); alpha = avg(alpha); var color = Color.FromArgb(alpha, red, green, blue); return color; } }
