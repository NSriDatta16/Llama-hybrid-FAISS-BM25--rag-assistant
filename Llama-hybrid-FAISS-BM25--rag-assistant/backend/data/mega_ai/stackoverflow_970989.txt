Seeing your clarification, I think you want something like this. This will call the timerFired: method every PERIOD seconds while the finger is down. - (void)touchesBegan:(NSSet *)touches withEvent:(UIEvent *)event { // timer is an ivar... timer = [[NSTimer scheduledTimerWithTimeInterval:PERIOD target:self selector:@selector(timerFired:) userInfo:nil repeats:YES] retain]; } - (void)touchesEnded:(NSSet *)touches withEvent:(UIEvent *)event { [timer invalidate]; [timer release]; timer = nil; } - (void)timerFired:(NSTimer *)theTimer { // do stuff }
