select DATEADD(month, -n.number, getdate()) OneMonthFromThisDate, avg(datediff(dd, acquisitiontime, createdate)) AverageInThisMonth from users join master..spt_values n on n.type='P' and n.number between 1 and 24 where createdate > DATEADD(month, -n.number, getdate()) and createdate This puts the most recent first.
