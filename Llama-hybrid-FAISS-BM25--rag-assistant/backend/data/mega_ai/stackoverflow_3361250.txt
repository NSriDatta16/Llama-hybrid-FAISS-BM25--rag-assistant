Create file file targetver.h and include it in your stdafx.h . The targetver.h looks like this: #pragma once #ifndef WINVER #define WINVER 0x0700 #endif #ifndef _WIN32_WINNT #define _WIN32_WINNT #endif #ifndef _WIN32_WINDOWS #define _WIN32_WINDOWS 0x0510 #endif #ifndef _WIN32_IE #define _WIN32_IE 0x0700 #endif
