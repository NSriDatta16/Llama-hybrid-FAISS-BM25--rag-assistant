Modify the touchesBegan and touchesMoved methods to be like the following. float oldX, oldY; BOOL dragging; The touchesBegan:withEvent: method. - (void)touchesBegan:(NSSet *)touches withEvent:(UIEvent *)event { UITouch *touch = [[event allTouches] anyObject]; CGPoint touchLocation = [touch locationInView:self.view]; if (CGRectContainsPoint(window.frame, touchLocation)) { dragging = YES; oldX = touchLocation.x; oldY = touchLocation.y; } } The touchesMoved:withEvent: method. - (void)touchesMoved:(NSSet *)touches withEvent:(UIEvent *)event { UITouch *touch = [[event allTouches] anyObject]; CGPoint touchLocation = [touch locationInView:self.view]; if (dragging) { CGRect frame = window.frame; frame.origin.x = window.frame.origin.x + touchLocation.x - oldX; frame.origin.y = window.frame.origin.y + touchLocation.y - oldY; window.frame = frame; } } The touchesEnded:withEvent: method. - (void)touchesEnded:(NSSet *)touches withEvent:(UIEvent *)event { dragging = NO; }
