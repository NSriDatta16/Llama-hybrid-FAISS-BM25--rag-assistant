This might work for you. update dashboardtasks1 set tasktype = s.tasktype, [desc] = s.[desc], duedate = s.duedate, compdate = s.compdate, comments = s.comments, agent = s.agent, compby = s.compby, graceperiod = s.graceperiod from staggingtasks as s where s.tour=dashboardtasks1.tour and s.taskname=dashboardtasks1.taskname and s.deptdate=dashboardtasks1.deptdate and not exists (select * from dashboardtasks1 as d where s.tour=d.tour and s.taskname=d.taskname and s.deptdate=d.deptdate and s.duedate=d.duedate ) insert into dashboardtasks1 (tour, taskname, deptdate, tasktype, [desc], duedate, compdate, comments, agent, compby, graceperiod) select tour, taskname, deptdate, tasktype, [desc], duedate, compdate, comments, agent, compby, graceperiod from staggingtasks as s where not exists (select * from dashboardtasks1 as d where s.tour=d.tour and s.taskname=d.taskname and s.deptdate=d.deptdate and s.duedate=d.duedate )
