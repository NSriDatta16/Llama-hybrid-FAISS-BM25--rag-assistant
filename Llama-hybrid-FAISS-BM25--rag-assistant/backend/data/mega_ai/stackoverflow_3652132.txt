u can customize the pickerView and inherit their touch event method. - (void) touchesEnded: (NSSet *) touches withEvent: (UIEvent *) event { if (!self.dragging) { [self.nextResponder touchesEnded: touches withEvent:event]; } //here u can keep track of touch event(click ,double click) [super touchesEnded: touches withEvent: event]; }
