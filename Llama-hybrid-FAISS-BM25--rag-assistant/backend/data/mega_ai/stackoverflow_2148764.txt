function JS_Utils_BuildModal() { // Create parent div var objModal = document.createElement("div"); objModal.setAttribute('id', 'Modal'); document.body.appendChild(objModal); // Create modal div var objContent = document.createElement("div"); objContent.setAttribute("id", "Content"); objModal.appendChild(objContent); // Create sending data paragraph var objData = document.createElement("p"); objContent.appendChild(objData); } The appendChild() method will work on any DOM element. document.body is a DOM element for the document's body.
