Here is the above criteria with dynamic keywords string searchQuery = "wine pasta"; ICriteria query = Session.CreateCriteria(typeof(Recipe), "r") .CreateCriteria("Ingredients", "i", JoinType.InnerJoin) .SetFetchMode("Images", FetchMode.Join) .SetFetchMode("Comments", FetchMode.Join) .SetFetchMode("Ingredients", FetchMode.Join) .SetResultTransformer(new DistinctRootEntityResultTransformer()); var keywords = searchQuery.Split(' '); Disjunction keywordsCriteria = Restrictions.Disjunction(); foreach (var keyword in keywords) { keywordsCriteria.Add(Restrictions.Like("i.IngredientName", string.Format("%{0}%", keyword))); keywordsCriteria.Add(Restrictions.Like("r.RecipeTitle", string.Format("%{0}%", keyword))); } query.Add(keywordsCriteria); List result = query.List ();
