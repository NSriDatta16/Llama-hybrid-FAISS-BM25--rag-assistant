Is there any reason you don't use the XML feeds? require 'open-uri' require 'nokogiri' d = Nokogiri.XML(open 'http://services.tvrage.com/feeds/showinfo.php?sid=15614') name = d.search('//showname').text # => 'Chuck' day = d.search('//airday').text # => 'Monday' time = d.search('//airtime').text # => '20:00' net = d.search('//network[@country="US"]').text # => 'NBC' puts "#{name} airs #{day}s at #{time} on #{net}." # Chuck airs Mondays at 20:00 on NBC.
