You can subclass your UITableViewCell and override the touchesBegan:withEvent: method. You can then get a CGPoint of where the touch was placed. - (void)touchesBegan:(NSSet*)touches withEvent:(UIEvent*)event { UITouch* touch = touches.anyObject; CGPoint location = [touch locationInView:self]; if (CGRectContainsPoint(myTestRect, location)) { // Touched inside myTestRect, do whatever... } else { // Let the default implementation take over. [super touchesBegan:touches withEvent:event]; } } Andrew
