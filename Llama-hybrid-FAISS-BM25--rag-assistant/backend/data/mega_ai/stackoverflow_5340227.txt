You can use $('#new_average, #max_occupancy').keyup(Myfunction); $(".industry_picker").change(Myfunction); function Myfunction(){ factor = $(".industry_picker option:checked").attr("title"); average = $('#new_average').val(); if($('#new_average').val().indexOf("$") != -1){ average = average.substring(1); } if($("#max_occupancy").val() != ""){ max_occupancy = $("#max_occupancy").val(); max_total = ((average * factor) * max_occupancy) * 30; if(factor != ""){ $("#new_calc").val("$" + max_total + ".00"); } } }
