I just brewed this together: uses ShLwApi; function RelToAbs(const ARelPath, ABasePath: string): string; begin SetLength(Result, MAX_PATH); if PathCombine(@Result[1], PChar(IncludeTrailingPathDelimiter(ABasePath)), PChar(ARelPath)) = nil then Result := '' else SetLength(Result, StrLen(@Result[1])); end; Thanks to Andreas and David for calling my attention to the Shell Path Handling Functions .
