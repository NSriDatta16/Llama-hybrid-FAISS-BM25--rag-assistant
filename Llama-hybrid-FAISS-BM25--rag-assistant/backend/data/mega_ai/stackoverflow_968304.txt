If you can change the configuration file format, you can directly write your file as a Python file. config.py job = { 'Name' : "host.domain.com-foo", 'Client' : "host.domain.com-fd", 'JobDefs' : "DefaultJob", 'FileSet' : "local", 'Write Bootstrap' : "/etc/foo/host.domain.com-foo.bsr", 'Pool' : 'storage-disk1' } yourscript.py from config import job print job['Name']
