- (void)touchesBegan:(NSSet *)touches withEvent:(UIEvent *)_event { CGPoint pt = [[touches anyObject] locationInView:self]; startLocation = pt; } - (void)touchesMoved:(NSSet *)touches withEvent:(UIEvent *)_event { CGPoint pt = [[touches anyObject] previousLocationInView:self]; CGFloat dx = pt.x - startLocation.x; CGFloat dy = pt.y - startLocation.y; CGPoint newCenter = CGPointMake(self.center.x + dx, self.center.y + dy); self.center = newCenter; } - (void)setRotation:(float)rotation { self.transform = CGAffineTransformRotate(self.transform, degreesToRadians(rotation)); }
