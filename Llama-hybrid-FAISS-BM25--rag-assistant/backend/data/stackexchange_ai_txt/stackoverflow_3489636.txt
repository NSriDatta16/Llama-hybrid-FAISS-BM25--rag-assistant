This code snippet is pretty close to drag from "item" (just replace the "test.txt" string) to any explorer window (including ftp), when user is moving the item manually. private void Form1_MouseDown(object sender, MouseEventArgs e) { string[] files = new string[] { @"c:\temp\test.txt" }; this.DoDragDrop(new DataObject(DataFormats.FileDrop, files), DragDropEffects.Copy); }
