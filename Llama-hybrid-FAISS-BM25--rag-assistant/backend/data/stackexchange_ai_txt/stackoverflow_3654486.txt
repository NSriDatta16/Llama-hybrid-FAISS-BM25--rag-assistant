This solution is in my opinion more elegant, also more natural, it is easier to control, to correctly edit values of delays etc. I hope you'll like it. $(document).ready(function () { var elementsList = $('#slideshow p'); function animationFactory(i) { var element = $(elementsList[i % elementsList.length]); return function () { element.delay(200).fadeIn('slow').delay(800).fadeOut('slow', animationFactory(i + 1)); }; } animationFactory(0)(); });
