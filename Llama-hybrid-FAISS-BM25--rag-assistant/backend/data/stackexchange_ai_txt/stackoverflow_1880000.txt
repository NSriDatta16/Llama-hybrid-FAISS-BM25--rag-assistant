First define the following: #ifdef _M_IX86 typedef unsigned int READETYPE; #else typedef unsigned __int64 READETYPE; #endif extern "C" { READETYPE __readeflags(); } #pragma intrinsic(__readeflags) You can then check the eflags register as follows: if ( (__readeflags() & 0x800)) { overflow = 1; }
