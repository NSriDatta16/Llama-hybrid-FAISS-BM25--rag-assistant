try: inp = raw_input except NameError: inp = input _sum = 0.0 _num = 0 while True: val = float(inp("Enter difficulty rating (-1 to exit): ")) if val==-1.0: break else: _sum += val _num += 1 if _num: print "The average is {0:0.3f}".format(_sum/_num) else: print "No values, no average possible!"
