Certainly not the most elegant piece, but this should get you going regarding the closing tags: std::string indent(std::string str, const std::string &indent = " ") { std::string indent_ = std::string("\n"); size_t token = 0; while ((token = str.find_first_of("<>,", token)) != std::string::npos) { switch(str[token]) { case ' ': indent_.erase(indent_.size() - indent.size()); str.insert(token, indent_); } token += indent_.size() + 1; const size_t nw = str.find_first_not_of(" ", token); if(nw != std::string::npos) { str.erase(token, nw-token); } } return str; }
