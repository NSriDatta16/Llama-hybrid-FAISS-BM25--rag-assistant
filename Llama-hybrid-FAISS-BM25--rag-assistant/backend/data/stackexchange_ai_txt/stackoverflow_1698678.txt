If you can't find a more elegant solution, you can always download the file, do the string replacements, then use that as the WSDL. $cached_wsdl_file = './cached_wsdl.xml'; if (filemtime($cached_wsdl_file) > time() - 3600) { $wsdl = file_get_contents('http://server/service?wsdl'); $wsdl = str_replace('server:80', 'server', $wsdl); file_put_contents($cached_wsdl_file, $wsdl); } $client = new SoapClient($cached_wsdl_file);
