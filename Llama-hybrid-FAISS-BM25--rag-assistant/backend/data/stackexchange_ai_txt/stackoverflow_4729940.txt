You use NSDateComponents like so: - (NSDate *)logicalOneYearAgo:(NSDate *)from { NSCalendar *gregorian = [[[NSCalendar alloc] initWithCalendarIdentifier:NSGregorianCalendar] autorelease]; NSDateComponents *offsetComponents = [[[NSDateComponents alloc] init] autorelease]; [offsetComponents setYear:-1]; return [gregorian dateByAddingComponents:offsetComponents toDate:from options:0]; }
