Much more elegant than the code above. function swapOptions(obj,i,j){ var o = obj.options; var i_selected = o[i].selected; var j_selected = o[j].selected; var temp = new Option(o[i].text, o[i].value, o[i].defaultSelected, o[i].selected); var temp2= new Option(o[j].text, o[j].value, o[j].defaultSelected, o[j].selected); o[i] = temp2;o[j] = temp;o[i].selected = j_selected;o[j].selected = i_selected; }
