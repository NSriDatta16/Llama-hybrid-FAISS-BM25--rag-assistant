Besides using the Thumb.DragCompleted event you can also use both ValueChanged and Thumb.DragStarted , this way you donâ€™t lose functionality when the user modifies the value by pressing the arrow keys or by clicking on the slider bar. Xaml: Code behind: private bool dragStarted = false; private void Slider_DragCompleted(object sender, DragCompletedEventArgs e) { DoWork(((Slider)sender).Value); this.dragStarted = false; } private void Slider_DragStarted(object sender, DragStartedEventArgs e) { this.dragStarted = true; } private void Slider_ValueChanged( object sender, RoutedPropertyChangedEventArgs e) { if (!dragStarted) DoWork(e.NewValue); }
