$('.tabs').each( function() { var currentTab, ul = $(this); $(this).find('a').each( function(i) { var a = $(this).bind( 'click', function() { if (currentTab) { ul.find('a.active').removeClass('active'); $(currentTab).hide(); } currentTab = $(this).addClass('active') .attr('href'); $(currentTab).show().jScrollPane({ showArrows: true, verticalGutter: 10, animateScroll: true, verticalDragMinHeight: 34, verticalDragMaxHeight: 34 }); return false; } ); $(a.attr('href')).hide(); } ); } ).find('a:first').trigger('click'); Using ('a:first') instead of .first() worked for me.
