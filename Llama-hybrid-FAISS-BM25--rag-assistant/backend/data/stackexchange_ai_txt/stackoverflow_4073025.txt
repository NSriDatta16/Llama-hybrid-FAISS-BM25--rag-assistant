This piece of code will retrieve the bytes - byte[] output; using (IsolatedStorageFile isolatedStorageFile = IsolatedStorageFile.GetUserStoreForApplication()) { IsolatedStorageFileStream isolatedStorageFileStream = isolatedStorageFile.OpenFile("data.dat", FileMode.Open, FileAccess.Read); output = new byte[isolatedStorageFileStream.Length]; isolatedStorageFileStream.Read(output, 0, output.Length); isolatedStorageFileStream.Dispose(); }
