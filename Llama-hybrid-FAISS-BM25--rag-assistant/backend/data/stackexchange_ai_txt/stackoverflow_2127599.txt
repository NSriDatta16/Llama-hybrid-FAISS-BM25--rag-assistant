Here's a link to great tutorial related to LINQ mapping: http://www.codeproject.com/KB/linq/linqtutorial.aspx My final code is: [Table(Name="Role")] public class Role { private EntitySet _Users = new EntitySet (); [Column(IsPrimaryKey = true, IsDbGenerated = true, AutoSync = AutoSync.OnInsert)] public int RoleID { get; set; } [Column] public string Name { get; set; } [Association(Name = "FK_User_Role", Storage = "_Users", ThisKey = "RoleID", OtherKey = "RoleID")] public EntitySet Users { get{ return this._Users; } set{ this._Users.Assign(value);} } }
