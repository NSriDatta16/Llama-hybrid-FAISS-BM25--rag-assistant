You can try this: try { File sd = Environment.getExternalStorageDirectory(); if (sd.canWrite()) { String sourcePath= "/path/to/source/file.mp3"; String destinationPath= "/path/to/destination/file.mp3"; File source= new File(sd, sourcePath); File destination= new File(sd, destinationPath); if (source.exists()) { FileChannel src = new FileInputStream(source).getChannel(); FileChannel dst = new FileOutputStream(destination).getChannel(); dst.transferFrom(src, 0, src.size()); src.close(); dst.close(); } } catch (Exception e) {}
