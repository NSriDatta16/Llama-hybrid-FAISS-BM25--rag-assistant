You can get the length of the selected HTML as follows: function getSelectionHtml() { var sel, html = ""; if (window.getSelection) { sel = window.getSelection(); if (sel.rangeCount) { var frag = sel.getRangeAt(0).cloneContents(); var el = document.createElement("div"); el.appendChild(frag); html = el.innerHTML; } } else if (document.selection && document.selection.type == "Text") { html = document.selection.createRange().htmlText; } return html; } alert(getSelectionHtml().length);
