In the UIView, implement touchesBegan: - (void)touchesBegan:(NSSet *)touches withEvent:(UIEvent *)event { // assign a UITouch object to the current touch UITouch *touch = [[event allTouches] anyObject]; // if the view in which the touch is found is myScrollView // (assuming myScrollView is the UIScrollView and is a subview of the UIView) if ([touch view] == myScrollView) { // do stuff here } } A side note: make sure userInteractionEnabled is set to YES in the UIView.
