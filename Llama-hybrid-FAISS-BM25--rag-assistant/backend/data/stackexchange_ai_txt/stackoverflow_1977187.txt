Assuming your classes always start out as "hide-para", this should work: $(document).ready(function() { $("a.hide-para").click(function(){ if($(this).hasClass('hide-para')){ $('p').hide(); $(this).html('Show Paragraphs').removeClass('hide-para').addClass('show-para'); } else { $('p').show(); $(this).html('Hide Paragraphs').removeClass('show-para').addClass('hide-para'); } }); });
