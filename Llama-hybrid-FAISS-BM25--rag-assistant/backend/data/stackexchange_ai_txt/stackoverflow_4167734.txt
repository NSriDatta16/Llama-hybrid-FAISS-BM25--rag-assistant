jQuery UI draggable+sortable is just what you are seeking for. $( "#list" ).sortable({ revert: true }); $( "#toolbox li" ).draggable({ connectToSortable: "#list", helper: "clone", revert: "invalid" }); $( "ul, li" ).disableSelection();
