function accessArray(array $array, $keys) { if (!preg_match_all('~\[([^\]]+)\]~', $keys, $matches, PREG_PATTERN_ORDER)) { throw new InvalidArgumentException; } $keys = $matches[1]; $current = $array; foreach ($keys as $key) { $current = $current[$key]; } return $current; } echo accessArray( array( 1 => array( 2 => 'foo' ) ), '[1][2]' ); // echos 'foo' Would be even nicer, if you passed in array(1, 2) , instead of [1][2] : One could avoid the (fragile) preg_match_all parsing.
