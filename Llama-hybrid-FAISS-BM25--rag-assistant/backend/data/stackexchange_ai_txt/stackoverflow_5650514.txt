You can use it like so: //Initialise private ProgressDialog m_ProgressDialog = null; private Runnable myRunnable; //Make a runnable myRunnable= new Runnable(){ @Override public void run(){ Thread.sleep(5000); setContentView(R.layout.blaa); m_ProgressDialog.dismiss(); } }; //initialize your thread Thread thread = new Thread(null, myRunnable, "MagentoBackground"); thread.start(); //open up you dialog m_ProgressDialog = ProgressDialog.show(SelectStation.this, "", "Loading, please wait...", true);
