I found a solution for my Problem, based on this tutorial : private void List_PreviewMouseLeftButtonDown(object sender, MouseButtonEventArgs e) { // Store the mouse position startPoint = e.GetPosition(null); } private void List_MouseMove(object sender, MouseEventArgs e) { // Get the current mouse position Point mousePos = e.GetPosition(null); Vector diff = startPoint - mousePos; if (e.LeftButton == MouseButtonState.Pressed && Math.Abs(diff.X) > SystemParameters.MinimumHorizontalDragDistance && Math.Abs(diff.Y) > SystemParameters.MinimumVerticalDragDistance) { if (listView1.SelectedItems.Count == 0) { return; } string[] files = GetSelection(); string dataFormat = DataFormats.FileDrop; DataObject dataObject = new DataObject(dataFormat, files); DragDrop.DoDragDrop(listView1, dataObject, DragDropEffects.Copy); } }
