public class Wrapper { public List GetInfo (Array data, EdmCmd edmCmd) { List info = new List (); switch (edmCmd.meCmdType) { case EdmCmdType.EdmCmd_PostAdd: foreach (EdmCmdData item in data) { info.Add(new EdmCmdPostAddInfo(item.mlObjectID1, item.mlObjectID2, item.mbsStrData1, item.mlLongData1)); } break; default: break; } return info.OfType ().ToList(); } } Assuming all of your new ECmd* structures are truly structs and not classes with a hierarchy, you can create a List of objects and cast this after you've processed the passed in array.
