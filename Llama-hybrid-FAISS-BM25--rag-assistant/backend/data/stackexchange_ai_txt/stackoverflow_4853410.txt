use WWW::Mechanize; my $url = 'marinetraffic2.aegean.gr/ais/getkml.aspx'; my $local_file_name = 'getkml.aspx'; my $mech = WWW::Mechanize->new; $mech->get( $url, ":content_file" => $local_file_name ); This in fact wraps around the LWP::UserAgent->get .
