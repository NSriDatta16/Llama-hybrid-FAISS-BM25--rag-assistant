So it turns out that I should have been setting the Margin instead of using TranslateTransform: def startDrag(self, sender, event): self.root.drag.Visibility = Visibility.Visible self.root.dragText.Text = sender.Text self.root.CaptureMouse() self._captured = True self.root.MouseLeftButtonUp += self.stopDrag self.root.MouseLeave += self.stopDrag self.onDrag(sender, event) def onDrag(self, sender, event): if self._captured: position = event.GetPosition(self.root.layout_root) self.root.drag.Margin = Thickness(position.X, position.Y, 0, 0) self.root.drag.UpdateLayout()
