(defun remove-nil-recursively (x) (if (listp x) (mapcar #'remove-nil-recursively (remove nil x)) x)) Works for your example: [1]> (remove-nil-recursively '((state L L L L) NIL (state L L R L) NIL)) ((STATE L L L L) (STATE L L R L)) And with nested lists: [2]> (remove-nil-recursively '(NIL (state L L nil R L) NIL)) ((STATE L L R L)) But watch out: [3]> (remove-nil-recursively '(NIL (state L L (nil) R L) NIL)) ((STATE L L NIL R L))
