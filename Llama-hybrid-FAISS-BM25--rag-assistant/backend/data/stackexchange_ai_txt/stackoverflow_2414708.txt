If you have a byte array that represents a file that WPF can decode (bmp, jpg, gif, png, tif, ico), you can do the following: BitmapSource LoadImage(Byte[] imageData) { using (MemoryStream ms = new MemoryStream(imageData)) { var decoder = BitmapDecoder.Create(ms, BitmapCreateOptions.PreservePixelFormat, BitmapCacheOption.OnLoad); return decoder.Frames[0]; } } Likewise, to convert it back, you can do the following: byte[] SaveImage(BitmapSource bitmap) { using (MemoryStream ms = new MemoryStream()) { var encoder = new BmpBitmapEncoder(); encoder.Frames.Add(BitmapFrame.Create(bitmap)); encoder.Save(ms); return ms.GetBuffer(); } }
