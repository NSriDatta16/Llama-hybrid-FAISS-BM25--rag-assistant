You could use jQuery data to store the appropriate function object right with the draggable elements: // declare the appropriate function for each element $("select elements that need function1").data("dragFunction", function() { // whatever }); $("select elements that need function2").data("dragFunction", function() { // whatever }); // and then... just execute whatever function is stored in "dragFunction" $("div.element").draggable({ drag: function() { return $(this).data("dragFunction")(); } });
