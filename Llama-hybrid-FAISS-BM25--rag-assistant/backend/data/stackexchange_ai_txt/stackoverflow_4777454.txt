If you're using Node.js you could leverage the CommonJS module system. math.js (your library) exports.add = function() { for (var i = arguments.length; i--;) { sum += arguments[i]; } return sum; }; program.js (someone using it...) var MyMath = require('math'); console.log(MyMath.add(1, 2)); // 3 // ... in different ways var add = require('math').add; console.log(add(1, 2)); // 3
