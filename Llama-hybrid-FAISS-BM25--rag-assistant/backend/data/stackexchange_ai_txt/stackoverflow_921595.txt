-- This is a very, very inefficient way of doing it but should be OK for -- small tables. It uses an auxiliary table of numbers as per Itzik Ben-Gan and simply -- looks for characters with bit 7 set. SELECT * FROM yourTable as t WHERE EXISTS ( SELECT * FROM msdb..Nums as NaturalNumbers WHERE NaturalNumbers.n 127)
