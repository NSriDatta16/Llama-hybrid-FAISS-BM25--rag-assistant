If it's just writing each paragraph or sentence in the reverse order than it appears: Sub Test() Dim currentDocument As Document Set currentDocument = ActiveDocument Dim sourceDocument As Document Set sourceDocument = Documents.Add("c:\words.txt") Dim i As Long For i = sourceDocument.Paragraphs.Count To 1 Step -1 currentDocument.Range.InsertAfter sourceDocument.Paragraphs(i).Range.Text DoEvents Next sourceDocument.Close wdDoNotSaveChanges End Sub
