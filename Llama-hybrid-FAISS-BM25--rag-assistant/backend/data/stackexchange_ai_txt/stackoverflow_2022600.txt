Only for the sake of OOP, don't use classes. If you really want to implement this functionality, use this class: class ArrayClass { protected $storage = array(); public function push($content) { $this->storage[] = $content; return $this; } public function get($index) { if (isset($this->storage[$index])) { return $this->storage[$index]; } else { //throw an error or return false; } } }
