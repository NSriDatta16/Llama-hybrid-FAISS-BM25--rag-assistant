has_key in python is O(1). Insertion and retrieval from a hash is also O(1). Loops through n items twice, so O(n). def unique(list): s = {} output = [] for x in list: count = 1 if(s.has_key(x)): count = s[x] + 1 s[x] = count for x in list: count = s[x] if(count > 0): s[x] = 0 output.append(x) return output
