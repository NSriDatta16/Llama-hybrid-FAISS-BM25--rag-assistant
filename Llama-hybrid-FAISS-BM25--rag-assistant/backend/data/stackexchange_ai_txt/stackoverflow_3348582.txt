Try to look for it in the shared HTTP cookies storage: for (NSHTTPCookie *cookie in [[NSHTTPCookieStorage sharedHTTPCookieStorage] cookies]) { NSLog(@"name: '%@'\n", [cookie name]); NSLog(@"value: '%@'\n", [cookie value]); NSLog(@"domain: '%@'\n", [cookie domain]); NSLog(@"path: '%@'\n", [cookie path]); } or if working in Swift: for cookie in HTTPCookieStorage.shared.cookies! { NSLog("name: \(cookie.name)") NSLog("value: \(cookie.value)") NSLog("domain: \(cookie.name)") NSLog("path: \(cookie.path)") }
