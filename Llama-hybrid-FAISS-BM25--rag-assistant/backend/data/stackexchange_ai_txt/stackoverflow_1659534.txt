As I understand, it's impossible to implement iframe printing without opening new window. My print function: if ($.browser.opera || (/chrome/.test(navigator.userAgent.toLowerCase()))) { var href = contentWindow.location.href; href = href.indexOf("?") > -1 ? href + "&print=1" : href + "?print=1"; var printWindow = window.open(href, "printWindow", "scrollbars=yes"); printWindow.focus(); } else { contentWindow.focus(); contentWindow.print(); } Also I added the following code to the end of the body (when print==1): function invokePrint() { if (document.readyState && document.readyState!='complete') setTimeout(function() { invokePrint(); }, 50); else if (document.body && document.body.innerHTML=='false') setTimeout(function() { invokePrint(); }, 50); else { focus(); print(); } } invokePrint();
