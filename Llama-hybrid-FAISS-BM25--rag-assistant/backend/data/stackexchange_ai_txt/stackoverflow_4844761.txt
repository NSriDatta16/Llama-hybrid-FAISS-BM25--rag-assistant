Put the conditional define in the header files of your library (if it's not already there) and place a compiler-search directive in the appropriate lib-file along with it (when it's included by clients). #ifdef MYMATH_USE_DOUBLE typedef double Real; $ifndef _LIB // only for clients #pragma comment( lib, "double_lib" ) // double_lib name of the library. #endif #else typedef float Real; $ifndef _LIB #pragma comment( lib, "float_lib" ) #endif #endif
