You can do this easily leveraging SQL Server's XML functions: WITH xmlData (xml_owner_id,parecelID) AS ( /* make into xml */ SELECT cast(' '+replace(owner_id,',',' ')+' ' as XML) AS xml_owner_id, parecelID FROM ImportData ) SELECT x.value('.','int') AS owner_id, parecelID /* split up */ FROM xmlData CROSS APPLY xmlData.xml_owner_id.nodes('//x') AS func(x)
