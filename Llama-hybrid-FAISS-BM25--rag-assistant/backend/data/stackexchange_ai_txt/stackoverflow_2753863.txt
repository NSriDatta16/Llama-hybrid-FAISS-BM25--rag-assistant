Here's a condensed version. Because you're return ing inside each if statement, you can eliminate your else s. Also, I eliminated the need for using exceptions for flow. public static string GetUa(HttpRequest hr) { string visitorBrowser = hr.UserAgent; string originalBrowser = hr.ServerVariables["X-OperaMini-Phone-UA"]; string anotherOriginalBrowser = hr.ServerVariables["X-Device-User-Agent"]; //novarra if (string.IsNullOrEmpty(visitorBrowser)) return "No UA Found"; if (!String.IsNullOrEmpty(originalBrowser)) return "OPERAMINI " + originalBrowser; if (!String.IsNullOrEmpty(anotherOriginalBrowser)) return "NOVARRA " + anotherOriginalBrowser; return visitorBrowser; }
