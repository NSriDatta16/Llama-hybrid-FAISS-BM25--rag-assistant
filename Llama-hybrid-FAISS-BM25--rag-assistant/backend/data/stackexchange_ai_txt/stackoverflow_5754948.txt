well, somewhat like this? def simple_format( text ) return text if ( text =~ /( )/ ) # return text unchanged start_tag = ' ' text = text.to_s.dup text.gsub!(/\r\n?/, "\n") # \r\n and \r => \n text.gsub!(/\n\n+/, " \n\n#{start_tag}") # 2+ newline => paragraph text.gsub!(/([^\n]\n)(?=[^\n])/, '\1 ') # 1 newline => br text.insert 0, start_tag text " end
