You can do this without reflection (both int and double are supported): public static double Average(this IEnumerable surveys, Func selector) { return surveys.Average(selector); } public static double Average(this IEnumerable surveys, Func selector) { return surveys.Average(selector); } Usage: var average1 = surveys.Average(survey => survey.Property1); var average2 = surveys.Average(survey => survey.Property2);
