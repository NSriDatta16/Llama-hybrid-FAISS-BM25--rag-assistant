SELECT equip.* FROM equip e WHERE equipid IN ( SELECT equipid FROM equip_equipcat ec JOIN equipcat c ON c.equipcatid = ee.equipcatid WHERE equipcat LIKE '%rake%' OR equipcatkeywords LIKE '%rake%' ) AND ( equipdesc LIKE '%rake%' OR keywords LIKE '%rake%' ) AND hideyn = 0 ORDER BY equipdesc
