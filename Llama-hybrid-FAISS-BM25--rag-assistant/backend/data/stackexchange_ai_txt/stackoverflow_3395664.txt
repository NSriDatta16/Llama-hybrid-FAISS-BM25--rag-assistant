from naugtur suggestion i found this good working solution: dragStart:function(){ var divt=document.createElement('div'); $(divt).attr('id','tempdragdiv'); divt.style.height='93%'; divt.style.width='100%'; $(divt).css('position','absolute').css('left','0') .css('top','0'); $(div).append(divt);//div where is iframe }, dragStop:function(){ $('#tempdragdiv').remove(); }
