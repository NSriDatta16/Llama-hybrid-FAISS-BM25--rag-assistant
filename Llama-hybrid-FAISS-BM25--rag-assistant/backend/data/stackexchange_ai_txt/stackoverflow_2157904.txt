You can use an enum to conceal this: class FetchableThing {} class Person extends FetchableThing {} class Organisation extends FetchableThing {} enum DAO { Person(Person.class), Organisation(Organisation.class); private Class myClass; private DAO(Class myClass) { this.myClass=myClass;} public delete(String id) { PersistenceManager pm = PMF.get().getPersistenceManager(); try { T p = (T) pm.getObjectById(myClass, id); pm.deletePersistent(p); } finally { pm.close(); } } } DAO.Person.delete("1234"); DAO.Organsation.delete("1234");
