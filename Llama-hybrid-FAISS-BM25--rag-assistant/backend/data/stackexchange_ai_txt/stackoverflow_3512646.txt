// Define a random integer function function random( n ){ return Math.floor( Math.random() * n ); } // Define some variables, hide all images and show just one of them. var transition_time = 500; var waiting_time = 500; var images = $('div#Slogan img'); var n = images.length; var current = random( n ); images.hide(); images.eq( current ).show(); // Periodically, we fadeOut the current image and fadeIn a random one var interval_id = setInterval( function(){ images.eq( current ).fadeOut( transition_time, function(){ current = random( n ); images.eq( current ).fadeIn( transition_time ); } ); }, 2 * transition_time + waiting_time); // You can then stop the effect with: // clearInterval( interval_id );
