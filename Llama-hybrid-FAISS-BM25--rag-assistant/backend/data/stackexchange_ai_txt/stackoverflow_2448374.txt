Because if you want to handle rjs using jQuery there are two way to do that: $("#droppable_ ").droppable({ drop: function(event, ui) { div_ids = ui.draggable.attr('id').split('_'); $.post('/tasks/' + div_ids[div_ids.length - 1], {'task[bin_id]' : ' ', _method:'PUT'}, function(data) { eval(data) #run js ajax returned }); } }); or $("#droppable_ ").droppable({ drop: function(event, ui) { div_ids = ui.draggable.attr('id').split('_'); $.ajax({ url:'/tasks/' + div_ids[div_ids.length - 1], data: {'task[bin_id]' : ' ', _method:'PUT'}, dataType: "script", type: "POST" }); } });
