This behavior is a little strange, but a handy shortcut in case of deep hierarchy! You can implement following UITabBarControllerDelegate methods to disable this system wide shortcut: #pragma mark - #pragma mark UITabBarControllerDelegate - (BOOL)tabBarController:(UITabBarController *)tbc shouldSelectViewController:(UIViewController *)vc { UIViewController *tbSelectedController = tbc.selectedViewController; if ([tbSelectedController isEqual:vc]) { return NO; } return YES; }
