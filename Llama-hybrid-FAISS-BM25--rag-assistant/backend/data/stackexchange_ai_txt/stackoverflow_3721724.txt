Ok, after playing around, I came up with this which works Protected conString As String = ConfigurationManager.AppSettings("sqldirectory") Dim cnn As New SqlConnection(conString) Dim cmd As New SqlCommand("select message from [database].[dbo].[table]", cnn) Dim dr As SqlDataReader Dim msg As String Try cnn.Open() dr = cmd.ExecuteReader(System.Data.CommandBehavior.CloseConnection) While dr.Read() msg = dr("message") lblMsg.Text = msg End While Catch ex As Exception dr = Nothing Finally cnn.Close() End Try
