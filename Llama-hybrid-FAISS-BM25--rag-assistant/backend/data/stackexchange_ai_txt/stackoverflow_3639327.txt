Open File: String pFile = "input.txt"; BufferedReader in; try { in = new BufferedReader(new FileReader(pFile)); } catch (FileNotFoundException ex) { System.out.println("File Not Found"); return; } Declare the HashMap: HashMap map = new HashMap (); // Read lines, Parse them and add to Hash-Map: String line; //loop till end of file while((line = in.readLine()) != null) { //remove spaces from front and back... line = line.trim(); //ignore if line is empty if(line.isEmpty()) continue; //split line into array String[] p=line.split(" "); //now according to your input-file's first line //p[0]="110001" //p[1]="commercial" //p[2]="500000.00" //p[3]="101" //so do whatever you want. String agentId = p[3]; //Parse the string to double. double property = Double.valueOf(p[2]); //Check if hash already contains key. if(!map.containsKey(agentId)) //if not then add the new key with new value map.put(agentId, property); else //if key is present, add the value.. map.put(agentId, map.get(agentId) + property); }
