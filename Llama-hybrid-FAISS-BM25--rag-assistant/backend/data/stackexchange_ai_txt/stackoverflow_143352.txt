Here is a fragment of code that I'm using. UIColor *mainTextColor = [UIColor whiteColor]; [mainTextColor set]; drawTextLjust(@"Sample Text", 8, 50, 185, 18, 16); And: static void drawTextLjust(NSString* text, CGFloat y, CGFloat left, CGFloat right, int maxFontSize, int minFontSize) { CGPoint point = CGPointMake(left, y); UIFont *font = [UIFont systemFontOfSize:maxFontSize]; [text drawAtPoint:point forWidth:right - left withFont:font minFontSize:minFontSize actualFontSize:NULL lineBreakMode:UILineBreakModeTailTruncation baselineAdjustment:UIBaselineAdjustmentAlignBaselines]; }
