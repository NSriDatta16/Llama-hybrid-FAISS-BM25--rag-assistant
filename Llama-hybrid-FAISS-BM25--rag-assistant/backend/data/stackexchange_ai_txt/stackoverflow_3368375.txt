Assuming we're working with strings here, here's another "elegant" solution using IndexOf() . public IQueryable GetFacilityItemRootByDescription(string description) { return this.ObjectContext.FACILITY_ITEM .Where(fi => fi.DESCRIPTION .IndexOf(description, StringComparison.OrdinalIgnoreCase) != -1); }
