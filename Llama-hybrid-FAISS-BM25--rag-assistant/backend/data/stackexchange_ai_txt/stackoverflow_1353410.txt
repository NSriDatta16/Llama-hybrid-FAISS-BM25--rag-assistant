OK, I got the dirty solution... $(document).ready(function () { src = null; options = { revert:true, axis: 'x', opacity: 0.8, start: function() { src = $(this).parent(); } }; $(".item").draggable(options); $(".container").droppable({ drop: function(event, ui) { src.append( $('.item', this).remove().clone() .removeClass().addClass("item") .css({"left": '', "opacity": ''}) .draggable(options) ); $(this).append( ui.draggable.remove().clone() .removeClass().addClass("item") .css({"left": '', "opacity": ''}) .draggable(options) ); } }); }); Hope, someone could improve this... :) Cheer...
