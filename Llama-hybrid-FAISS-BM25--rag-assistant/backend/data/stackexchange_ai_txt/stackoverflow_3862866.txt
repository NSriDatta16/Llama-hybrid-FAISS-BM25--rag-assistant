Here is a real/pseudo code version - hope it helps... -(void) touchesBegan:(NSSet *)touches withEvent:(UIEvent *)even{ UITouch *touch = [touches anyObject]; if(touch.x = 240){ globalTouchRight = touch; if(!globalTouchLeft){ startRightTimer; //again I don't know what your timere/variable is } } } -(void)touchesEnded:(NSSet *)touches withEvent:(UIEvent *)event{ UITouch *touch = [touches anyObject]; if(touch == globalLeftTouch){ stopLeftTimer globalLeftTouch = nil; if(globalRightTouch){ startRightTimer; } } if(touch == globalTouchRight){ stopRightTimer globalRightTouch = nil; if(globalLeftTouch){ startLeftTimer; } } } cheers, Michael O'Brien
