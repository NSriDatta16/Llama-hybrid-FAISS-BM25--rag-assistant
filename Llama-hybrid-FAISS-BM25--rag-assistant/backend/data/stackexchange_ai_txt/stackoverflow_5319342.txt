FieldInfo fi = dataGridView1.GetType().GetField( "m_sbVert", BindingFlags.NonPublic | BindingFlags.GetField | BindingFlags.Instance ); ( ( VScrollBar ) fi.GetValue( dataGridView1 ) ).Width = 50; fi = ultraGrid1.GetType().GetField( "m_sbHorz", BindingFlags.NonPublic | BindingFlags.GetField | BindingFlags.Instance ); ( ( HScrollBar ) fi.GetValue( ultraGrid1 ) ).Height = 0; Where 50 is the vertical scrollbar width and 0 is the horizontal scrollbar height. using System.Reflection; all at the beginning of your form. It works for WinCE 5.0 .
