There's System.Collections.ObjectModel. KeyedCollection , which derives from Collection . Retrieval is O(1) . class IndexableDictionary : KeyedCollection { Dictionary keys = new Dictionary (); protected override string GetKeyForItem(TItem item) { return keys[item];} public void Add(string key, TItem item) { keys[item] = key; this.Add(item); } }
