Though not as elegant as the LISP code, you could create a function dynamically that acts like an operator (on numbers), but it's not. function op(o) { return new Function("a", "b", "return a " + o + " b"); } function aPlusAbsB(a, b) { return (b > 0 ? op('+') : op('-'))(a, b); } Additionally, we can hide the complexity of generating these inside an if wrapper, but that's the closest I can get :) function is(expr, op1, op2) { return expr ? op(op1) : op(op2); } function aPlusAbsB(a, b) { return (is(b > 0, '+', '-')(a, b)); }
