CriteriaBuilder criteriaBuilder = em.getCriteriaBuilder(); Metamodel m = em.getMetamodel(); EntityType WS = m.entity(Workspace.class); CriteriaQuery criteriaQuery = criteriaBuilder.createQuery(Workspace.class); Root from = criteriaQuery.from(Workspace.class); Predicate condition = criteriaBuilder.equal(from.get(WS.username), username); http://download.oracle.com/javaee/6/tutorial/doc/gjivm.html
