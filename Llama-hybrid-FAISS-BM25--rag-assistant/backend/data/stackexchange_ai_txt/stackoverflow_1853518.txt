select u.user_id, (select count(user_id) from votes v where v.user_id = u.user_id) as num_votes, (select sum(value) from votes v where v.user_id = u.user_id) as sum_votes from users u # if you want: order by ... EDIT: The subselects are not the most elegant solution, and not really necessary. So here's an outer join version: select u.user_id, count(v.user_id) as num_votes, sum(v.value) as sum_votes from users u left outer join votes v on u.user_id = v.user_id group by u.user_id
