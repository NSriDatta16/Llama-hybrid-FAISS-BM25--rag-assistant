I found a solution: JNIPATH := $(call my-dir) LOCAL_PATH := $(JNIPATH) include $(call all-subdir-makefiles) LOCAL_PATH := $(JNIPATH) include $(CLEAR_VARS) LOCAL_MODULE := gaitfuncs LOCAL_SRC_FILES := gaitfuncs.c LOCAL_STATIC_LIBRARIES := htk include $(BUILD_SHARED_LIBRARY) Calling the CLEAR_VARS function before calling the subdir-makefiles function wasn't exactly elegant ;)
