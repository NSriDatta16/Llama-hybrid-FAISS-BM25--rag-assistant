A simple but fast implementation: private int[] values = new int [10]; // all 0's initially private int sum = 0; private int pos = 0; public void AddValue (int v) { sum -= values[pos]; // only need the array to subtract old value sum += v; values[pos] = v; pos = (pos + 1) % values.Length; } public int Average => sum / values.Length;
