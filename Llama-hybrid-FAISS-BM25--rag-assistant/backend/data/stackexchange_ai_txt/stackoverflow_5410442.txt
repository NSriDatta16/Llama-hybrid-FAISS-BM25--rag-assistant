Try adding a hash parameter - the browser will see this as a new request, and won't use its cache, but everything beyond the hash gets stripped out before the request is made. Give this a whirl and see it it works: var callback = '#call' + new Date().getTime(); $.ajax({ "url": "http://myserver.com/echoDate" + callback, "dataType": 'jsonp', "jsonp": "cb", "jsonpCallback": plcb, "success": function(resp){ $("#pants").html($("#pants").html() + resp + " "); } });
