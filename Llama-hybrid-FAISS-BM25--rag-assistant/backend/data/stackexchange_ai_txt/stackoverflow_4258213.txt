The variable this is person in your case, also firstName and lastName are defined as local variables, but not as properties of the person , so you can just access them within the anonymous function by names: var secretAgent = (function(){ var person = {}, firstName = "James", lastName = "Bond"; person.WhoAreYou = function() { alert("My name is " + lastName + ", " + firstName + " " +lastName); }; return person; })(); secretAgent.WhoAreYou();
