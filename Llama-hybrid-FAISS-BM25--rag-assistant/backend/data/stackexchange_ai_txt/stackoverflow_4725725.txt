SELECT n.cid, n.Title, GROUP_CONCAT(DISTINCT co.name ORDER BY co.sort) as organizers, GROUP_CONCAT(cp.name ORDER BY cp.sort) FROM Content n LEFT JOIN Organizers co ON n.cid = co.cid LEFT JOIN Participants cp ON n.cid = cp.cid GROUP BY n.cid
