If "exec" is activated /** * Return the current server load * It needs "exec" activated * * e.g. * 15:06:37 up 10 days, 5:59, 12 users, load average: 1.40, 1.45, 1.33 * returns * float(1.40) */ public function getServerLoad() { preg_match('#(\d\.\d+),[\d\s\.]+,[\d\s\.]+$#', exec("uptime"), $m); return (float)$m[1]; }
