i want if i change dropdown of searchbox-purchase form, only dropdown city in that particular form should be changed. Your code is fine: $("form[name='searchbox-purchase'] select[name='city']").change(function(){ }); You just specify same for all fours forms and it will affect only dropdown of those forms: $("form[name='searchbox-purchase'] select[name='city']").change(function(){ // this will affect city dropdown inside searchbox-purchase }); $("form[name='searchbox-rental'] select[name='city']").change(function(){ // this will affect city dropdown inside searchbox-rental }); $("form[name='searchbox-agents'] select[name='city']").change(function(){ // this will affect city dropdown inside searchbox-agents }); $("form[name='searchbox-developments'] select[name='city']").change(function(){ // this will affect city dropdown inside searchbox-developments });
