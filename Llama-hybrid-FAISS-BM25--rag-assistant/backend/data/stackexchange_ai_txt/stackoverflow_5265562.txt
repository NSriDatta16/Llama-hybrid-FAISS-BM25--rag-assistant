I would suggest you leverage the MVC Framework for this. You can then easily call Controller's Action methods via client-side AJAX calls. Here's a simplified example: // jQuery AJAX call. function getContacts() { $.ajax({ type: 'get', url: '/Contacts/GetContacts', dataType: 'json', success: function (response) { var contacts = response; }, error: function (e) { alert('oops!'); } }); } // Server-side. public class ContactsController : Controller { [HttpGet] public JsonResult GetContacts() { JsonResult result = new JsonResult { JsonRequestBehavior = JsonRequestBehavior.AllowGet }; List contacts = DataAccess.GetContacts(); result.Data = contacts; return result; } }
