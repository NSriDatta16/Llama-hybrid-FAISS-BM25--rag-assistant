You need to pass an instantiated class as a parameter of the DefaultIfEmpty . class Program { static void Main(string[] args) { var lTest = new List (); var s = lTest.DefaultIfEmpty(new Test() { i = 1, name = "testing" }).First().name; Console.WriteLine(s); Console.ReadLine(); } } public class Test { public int i { get; set; } public string name { get; set; } } To add to it and make it a bit more elegant (IMO) add a default constructor: class Program { static void Main(string[] args) { var lTest = new List (); var s = lTest.DefaultIfEmpty(new Test()).First().name; Console.WriteLine(s); Console.ReadLine(); } } public class Test { public int i { get; set; } public string name { get; set; } public Test() { i = 2; name = "testing2"; } }
