select sm.NAME, sm.AverageScore, sm.best_score, s.SUBJECT from ( SELECT NAME, Avg(SCORE) AS AverageScore, MAX(SCORE) AS best_score FROM Scoretable GROUP BY NAME ) sm inner join Scoretable s on sm.NAME = s.NAME and sm.best_score = s.SCORE
