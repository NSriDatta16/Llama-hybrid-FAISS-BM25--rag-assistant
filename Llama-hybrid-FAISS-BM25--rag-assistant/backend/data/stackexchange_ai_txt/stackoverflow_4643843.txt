Use Apache Commons StringUtils org.apache.commons.lang.StringUtils to get the count. String str="my name is neo and my other name is also neo because I am neo"; // Make a unique list (java.util.Set) of words. Set stSet = new HashSet (Arrays.asList(str.split(" "))); int sz = stSet.size(); int[] counts = new int[sz]; Map matches = new HashMap (sz); int i = 0; for (String s : stSet) { // saves the individual word count in a sortable array. counts[i] = StringUtils.countMatches(str,s)); // saves the word count and the word in a HashMap for easy retrieval. matches.put(counts[i],s); i++; } Arrays.sort(counts); int max = counts.length - 1; System.out.println("The the word with the most occurrances is: "+matches.get(counts[max])+", the number of occurrances is: "+counts[max]);
