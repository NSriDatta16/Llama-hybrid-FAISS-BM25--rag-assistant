Well, with a small change, the code posted by Tim Down worked for me : if (typeof Range.prototype.createContextualFragment == "undefined") { Range.prototype.createContextualFragment = function (html) { var doc = window.document; var container = doc.createElement("div"); container.innerHTML = html; var frag = doc.createDocumentFragment(), n; while ((n = container.firstChild)) { frag.appendChild(n); } return frag; }; }
