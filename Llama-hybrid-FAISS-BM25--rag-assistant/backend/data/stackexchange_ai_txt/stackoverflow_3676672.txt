Since that xamlString contains a FlowDocument, XamlReader.Load will return a FlowDocument object rather than a Paragraph. If you want to handle various kinds of content, you could do something like this: XmlReader xmlReader = XmlReader.Create(new StringReader(xamlString)); object template1 = XamlReader.Load(xmlReader); FlowDocument newFL; if (template1 is FlowDocument) { newFL = (FlowDocument)template1; } else { newFL = new FlowDocument(); if (template1 is Block) { newFL.Blocks.Add((Block)template1); } else if (template1 is Inline) { newFL.Blocks.Add(new Paragraph((Inline)template1)); } else if (template1 is UIElement) { newFL.Blocks.Add(new BlockUIContainer((UIElement)template1)); } else { // Handle unexpected object here } } RichTextBox1.Document = newFL;
