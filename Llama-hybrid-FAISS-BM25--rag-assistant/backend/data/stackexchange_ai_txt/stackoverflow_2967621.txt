lowest = arr.min highest = arr.max total = arr.inject(:+) len = arr.length average = total.to_f / len # to_f so we don't get an integer result sorted = arr.sort median = len % 2 == 1 ? sorted[len/2] : (sorted[len/2 - 1] + sorted[len/2]).to_f / 2
