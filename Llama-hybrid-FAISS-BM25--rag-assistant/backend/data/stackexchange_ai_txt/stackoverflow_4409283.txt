'/Maildir/') ); Processing mail code: foreach ($mailbox as $id => $message) { $uniqueid = $mailbox->getUniqueId($id); /* ... mail processing code ... */ // mark as read $currentmessageid = $mailbox->getNumberByUniqueId($uniqueid); $mailbox->setFlags($currentmessageid, array(Zend_Mail_Storage::FLAG_SEEN)); // or uncomment to delete it //$mailbox->removeMessage($currentmessageid); }
