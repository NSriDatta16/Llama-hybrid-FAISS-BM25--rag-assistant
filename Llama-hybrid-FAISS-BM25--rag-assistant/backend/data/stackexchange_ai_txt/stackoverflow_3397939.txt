Just a spoon full of reflection helps the medicine go down ;-) private static bool IsLazyLoadedProperty(PropertyInfo property) { var column = property.GetCustomAttributes(typeof(ColumnAttribute), true)[0] as ColumnAttribute; var field = property.DeclaringType.GetField(column.Storage, BindingFlags.Instance | BindingFlags.NonPublic); if (!field.FieldType.IsGenericType) { return false; } return field.FieldType.GetGenericTypeDefinition() == typeof(Link<>); }
