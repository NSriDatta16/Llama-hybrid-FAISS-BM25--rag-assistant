Try this. Add this function to ChildBrowserCommand.m -(void) deleteCookies:(NSMutableArray*)arguments withDict:(NSMutableDictionary*)options{ NSHTTPCookieStorage *cookieStorage = [NSHTTPCookieStorage sharedHTTPCookieStorage]; for (NSHTTPCookie *each in [cookieStorage cookies]) { [cookieStorage deleteCookie:each]; } } Add this to Childbrowser.js ChildBrowser.prototype.LogOut = function() { PhoneGap.exec("ChildBrowserCommand.deleteCookies"); } Add this to FBConnect.js FBConnect.prototype.Logout = function() { window.plugins.childBrowser.LogOut(); } At your page add this code to logout button on click function Logout() { var fb=FBConnect.install(); fb.Logout(); } Enjoy.
