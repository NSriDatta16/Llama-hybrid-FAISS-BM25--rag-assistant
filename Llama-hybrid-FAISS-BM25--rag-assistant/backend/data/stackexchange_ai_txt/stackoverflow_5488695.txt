Andrea's answer led me to this for iframes and modals with the overlay working. Had to set the width and height after opening and append the overlay to the scoped div as well. var d = $(" ") .dialog( { title: "Import", autoOpen: false, width: 1000, height: 600, modal: true, resizable: false, draggable: false }); var scope = $(" ").addClass("jquery-ui-scope").appendTo("body"); d.parent(".ui-dialog").appendTo(scope); d.dialog("open").width(990).height(590); $(".ui-widget-overlay").appendTo(scope);
