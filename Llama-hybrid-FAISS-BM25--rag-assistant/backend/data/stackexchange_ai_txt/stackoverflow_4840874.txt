This will make sure things don't get duplicated in any list. It uses the data-id attribute to give each product a product id. $("#products li").draggable({ appendTo: "body", helper: "clone" }); $(".shoppingCart ol").droppable({ activeClass: "ui-state-default", hoverClass: "ui-state-hover", drop: function(event, ui) { var self = $(this); self.find(".placeholder").remove(); var productid = ui.draggable.attr("data-id"); if (self.find("[data-id=" + productid + "]").length) return; $(" ", { "text": ui.draggable.text(), "data-id": productid }).appendTo(this); // To remove item from other shopping chart do this var cartid = self.closest('.shoppingCart').attr('id'); $(".shoppingCart:not(#"+cartid+") [data-id="+productid+"]").remove(); } }).sortable({ items: "li:not(.placeholder)", sort: function() { $(this).removeClass("ui-state-default"); } }); Example: http://jsfiddle.net/petersendidit/S4QgX/
