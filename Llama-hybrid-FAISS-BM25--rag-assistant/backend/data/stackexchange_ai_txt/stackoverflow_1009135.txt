I can help you somewhat. Here's some code that will allow you to drag something out of the listbox, and when dropped on the desktop, it will create a copy of a file that exists on your machine to the desktop. public partial class Form1 : Form { public Form1() { InitializeComponent(); this.listBox1.Items.Add("foo.txt"); this.listBox1.MouseDown += new MouseEventHandler(listBox1_MouseDown); this.listBox1.DragOver += new DragEventHandler(listBox1_DragOver); } void listBox1_DragOver(object sender, DragEventArgs e) { e.Effect = DragDropEffects.Copy; } void listBox1_MouseDown(object sender, MouseEventArgs e) { string[] filesToDrag = { "c:/foo.txt" }; this.listBox1.DoDragDrop(new DataObject(DataFormats.FileDrop, filesToDrag), DragDropEffects.Copy); } }
