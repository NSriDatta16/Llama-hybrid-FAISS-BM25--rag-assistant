function movebefore(a, b) { // If "a" is the node you're dragging and "b" is the node on top of which you dropped "a" b.parentNode.insertBefore(a, b); // Or "b.nextSibling", depending on what you want to do } It's that simple! The DOM will take care of removing the element from its previous position and adjusting the other elements.
