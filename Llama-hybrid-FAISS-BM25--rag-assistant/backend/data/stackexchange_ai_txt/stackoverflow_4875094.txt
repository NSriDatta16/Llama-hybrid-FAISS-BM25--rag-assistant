Override the underlying storage implementation for instances of FileField on your model dynamically: def setUp(self): self._field = Package._meta.get_field_by_name('file')[0] self._default_storage = self._field.storage test_storage = FileSystemStorage(location=mkdtemp(), base_url="/") self._field.storage = test_storage def tearDown(self): self._field = self._default_storage
