Well one way is to use WNDPROC g_OldProc; LRESULT CALLBACK MyEditWindowProc( HWND hwnd, UINT uMsg, WPARAM wParam, LPARAM lParam ) { if ( uMsg == WM_KEYDOWN ) { // Handle key down. } return g_OldProc( hwnd, uMsg, wParam, lParam ); } then at some opportune moment g_OldProc = (WNDPROC)GetWindowLongPtr( hEdit, GWLP_WNDPROC ); SetWindowLongPtr( hEdit, GWLP_WNDPROC, (LONG_PTR)MyEditWindowProc ); This will replace the window procedure of the hEdit control with your own window procedure that, in turn, calls the original window procedure.
