Be sure to pay attention to the CloseReason so you won't block Windows trying to close your form. Like this: protected override void OnFormClosing(FormClosingEventArgs e) { if (e.CloseReason == CloseReason.UserClosing) { this.Hide(); e.Cancel = true; } else base.OnFormClosing(e); }
