Try this instead: public object Convert(object value, Type targetType, object parameter, CultureInfo culture) { Binary binary = value as Binary; if (binary != null) { using (var stream = new MemoryStream(binary.ToArray())) { BitmapImage bitmap = new BitmapImage(); bitmap.BeginInit(); bitmap.CacheOption = BitmapCacheOption.OnLoad; bitmap.StreamSource = stream; bitmap.EndInit(); bitmap.Freeze(); return bitmap; } } return null; } In your non-working version, your using block means the stream is closed before the image is actually decoded.
