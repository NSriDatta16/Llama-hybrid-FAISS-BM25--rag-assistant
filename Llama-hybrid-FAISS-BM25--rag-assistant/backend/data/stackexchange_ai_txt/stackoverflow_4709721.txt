After thrashing around with SoapVar and SoapParam for days and getting nowhere, found the following simple solution: $oClient = new SoapClient($wsdlFilespec, $arguments); $parameters = array( 'header' => array( 'source' => $_POST['source'], 'accountNo' => $_POST['accountNo'], 'userAccessKey' => $connection['userAccessKey'] ), 'consignmentId' => array('PDH44109', 'PDH44110') ); $request = array('parameters' => $parameters); $result = $oClient->__soapCall($operation, $request);
