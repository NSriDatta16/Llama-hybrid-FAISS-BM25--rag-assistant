S=200506* if [ ${#S} -gt 6 ]; then echo haz filez! else echo no filez fi not a very elegant one, but w/o any external tools/commands (if don't think of "[" as an external one) the clue is if there is some files matched, then "S" variable will contain their names delimited with space. Otherwise it will contain a "200506*" string itself.
