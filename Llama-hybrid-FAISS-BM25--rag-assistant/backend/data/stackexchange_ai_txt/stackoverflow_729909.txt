You should use sql parameters: using( SqlConnection cnn = GetConnection() ) { using( SqlCommand cmd = cnn.CreateCommand() ) { cmd.CommandText = "INSERT INTO [ProfileGallery] ([UserId], [Title], [OriginalImage], [ThumbImage]) VALUES (@UserId, @Title, @OriginalPhoto, @Thumbnail)"; cmd.Parameters.AddWithValue( "@UserId", User.Identity.Name ); cmd.Parameters.AddWithValue( "@Title", Title ); cmd.Parameters.AddWithValue( "@OriginalPhoto", OriginalPhoto ); cmd.Parameters.AddWithValue( "@Thumbnail", Thumbnail ); cnn.Open(); cmd.ExecuteNonQuery(); cnn.Close(); } }
