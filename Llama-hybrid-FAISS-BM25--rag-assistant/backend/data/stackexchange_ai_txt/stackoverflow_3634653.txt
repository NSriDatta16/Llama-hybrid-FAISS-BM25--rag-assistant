It's not a complete wrap onto cylinder, but you will get a starting idea/example: (code is in GLSL, not in HLSL, but i think it will be not hard to convert it) uniform sampler2D tex; void main() { vec2 cen = vec2(0.5,gl_TexCoord[0].y)-gl_TexCoord[0].xy; cen = vec2(pow(cen.x,1.7),pow(cen.y,2.2)); cen.y = 3.9*sin(1.8*gl_TexCoord[0].x-2.3); vec2 mcoord = 1.65*vec2(-0.22+gl_TexCoord[0].x,1.95+gl_TexCoord[0].y); gl_FragColor = texture2D(tex, mcoord+cen); } From this you will get something like that Good luck
