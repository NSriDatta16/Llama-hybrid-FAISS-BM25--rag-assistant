I think you may need to modify your query like : List newVisionData = (from data in visionXML.Descendants(ns + "gcDiversityVision") select new VisionData(new List (data.Descendants(ns + "Employee").Select(f => f.Element(ns + "EmployeeName").Value)), new List (data.Descendants(ns + "Employee").Select(f => f.Element(ns + "EmployeeTitle").Value)), new List (data.Descendants(ns + "Employee").Select(f => f.Element(ns + "include").Attribute("externalPath").Value)), new List (data.Descendants(ns + "Employee").Select(f => f.Element(ns + "EmployeeVideo").Value)), data.Element(ns + "LinkHeaderText").Value, new List (data.Descendants(ns + "Links").Select(f => f.Element(ns + "LinkURL").Value)), new List (data.Descendants(ns + "Links").Select(f => f.Element(ns + "LinkText").Value)), new List (data.Descendants(ns + "EmployeeStoryHeaderParagraph").Select(f => f.Element(ns + "EmployeeHeader").Value)), new List (data.Descendants(ns + "EmployeeStoryHeaderParagraph").Select(f => f.Descendants(ns + "EmployeeParagraphs").Select(ep => ep.Element(ns + "EmployeeParagraph").Value).ToArray())) )).ToList();
