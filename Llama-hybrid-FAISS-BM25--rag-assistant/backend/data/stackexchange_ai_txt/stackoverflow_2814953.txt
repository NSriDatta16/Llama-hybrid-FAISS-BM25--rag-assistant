If you're willing to subclass (ugh!) your table view, something like this might work: - (void)touchesBegan:(NSSet *)touches withEvent:(UIEvent *)event { BOOL backgroundTouched = YES; for (UITouch *touch in touches) { CGPoint location = [touch locationInView:self]; for (UITableViewCell *cell in self.visibleCells) { if (CGRectContainsPoint(cell.frame, location)) { backgroundTouched = NO; break; } } } if (backgroundTouched) { for (UITableViewCell *cell in self.visibleCells) { // This presumes the first subview is the text field you want to resign. [[cell.contentView.subviews objectAtIndex:0] resignFirstResponder]; } } [super touchesBegan:touches withEvent:event]; }
