FYI, I made a C# function based on Bert's response. Thanks Bert! public static string GetLogMessage(string uri, int revision) { string message = null; using (SvnClient cl = new SvnClient()) { SvnLogArgs la = new SvnLogArgs(); Collection col; la.Start = revision; la.End = revision; bool gotLog = cl.GetLog(new Uri(uri), la, out col); if (gotLog) { message = col[0].LogMessage; } } return message; }
