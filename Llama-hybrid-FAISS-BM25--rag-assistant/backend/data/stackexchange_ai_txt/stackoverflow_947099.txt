class Proxy implements IteratorAggregate { public $data = array(); public function __get($name) { return $data[$name]; } public function getIterator() { $o = new ArrayObject($this->data); return $o->getIterator(); } } $p = new Proxy(); $p->data = array(2, 4, 6); foreach ($p as $v) { echo $v; } Output is: 246 . See Object Iteration in the PHP docs for more details.
