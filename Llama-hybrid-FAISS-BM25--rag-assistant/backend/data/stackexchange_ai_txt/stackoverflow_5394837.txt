Not sure how this relates to an element (jQuery plugin), but here is how you can structure your prototype: var DataHelper = function(_url, _function){ this.dataurl = _url; this.datafunction = _function; }; DataHelper.prototype = { GetData: function(){ // loads data from storage // using this.dataurl, and this.datafunction } }; Usage: var dataHelper = new DataHelper('/mysite/url/', function(){ // data function }); dataHelper.GetData();
