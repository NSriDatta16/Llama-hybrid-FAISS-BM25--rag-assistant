Also you could probably do something with the mousemove and mousedown events together to disable the click event: var dragging = 0; $('.drag').mousedown(function() { $(document).mousemove(function(){ dragging = 1; }); }); $(document).mouseup(function(){ dragging = 0; $(document).unbind('mousemove'); }); $('.class').click(function() { if (dragging == 0){ // default behaviour goes here } else return false; )};
