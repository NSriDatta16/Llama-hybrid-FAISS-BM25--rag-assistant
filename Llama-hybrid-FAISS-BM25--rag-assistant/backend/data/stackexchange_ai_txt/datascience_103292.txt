Solution for sklearn API (checked on v3.3.0): import lightgbm as lgb param = {'objective': 'binary', "is_unbalance": 'true', 'metric': 'average_precision'} model_skl = lgb.sklearn.LGBMClassifier(**param) # early stopping and verbosity # it should be 0 or False, not -1/-100/etc callbacks = [lgb.early_stopping(10, verbose=0), lgb.log_evaluation(period=0)] # train model_skl.fit(x_train, y_train, eval_set=[(x_train, y_train), (x_val, y_val)], eval_names=['train', 'valid'], eval_metric='average_precision', callbacks=callbacks)
