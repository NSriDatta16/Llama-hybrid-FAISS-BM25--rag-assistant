You need at least 1 level of inheritance and an upcast to demonstrate it. Here is a very simple example: class Animal { public: // turn the following virtual modifier on/off to see what happens //virtual std::string Says() { return "?"; } }; class Dog: public Animal { public: std::string Says() { return "Woof"; } }; void test() { Dog* d = new Dog(); Animal* a = d; // refer to Dog instance with Animal pointer std::cout Says(); // always Woof std::cout Says(); // Woof or ?, depends on virtual }
