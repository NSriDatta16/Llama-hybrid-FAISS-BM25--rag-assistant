You probably need to evaluate scripts in HTML pushed from server. Something like this: function extractScripts(html) { // based on PrototypeJs var ScriptFragment = " ]*>([\\S\\s]*?) "; var matchAll = new RegExp(ScriptFragment, "img"); var matchOne = new RegExp(ScriptFragment, "im"); return (html.match(matchAll) || []).map(function(scriptTag) { return (scriptTag.match(matchOne) || ['', ''])[1]; }); } function evalScripts(html) { return extractScripts(html).map(function(script) { return eval(script) }); }
