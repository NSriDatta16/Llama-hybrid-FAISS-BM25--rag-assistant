Vertical alignment you'll have to do yourself ((height of view + height of string)/2). Horizontal alignment you can do with: NSMutableParagraphStyle *style = [[NSParagraphStyle defaultParagraphStyle] mutableCopy]; style.alignment = NSTextAlignmentCenter; NSDictionary *attr = [NSDictionary dictionaryWithObject:style forKey:NSParagraphStyleAttributeName]; [myString drawInRect:someRect withAttributes:attr];
