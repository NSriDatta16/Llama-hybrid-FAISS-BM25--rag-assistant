How about this? import java.util.Scanner; public class Average{ public static void main(String[] args) { int mark = 0; int countTotal = 0; // to count the number of entered marks int avg = 0; // to calculate the total average Scanner scan = new Scanner(System.in); System.out.println("Enter student name: "); String name = scan.next(); System.out.println("Please enter marks one by one?"); while(scan.hasNextInt()){ int tempMark = scan.nextInt(); mark += tempMark; countTotal+=1; System.out.println("If you are done with mark, type \"Done!\""); } avg = mark/countTotal; System.out.print( name + " " + avg ); } }
