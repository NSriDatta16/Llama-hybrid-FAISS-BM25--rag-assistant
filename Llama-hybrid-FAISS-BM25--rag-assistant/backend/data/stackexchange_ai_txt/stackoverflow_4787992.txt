Here is what I finally did that worked: $(".box-clone").live('mouseover', function() { $(this).draggable({ axis: 'y', containment: 'html' }); }); $(".box").draggable({ axis: 'y', containment: 'html', helper: 'clone', stop: function(event, ui) { $(ui.helper).clone(true).removeClass('box ui-draggable ui-draggable-dragging').addClass('box-clone').appendTo('body'); } });
