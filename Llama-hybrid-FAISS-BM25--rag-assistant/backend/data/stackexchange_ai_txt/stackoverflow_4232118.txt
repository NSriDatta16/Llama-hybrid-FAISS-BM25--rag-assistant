this isn't elegant, but does the job: import re from urlparse import urljoin relative_urls_re = re.compile('( ]+href\s*=\s*["\']?)(?!http)([^"\'>]+)', re.IGNORECASE) relative_urls_re.sub(lambda m: m.group(1) + urljoin(base_url, m.group(2)), html)
