There's no way to simply detect the size of Flash stage from JS. You'd better call a JS function from inside Flash to set the stage size. So, you'll have to put this code in the Flash movie (AS2): import flash.external.ExternalInterface; ExternalInterface.call("resizeFlash", Stage.width, Stage.height); Then in the HTML page you'd have the JS function that resizes the Flash element: var max_flash_width = 800; var max_flash_height = 600; function resizeFlash(w, h) { // force the w and h to be within limits set above w = Math.min(w, max_flash_width); h = Math.min(h, max_flash_height); // then resize the flash element using w and h var flash_elem = document.getElementById("example_flashGame"); flash_elem.width = w; flash_elem.height = h; if (navigator.userAgent.toLowerCase().indexOf("safari") !=- 1) { var flash_elem = document.getElementById("example_flashGame"); flash_elem.width = w; flash_elem.height = h; } }
