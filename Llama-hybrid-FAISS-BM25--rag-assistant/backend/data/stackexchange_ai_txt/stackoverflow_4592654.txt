You could use TestFixtureSetUp to prepare the fake data for each unit test: [TestFixture] public class AuditTrailRepositoryTests { private Guid orgId1; private IList fakeAuditData; [TestFixtureSetUp] public void Init() { orgId1 = Guid.NewGuid(); fakeAuditData = new List { new AuditTrail { ChangeDate = new DateTime(2011, 1, 1), EntityName = "Test", OrganisationId = orgId1, Username = "admin" } }; } }
