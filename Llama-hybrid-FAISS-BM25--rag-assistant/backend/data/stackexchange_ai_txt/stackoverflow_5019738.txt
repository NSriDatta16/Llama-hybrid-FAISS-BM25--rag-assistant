Have a delete statement with the same criteria you have for insert except that you now use EXISTS instead of NOT EXISTS . Something like this: INSERT INTO dashboardtasks SELECT [tour], tourname, [taskname], [deptdate], [tasktype], [desc], [duedate], [compdate], [comments], [agent], [compby], [graceperiod], completed , canceled FROM staggingtasks WHERE NOT EXISTS ( SELECT * FROM dashboardtasks WHERE ( staggingtasks.tour=dashboardtasks.tour and staggingtasks.taskname=dashboardtasks.taskname and staggingtasks.deptdate=dashboardtasks.deptdate and staggingtasks.duedate=dashboardtasks.duedate and staggingtasks.tourname=dashboardtasks.tourname ) ) DELETE FROM staggingtasks WHERE EXISTS ( SELECT * FROM dashboardtasks WHERE ( staggingtasks.tour=dashboardtasks.tour and staggingtasks.taskname=dashboardtasks.taskname and staggingtasks.deptdate=dashboardtasks.deptdate and staggingtasks.duedate=dashboardtasks.duedate and staggingtasks.tourname=dashboardtasks.tourname ) )
