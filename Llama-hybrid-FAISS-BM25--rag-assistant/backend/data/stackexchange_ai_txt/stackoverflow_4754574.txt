You could insert a function like this to change the user agent on the fly before you make your http request: function changeuserAgent() { var altuserAgentGetter = function () { return "Mozilla/5.0 (Windows; U; Windows NT 6.1; en-US; rv:1.9.2) Gecko/20100115 "; }; if (Object.defineProperty) { Object.defineProperty(navigator, "userAgent", { get: altuserAgentGetter }); } else if (Object.prototype.__defineGetter__) { navigator.__defineGetter__("userAgent", altuserAgentGetter); } }
