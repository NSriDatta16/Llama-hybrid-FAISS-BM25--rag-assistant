You could use jQuery's Live and Draggable/Resizable together, as illustrated here . Example: $(function() { $('.window').live("mouseover", function() { if (!$(this).data("init")) { $(this).data("init", true); $(this).draggable({ containment: '#desktop', scroll: false, opacity: 0.6, handle:'.winTitle' }); } }); });
