When an element is selected it gets the ui-selected class added. So you could get all selected elements with $(".ui-selected") This might not work exactly but I think the idea would be something like this: $('#content_td_account').selectable({ filter: 'li:not(".non_draggable")', selecting: function(event, ui) { var p = $(this).parent(); $(".ui-selected").each(obj, function() { if(obj.parent() == p) { // get rad } }); } });
