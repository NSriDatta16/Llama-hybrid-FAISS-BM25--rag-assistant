Just for some variety, how about testing the string itself, instead of TryParse, which would require extra storage, and potentially an unneeded cast? static void Main(string[] args) { var isFalse = "t".IsInt(); var isTrue = "123".IsInt(); var isAlsoFalse = "123.1".IsInt(); } static bool IsInt(this IEnumerable s) { return s.All(x => char.IsNumber(x)); }
