if you want to use, in my opinion, something more proper approach, than regexp: from urlparse import * urlparsed = urlparse('www.example.com?u=userpage&as=233&p=1') # -> ParseResult(scheme='', netloc='', path='www.example.com', params='', query='u=userpage&as=233&p=1', fragment='') qdict = dict(parse_qsl(urlparsed.query)) # -> {'as': '233', 'p': '1', 'u': 'userpage'} qdict.get('p') == '1' and qdict.get('u') == 'userpage' # -> True
