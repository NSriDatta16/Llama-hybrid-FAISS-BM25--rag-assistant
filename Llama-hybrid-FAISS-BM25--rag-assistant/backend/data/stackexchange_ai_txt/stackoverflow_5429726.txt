The following code might help you - (void)touchesBegan:(NSSet *)touches withEvent:(UIEvent *)event { timeStampStart = event.timestamp; } - (void)touchesEnded:(NSSet *)touches withEvent:(UIEvent *)event { timeStampEnd = event.timestamp; touchDuration = timeStampEnd - timeStampStart; if(touchDuration > smallTimeStamp) [super touchesEnded:touches withEvent:event]; else [self zoomMyTableView]; }
