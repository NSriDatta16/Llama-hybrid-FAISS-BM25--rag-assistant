Try this: [DataContract(Namespace = Constants.OrgStructureNamespace)] public class EntityInfo : IExtensibleDataObject { public Guid EntityID { get; set; } [DataMember(Name="EntityID")] byte[] EntityIDBytes { get { return this.EntityID.ToByteArray(); } set { this.EntityID = new Guid(value); } } [DataMember] public EntityType EntityType { get; set; } [DataMember] public IList EntityVersions { get; set; } [DataMember] public IList OrganisationStructures { get; set; } #region IExtensibleDataObject Members // ... #endregion } It looks like the DataContractSerializer handles byte arrays by Base64-encoding them, whereas it appears to just use the Guid.ToString method for Guids.
