You should use the method decoded_content from the class HTTP::Message instead. Manual decoding is not necessary. #!/usr/bin/env perl use utf8; use strict; use LWP::UserAgent; my $html = LWP::UserAgent->new ->get('http://www.boc.cn/sourcedb/whpj/') ->decoded_content; my $str_chinese = '首页'; my $str_english = 'English'; if ($html =~ /$str_chinese/) { print "chinese word matched\n"; } else { print "chinese word unmatched\n"; } if ($html =~ /$str_english/i) { print "english word matched\n"; } else { print "english word unmatched\n"; } Output: chinese word matched english word matched
