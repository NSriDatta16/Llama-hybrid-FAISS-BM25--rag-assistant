Here's a script which when run on your paragraph produces an array with the following values: Array ( [0] => Nick Swisher [1] => James Shields [2] => New York Yankees [3] => Tampa Bay [4] => Tuesday [5] => Rays [6] => AL East. ) Is this helpful? $proper_nouns = Array(); $words = explode(' ', $paragraph); for ($i = 0; $i 0) { $proper_noun = $words[$i]; $index = 1; while (true) { if ($i + $index 0) { $proper_noun = $proper_noun." ".$words[$i + $index]; $index++; } else { $i = $i + $index - 1; break; } } else { break; } } array_push($proper_nouns, $proper_noun); } }
