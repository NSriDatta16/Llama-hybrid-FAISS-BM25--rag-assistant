From the docs: https://docs.djangoproject.com/en/1.7/ref/forms/validation/#cleaning-and-validating-fields-that-depend-on-each-other from django.forms.util import ErrorList def clean(self): if self.cleaned_data['type'].organized_by != self.cleaned_data['organized_by']: msg = 'The type and organization do not match.' self._errors['type'] = ErrorList([msg]) del self.cleaned_data['type'] if self.cleaned_data['start'] > self.cleaned_data['end']: msg = 'The start date cannot be later than the end date.' self._errors['start'] = ErrorList([msg]) del self.cleaned_data['start'] return self.cleaned_data
