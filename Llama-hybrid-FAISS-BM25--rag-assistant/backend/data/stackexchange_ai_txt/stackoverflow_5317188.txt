You can compare part by part: string left = (string)x; string right = (string)y; string lname = left.Substring(3, 3); string rname = right.Substring(3, 3); int result = lname.CompareTo(rname); if (result != 0) return result; string lage = left.Substring(0, 3); string rage = right.Substring(0, 3); int result = lage.CompareTo(rage); if (result != 0) return result; string lamt = left.Substring(6); string ramt = right.Substring(6); return -lamt.CompareTo(ramt);
