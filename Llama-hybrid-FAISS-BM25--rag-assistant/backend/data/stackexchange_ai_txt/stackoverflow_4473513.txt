private void TabControl_DragEnter(object sender, DragEventArgs e) { TabItem item = e.Source as TabItem; TabControl tabControl = sender as TabControl; if (item != null && tabControl != null) { if (tabControl.SelectedItem != item) tabControl.SelectedItem = item; } }
