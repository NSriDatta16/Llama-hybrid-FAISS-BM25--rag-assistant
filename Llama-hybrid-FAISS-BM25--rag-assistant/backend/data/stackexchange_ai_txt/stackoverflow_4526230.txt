You can try: function color_avg($color1,$color2,$factor) { // extract RGB values for color1. list($r1,$g1,$b1) = str_split(ltrim($color1,'#'),2); // extract RGB values for color2. list($r2,$g2,$b2) = str_split(ltrim($color2,'#'),2); // get the average RGB values. $r_avg = (hexdec($r1)*(1-$factor)+hexdec($r2)*$factor); $g_avg = (hexdec($g1)*(1-$factor)+hexdec($g2)*$factor); $b_avg = (hexdec($b1)*(1-$factor)+hexdec($b2)*$factor); // construct the result color. $color_avg = '#'.sprintf("%02s",dechex($r_avg)). sprintf("%02s",dechex($g_avg)). sprintf("%02s",dechex($b_avg)); // return it. return $color_avg; } See it
