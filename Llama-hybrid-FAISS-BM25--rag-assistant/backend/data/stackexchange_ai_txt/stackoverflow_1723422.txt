While it may not be the most elegant solution, you can add another variable to keep track of what step you are on: int step = 0; try { classVar = functionCall(input, sEnum.First); step++; classVar = functionCall(input, sEnum.Second); step++; classVar = functionCall(input, sEnum.Third); } catch (Exception ex) { //examine the step variable here Debug.Assert(false, ex.ToString()); }
