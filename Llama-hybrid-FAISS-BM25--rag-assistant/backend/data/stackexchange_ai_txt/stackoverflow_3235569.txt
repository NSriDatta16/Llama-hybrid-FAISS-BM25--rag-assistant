Solved it!! First i saved it as a html page then opened it by word programatically and saved as an rtf file. The Word component made the whole conversion. protected override bool HandleDropEvent(DragEventArgs e) { bool result = false; if (e.Data.GetDataPresent(DataFormats.Text)) { System.Windows.Forms.IDataObject data = e.Data; if (data.GetDataPresent("HTML Format")) { byte[] rawHtmlBytes = HtmlFromIDataObject.GetHtml(data); string _html = Encoding.UTF8.GetString(rawHtmlBytes); object vk_missing = System.Reflection.Missing.Value; //1.Save the html file (in %temp% folder) string file = IOUtil.CreateTextFile(_html, ".html"); object oFile = file; Microsoft.Office.Interop.Word.Application wordApp = new Microsoft.Office.Interop.Word.Application(); //2.open the html file Microsoft.Office.Interop.Word.Document doc = wordApp.Documents.Open(ref oFile, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing); //3.saving as rtf string directory = Path.GetDirectoryName(file); string fileName = Path.GetFileNameWithoutExtension(file); file = Path.Combine(directory, fileName+ "_" + ".rtf"); object oSave = file; doc.SaveAs(ref oSave, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing, ref vk_missing); result = true; } } return result; }
