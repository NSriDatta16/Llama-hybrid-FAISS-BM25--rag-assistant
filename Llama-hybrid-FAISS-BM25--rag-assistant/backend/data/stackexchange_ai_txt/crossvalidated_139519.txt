Here's the complete Stata Code for anyone interested: #delimit; clear; /* Create Fake Data */ input str1 cohort str1 sex smoker n; "A" "M" 0 18; "A" "F" 0 14; "B" "M" 0 4; "B" "F" 0 8; "A" "M" 1 2; "A" "F" 1 6; "B" "M" 1 16; "B" "F" 1 12; end; sencode cohort, replace; sencode sex, replace; expand n; drop n; /* Check the Odds */ table sex cohort, c(mean smoker); logit smoker i.cohort##i.sex, or nolog; margins sex, dydx(cohort); margins r.sex, dydx(cohort);
