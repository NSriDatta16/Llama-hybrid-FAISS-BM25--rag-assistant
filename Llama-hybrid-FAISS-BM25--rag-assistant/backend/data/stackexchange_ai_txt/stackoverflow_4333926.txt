Here's some working code to post the âŒ˜+key event (assuming a known keycode): // some common keycodes #define KEY_CODE_x ((CGKeyCode)7) #define KEY_CODE_c ((CGKeyCode)8) #define KEY_CODE_v ((CGKeyCode)9) void DCPostCommandAndKey(CGKeyCode key) { CGEventSourceRef source = CGEventSourceCreate(kCGEventSourceStateCombinedSessionState); CGEventRef keyDown = CGEventCreateKeyboardEvent(source, key, TRUE); CGEventSetFlags(keyDown, kCGEventFlagMaskCommand); CGEventRef keyUp = CGEventCreateKeyboardEvent(source, key, FALSE); CGEventPost(kCGAnnotatedSessionEventTap, keyDown); CGEventPost(kCGAnnotatedSessionEventTap, keyUp); CFRelease(keyUp); CFRelease(keyDown); CFRelease(source); }
