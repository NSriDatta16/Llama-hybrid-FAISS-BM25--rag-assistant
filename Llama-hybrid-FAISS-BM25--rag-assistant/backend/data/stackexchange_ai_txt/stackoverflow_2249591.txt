The following works for limiting the amount of text if you don't mind losing the paragraph tags. var shortText = $("#content").text() // get the text within the div .trim() // remove leading and trailing spaces .substring(0, 600) // get first 600 characters .split(" ") // separate characters into an array of words .slice(0, -1) // remove the last full or partial word .join(" ") + "..."; // combine into a single string and append "..."
