If you have subviews in your AnimalView, you can iterate through the superview hierarchy to see if the touch was handled by a subview of your AnimalView by using code like this: - (void)touchesBegan:(NSSet *)touches withEvent:(UIEvent *)event { UITouch *touch = [touches anyObject]; UIView *hitView = touch.view; AnimalView *animal = nil; while (hitView != nil) { if ([hitView isKindOfClass:[AnimalView class]]) { animal = (AnimalView *) hitView; break; } hitView = [hitView superview]; } if (animal) { [animal doSomething]; } }
