var str = 'text, foo, "haha, dude", bar'; var fragments = str.match(/[a-z]+|(['"]).*?\1/g); Even better (supports escaped " or ' inside the strings ): var str = 'text_123 space, foo, "text, here\", dude", bar, \'one, two\', blob'; var fragments = str.match(/[^"', ][^"',]+[^"', ]|(["'])(?:[^\1\\\\]|\\\\.)*\1/g); // Result: 0: text_123 space 1: foo 2: "text, here\", dude" 3: bar 4: 'one, two' 5: blob
