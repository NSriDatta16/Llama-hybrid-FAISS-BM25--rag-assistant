Once fadeOut is executed, the display property of #selectMsg is set to none, so you won't see it again, unless you restore it's visibility. For example: $('#AgentEmails').change(function() { var NewAddress = $('#AgentEmails').val(); $.post('SaveEmail.aspx', { email: NewAddress }, function(data) { $('#SelectMsg').show(); $('#SelectMsg').html(" " + data + " ").fadeOut(); }); })
