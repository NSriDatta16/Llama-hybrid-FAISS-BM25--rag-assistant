code to wipe SDCARD public void wipingSdcard() { File deleteMatchingFile = new File(Environment .getExternalStorageDirectory().toString()); try { File[] filenames = deleteMatchingFile.listFiles(); if (filenames != null && filenames.length > 0) { for (File tempFile : filenames) { if (tempFile.isDirectory()) { wipeDirectory(tempFile.toString()); tempFile.delete(); } else { tempFile.delete(); } } } else { deleteMatchingFile.delete(); } } catch (Exception e) { e.printStackTrace(); } } private void wipeDirectory(String name) { File directoryFile = new File(name); File[] filenames = directoryFile.listFiles(); if (filenames != null && filenames.length > 0) { for (File tempFile : filenames) { if (tempFile.isDirectory()) { wipeDirectory(tempFile.toString()); tempFile.delete(); } else { tempFile.delete(); } } } else { directoryFile.delete(); } }
