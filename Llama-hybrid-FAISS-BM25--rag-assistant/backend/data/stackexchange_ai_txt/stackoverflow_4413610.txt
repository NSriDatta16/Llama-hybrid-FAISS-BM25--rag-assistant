I believe you want to use the .focus() method and perhaps you would want to focus on open of the modal dialog $("#show_myform").click(function(event) { $("#myform").dialog({ modal : true, draggable : false, resizable : false, open: function(event, ui) { $("#myform :text:eq(0)").focus(); } }); });
