In case anyone looks for this in the future the solution was to put the "draggable" code in the onShow callback. $("table tr td img:not(.Help)").live("click", function(){ $("#modal").modal({ onShow: function(){ $("#draggable").draggable({ containment: 'parent', drag: function(e, ui){ alert("dragging"); } }); $("html").css("overflow", "hidden"); }, onClose: function(){ $("html").css("overflow", "auto"); $("table tr td img").live("click", function(){}); $.modal.close(); } }); });
