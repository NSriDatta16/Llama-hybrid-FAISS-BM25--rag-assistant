Update OK, try this: public class GenericThingy { private Class mClazz; private T mValue; public final T[] mCandidates; public GenericThingy(Class clazz, T[] pCandidates, T pInitValue) { mClazz = clazz; mCandidates = pCandidates; mValue = pInitValue; } public void setValue(Object newValue) throws ClassCastException { mValue = mClazz.cast(newValue); } }
