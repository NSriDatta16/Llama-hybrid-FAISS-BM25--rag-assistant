You can use input:not(:checked) and just add it into your data object: $('#agent_region').click(function() { var ids = $('#agent_favourites_form input:checked') .map(function(){ return this.value; }).get(); var nonids = $('#agent_favourites_form input:not(:checked)') .map(function(){ return this.value; }).get(); $.ajax({ type: "POST", url: "set_agent_favourites.php", data: {map: ids, unmap: nonids}, success: function(msg){ $.jnotify(msg, 3000); } }); return false; });
