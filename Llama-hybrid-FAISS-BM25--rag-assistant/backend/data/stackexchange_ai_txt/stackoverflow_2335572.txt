If you want your picker view just to display an incremented number then you might use something like this: #define kPickerValuesAmount 200 #pragma mark - #pragma mark UIPickerViewDataSource methods - (NSInteger)numberOfComponentsInPickerView:(UIPickerView *)pickerView { return 1; } - (NSInteger)pickerView:(UIPickerView *)pickerView numberOfRowsInComponent:(NSInteger)component { return kPickerValuesAmount; } - (NSString *)pickerView:(UIPickerView *)pickerView titleForRow:(NSInteger)row forComponent:(NSInteger)component { return [NSString stringWithFormat:@"%i unit%@", (row + 1), (row == 0 ? @"" : @"s")]; }
