Here is a rather ugly yet pragmatic solution: import string decoded_str = " Name(++info++)Age(++info++)Adress of the emp(++info++)" letters = list(string.lowercase) token = "(++info++)" rep_token = "(++info %s++)" i = 0 while (token in decoded_str): decoded_str = decoded_str.replace(token, rep_token % letters[i], 1) i += 1 print decoded_str
