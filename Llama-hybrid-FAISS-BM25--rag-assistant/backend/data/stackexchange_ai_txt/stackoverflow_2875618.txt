INSERT INTO File SELECT DISTINCT [FileRef], [FileDesc], [FilePath] FROM [DataToImport] di WHERE NOT EXISTS ( SELECT di.FileRef, di.FileDesc, di.FilePath INTERSECT SELECT FileRef, FileDesc, FilePath FROM File ) It's not important in your case, but this would also handle the NULL values as DISTINCT correctly if the columns were nullable. @Philip Kelley 's solution is more elegant, though.
