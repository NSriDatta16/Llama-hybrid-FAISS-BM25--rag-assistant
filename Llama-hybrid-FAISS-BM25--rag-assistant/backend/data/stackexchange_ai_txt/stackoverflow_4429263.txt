You'll probably find better luck doing this in the SDK code rather than modifying file systems. Try: //Delete previous cookies NSHTTPCookieStorage *cookieStorage = [NSHTTPCookieStorage sharedHTTPCookieStorage]; for (NSHTTPCookie *each in [[[cookieStorage cookiesForURL:YOUR_URL] copy] autorelease]) { [cookieStorage deleteCookie:each]; }
