Assuming your paragraphs are in a wrapper div called 'info': (I haven't tested it, but it would be something like this) $('info').select('P').each(function(element) { Event.observe(element,'click',function(event){ flipIt(event) }) }) function flipIt(evt) { var element = evt.element(); if($(element).hasClassName('active')) { $(element).removeClassName('active') } else { $(element).addClassName('active') } }
