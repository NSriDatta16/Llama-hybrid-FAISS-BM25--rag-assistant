To achive that you could simple run a product search using LastModified field to filter recently created products by date. You'll need to set the catalogs you want to search for, since you said you have to search in all of them. I believe the code snippet below is what you need. Remember you'll need full-text search up and running to get results from the search. var queryBuilder = new CommerceQuery (); queryBuilder.SearchCriteria.FullTextSearchType = CommerceFullTextSearchType.FreeText; queryBuilder.SearchCriteria.Catalogs.Add("MyCatalogName1"); queryBuilder.SearchCriteria.Catalogs.Add("MyCatalogName2"); queryBuilder.SearchCriteria.Catalogs.Add("MyCatalogName3"); queryBuilder.SearchCriteria.WhereClause = "LastModified > '2010-10-01'"; CommerceQueryOperationResponse response = (CommerceQueryOperationResponse)new OperationServiceAgent() .ProcessRequest(requestContext, queryBuilder.ToRequest()) .OperationResponses[0]; List products = new List (); if (response.CommerceEntities != null && response.CommerceEntities.Count > 0) { foreach(var p in response.CommerceEntities) { products.Add((Product)p); } } return products;
