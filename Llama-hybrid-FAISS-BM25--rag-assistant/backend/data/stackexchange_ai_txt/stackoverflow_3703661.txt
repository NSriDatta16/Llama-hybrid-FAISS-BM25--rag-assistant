You can use Restrictions.Or or use a Disjunction for multiple or's. session.CreateCriteria () .CreateAlias("References", "r", JoinType.LeftOuterJoin) .Add(Restrictions.Or( Restrictions.Eq("Reference", reference), Restrictions.Eq("r.Reference", reference))) .SetResultTransformer(new DistinctRootEntityResultTransformer()) .List ();
