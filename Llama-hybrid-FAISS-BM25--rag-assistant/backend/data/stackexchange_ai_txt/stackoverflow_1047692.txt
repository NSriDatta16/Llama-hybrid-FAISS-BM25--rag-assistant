This is almost the exact same algorithm described in Anurag Uniyal's answer, except using the file as an iterator instead of readline() : from collections import defaultdict try: from io import StringIO # 2.6+, 3.x except ImportError: from StringIO import StringIO # 2.5 data = defaultdict(int) #with open("filename", "r") as f: # if a real file with StringIO("10\n5\n-1\n-1\n-1\n1\n1\n0\n2") as f: for line in f: data[int(line)] += 1 for number, count in data.iteritems(): print number, "was found", count, "times"
