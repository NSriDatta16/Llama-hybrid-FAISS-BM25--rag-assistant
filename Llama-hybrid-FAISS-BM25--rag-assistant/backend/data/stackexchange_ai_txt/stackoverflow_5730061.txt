IN detailviewcotroller.h , create outlet as follows, IBOutlet UIWebView *WEBVIEW; connect outlet in xib by drag n droping uiwebview control. In your detailviewcontroller.m's viewdidload method u can write as follows- NSString *str=@"yr default webpage link here"; NSURL *url=[NSURL URLWithString:str]; NSURLRequest *req=[NSURLRequest requestWithURL:url]; [WEBVIEW loadRequest:req];
