thank your provides a solution. I have found an other solution without need an outer div for this problem. I am use "distance" option to detect how long mouse has moved, then use "stop" option to remove element. $(".droppable").droppable({ drop: function(event, ui) { var obj = $(ui.draggable).clone(); obj.draggable({ distance: 100,//used to measure how far my mouse moved. helper: 'clone', opacity : 0.35, stop: function(event, ui) { $(this).remove();//remove this element. } } );//obj.draggable }//drop })
