thanks for sharing, i wrote up this quick hack, not elegant at all.... def search(request): query = request.GET.get('q', '') query = query.split() product_results = [] category_results = [] if query: for x in query: product_results.extend(Product.objects.filter(Q(title__icontains=x)| Q(brand__icontains=x)| Q(description__icontains=x))) category_results.extend(Category.objects.filter(title__icontains=x)) query = request.GET.get('q', '') product_results = list(set(product_results)) category_results = list(set(category_results)) return render_to_response('search_results.html', {'query': query, 'product_results': product_results, 'category_results': category_results})
