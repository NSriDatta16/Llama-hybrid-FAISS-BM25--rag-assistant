I removed the duplicated code and solved the problem by saving the previous element id by the data() function. JS Bin $(document).ready(function() { $('#area1').append(' '); $(".shoulddraggable").draggable({ scroll: false, revert: "invalid", scope: "items", }); $(".shoulddraggable").data('previousId', 'area1'); $('.droppable').droppable({ scope: "items", drop: function(event, ui) { alert('Previous container id: ' + ui.draggable.data('previousId')); alert("Element: "+ui.draggable.data("elementid")+" dragged into" + $(this).attr('id')); ui.draggable.data('previousId', $(this).attr('id')); } }); });
