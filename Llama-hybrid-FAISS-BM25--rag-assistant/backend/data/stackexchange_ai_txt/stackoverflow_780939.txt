Yuo can sove it by overriding showDropFeedback() method. My code below: import mx.controls.HorizontalList; import mx.controls.listClasses.IListItemRenderer; import mx.core.mx_internal; import mx.events.DragEvent; use namespace mx_internal; public class HList extends HorizontalList { public function HList() { super(); } override public function showDropFeedback(event:DragEvent):void { super.showDropFeedback(event); dropIndicator.setActualSize(rowHeight - 4, 4); DisplayObject(dropIndicator).rotation = 90; } }
