I haven't gotten a chance to test it yet, but I wrote this script to fire the contextmenu event on an element after a long press of 1.5 seconds or more. Try it out. UPDATE: finally got a chance to test it, it works as intended. I lowered the delay from 1500 ms to 1200 ms since the delay seemed too long for my taste. (function() { var EM = Ext.EventManager, body = document.body, activeTouches = {}, onTouchStart = function(e, t) { var be = e.browserEvent; Ext.id(t); if(be.touches.length === 1) { activeTouches[t.id] = fireContextMenu.defer(1200, null, [e, t]); } else { cancelContextMenu(e, t); } }, fireContextMenu = function(e, t) { var touch = e.browserEvent.touches[0]; var me = document.createEvent("MouseEvents"); me.initMouseEvent("contextmenu", true, true, window, 1, // detail touch.screenX, touch.screenY, touch.clientX, touch.clientY, false, false, false, false, // key modifiers 2, // button null // relatedTarget ); t.dispatchEvent(me); }, cancelContextMenu = function(e, t) { clearTimeout(activeTouches[t.id]); }; if(navigator.userAgent.match(/iPad/i) != null) { Ext.onReady(function() { EM.on(body, "touchstart", onTouchStart); EM.on(body, "touchmove", cancelContextMenu); EM.on(body, "touchend", cancelContextMenu); }); } })();
