Without using global variables or timeOuts you could leverage the callback for your animation to reassign the function. function DoTheAnimation(){ var $a = $(this); var h = $("div").height() + 100; $("div").animate({'height': h}, "slow",function(){ $a.one("click.animate", DoTheAnimation); }); } $("a").click(function(event){event.preventDefault();}).one("click.animate", DoTheAnimation); Code example on jsfiddle .
