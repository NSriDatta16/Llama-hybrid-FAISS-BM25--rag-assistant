I recommend that you create a custom UITableViewCell and use the touchesBegan and touchesEnded . - (void)touchesBegan:(NSSet *)touches withEvent:(UIEvent *)event { [super touchedBegan:touched withEvent:event]; // ... display image 2, hide image 1 } - (void)touchesEnded:(NSSet *)touches withEvent:(UIEvent *)event { [super touchedEnded:touched withEvent:event]; // ... display image 1, hide image 2 }
