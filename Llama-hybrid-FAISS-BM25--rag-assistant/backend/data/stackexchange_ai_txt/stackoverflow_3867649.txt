Some combination of String.split() and jQuery.wrap() will probably do the trick. Something like this: (untested, just whipped this up quickly) var paragraph = $(' I want to put a ding in the universe. '); var text = paragraph.text(); var newText = ''; $.each(text.split(), function(i, val) { if (val != ' ') { newText = newText + $(val).wrap(' '); } else { newText = newText + val; } }); paragraph.html(newText);
