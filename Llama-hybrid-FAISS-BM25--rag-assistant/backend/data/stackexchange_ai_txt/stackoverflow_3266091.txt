use SharedPreferences SharedPreferences Example (from webworld): /** * get if this is the first run * * @return returns true, if this is the first run */ public boolean getFirstRun() { return mPrefs.getBoolean("firstRun", true); } /** * store the first run */ public void setRunned() { SharedPreferences.Editor edit = mPrefs.edit(); edit.putBoolean("firstRun", false); edit.commit(); } SharedPreferences mPrefs; /** * setting up preferences storage */ public void firstRunPreferences() { Context mContext = this.getApplicationContext(); mPrefs = mContext.getSharedPreferences("myAppPrefs", 0); //0 = mode private. only this app can read these preferences }
