Or use a regex as shown below if you want to check if IP addresses are of the format 0-255.0-255.0-255.0-255 import java.util.regex.Matcher; import java.util.regex.Pattern; /** * * @author ragavan */ public class MatchIPv4Address { public static void main(String[] args) { String range = "(([01][0-9][0-9])|(2([0-4][0-9]|5[0-5]))|([0-9]?[0-9]))"; String pattern = range+"\\."+range+"\\."+range+"\\."+range; System.out.println(pattern); Pattern p = Pattern.compile(pattern); Matcher m = p.matcher("144.229.1.99"); System.out.println(m.matches()); } }
