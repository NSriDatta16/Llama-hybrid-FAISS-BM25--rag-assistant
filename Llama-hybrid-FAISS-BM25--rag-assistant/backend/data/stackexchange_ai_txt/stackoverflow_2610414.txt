I have discovered an even more elegant solution that does not require the modifications to the aspnet.config file. // get the current WindowIdentity of the main thread var currentIdentity = WindowsIdentity.GetCurrent(); Thread thread; thread = new Thread(() => { // using lambda closure, access the current identity // inside the background thread scope and call Impersonate var context = currentIdentity.Impersonate(); // do multi-threaded stuff // undo the impersonation before leaving the thread context.Undo(); }) { Name = "BackgroundThread", IsBackground = true }; thread.Start();
