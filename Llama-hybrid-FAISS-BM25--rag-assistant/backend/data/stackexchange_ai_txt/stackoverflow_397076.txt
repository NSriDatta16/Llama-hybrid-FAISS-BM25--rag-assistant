Every function inside a class, and every class variable must take the self argument as pointed. class mystuff: def average(a,b,c): #get the average of three numbers result=a+b+c result=result/3 return result def sum(self,a,b): return a+b print mystuff.average(9,18,27) # should raise error print mystuff.sum(18,27) # should be ok If class variables are involved: class mystuff: def setVariables(self,a,b): self.x = a self.y = b return a+b def mult(self): return x * y # This line will raise an error def sum(self): return self.x + self.y print mystuff.setVariables(9,18) # Setting mystuff.x and mystuff.y print mystuff.mult() # should raise error print mystuff.sum() # should be ok
