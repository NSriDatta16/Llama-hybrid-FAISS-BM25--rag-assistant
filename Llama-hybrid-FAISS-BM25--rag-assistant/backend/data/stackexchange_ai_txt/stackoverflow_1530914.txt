Looks like its possible to leverage the KeyedCollection class: public class Set : KeyedCollection { public Set() {} public Set(IEqualityComparer comparer) : base(comparer) {} public Set(IEnumerable collection) { foreach (T elem in collection) { Add(elem); } } protected override T GetKeyForItem(T item) { return item; } }
