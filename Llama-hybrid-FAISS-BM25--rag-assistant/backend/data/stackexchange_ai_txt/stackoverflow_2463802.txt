When creating the paragraph, include a non-breaking space. var newP = $(" &#160; ").get(0); For Firefox and standards-compliant browsers, var rng = document.createRange(); rng.selectNodeContents( newP ); var sel = document.defaultView.getSelection(); sel.removeAllRanges(); sel.addRange(rng); For IE, var rng = document.body.createTextRange(); rng.moveToElementText( newP ); rng.select();
