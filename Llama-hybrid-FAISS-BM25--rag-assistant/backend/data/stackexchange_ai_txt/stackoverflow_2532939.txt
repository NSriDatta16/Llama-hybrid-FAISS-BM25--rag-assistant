Walk the graph and keep a history of visited nodes. If you visit a node again, you've detected a cycle: void CheckCycles(IOrganizationUnit unit) { var visited = new HashSet (); for (var current = unit; current != null; current = current.ManagedBy) { if (!visited.Add(current)) { throw new Exception(); // cycle detected } } }
