Bit of a kludge but this works. DragMode on the ListBox is set to dmAutomatic. procedure TForm1.ListBox1DragDrop(Sender, Source: TObject; X, Y: Integer); var iDropIdx, i: Integer; pDropPoint: TPoint; slSelected: TStrings; begin {Which item is being dropped onto?} pDropPoint := Point(X, Y); iDropIdx := ListBox1.ItemAtPos(pDropPoint, False); slSelected := TStringList.Create; try {Copy the selected items to another string list} for i := 0 to Pred(ListBox1.Items.Count) do begin if (ListBox1.Selected[i]) then slSelected.Append(ListBox1.Items[i]); end; {Find the selected items in the listbox and swap them with the drop target} for i := 0 to Pred(slSelected.Count) do begin ListBox1.Items.Exchange(ListBox1.Items.IndexOf(slSelected[i]), iDropIdx); inc(iDropIdx); end; finally slSelected.Free; end; end;
