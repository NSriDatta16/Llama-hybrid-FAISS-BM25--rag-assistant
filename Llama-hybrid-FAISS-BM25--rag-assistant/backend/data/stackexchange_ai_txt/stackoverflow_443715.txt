It's possible to use NUnit to test unmanaged code, example: // Tests.h #pragma once #include using namespace System; using namespace NUnit::Framework; namespace Tests { [TestFixture] public ref class UnitTest { public: UnitTest(void) {} [Test] void TestCos() { Assert::AreEqual(1, cos(0.0)); } }; }
