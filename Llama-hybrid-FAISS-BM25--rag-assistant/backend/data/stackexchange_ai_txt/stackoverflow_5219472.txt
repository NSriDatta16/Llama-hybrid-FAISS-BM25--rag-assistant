It is not elegant. You should pass that var inside the constructor and save it into a private $var inside the class to be used in every function with an easy $this->var . $myUnicorn = "Pink"; class Unicorn { private $currentUnicorn; public function __construct($current) { $this->currentUnicorn = $current; } public function echoIt() { echo $this->currentUnicorn; } public function killIt() { unset($this->currentUnicorn); // :( } } $u = new Unicorn($myUnicorn); $u->killIt();
