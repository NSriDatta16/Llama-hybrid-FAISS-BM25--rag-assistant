for any (x,y) if NAN you can impute to average of surrounding pixels as: if((x==0 & y==0): return (x+1)+(y+1))/2 else if(x==x_max & y==y_max): return (x-1)+(y-1))/2 else if(x==0 & y==y_max): return (x+1)+(y-1))/2 else if(x==x_max & y==0): return (x-1)+(y+1))/2 else if(x==0): return ((x+1)+(y-1)+(y+1))/3 else if(x==x_max): return ((x-1)+(y-1)+(y+1))/3 else if(y==0): return ((x+1)+(x-1)+(y+1))/3 else if(y==y_max): return ((x-1)+(x+1)+(y-1))/3 else : return ((x-1)+(x+1)+(y-1)+(y+1))/4
