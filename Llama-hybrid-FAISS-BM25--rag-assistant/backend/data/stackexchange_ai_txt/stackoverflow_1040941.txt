Revert executes when dragging has stopped (with optional delay), not on click. $(document).ready(function() { var originalTop = $('#draggable').position().top; var originalLeft = $('#draggable').position().left; $('#Zoom').toggle( function() { $("#draggable").draggable({});}, function() { $("#draggable").draggable('destroy'); $('#draggable').position().top = originalTop; $('#draggable').position().left= originalLeft; }); });
