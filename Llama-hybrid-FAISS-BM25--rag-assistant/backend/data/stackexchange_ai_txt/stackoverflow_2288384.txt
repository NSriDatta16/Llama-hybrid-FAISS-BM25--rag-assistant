Workaround: typedef void (* CLASSPROC) (void *); template void class_proc(void * ptr) { (static_cast (ptr)->*proc)(); } class ClassA { CLASSPROC m_pProc; void * m_pInstance; public: void SetProc(void *pInstance, CLASSPROC pProc) { m_pInstance = pInstance; m_pProc = pProc; } void OnSomeEvent() { m_pProc(m_pInstance); } }; class ClassB { ClassA* pCA; void Proc() { /* ... */ } void Init() { // Assume pCA != NULL pCA->SetProc(this, class_proc ); } };
