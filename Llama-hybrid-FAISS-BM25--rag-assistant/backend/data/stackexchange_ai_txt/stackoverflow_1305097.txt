Hi by using the below code i am able to insert a person into ldap from jndi program Attributes attributes=new BasicAttributes(); Attribute objectClass=new BasicAttribute("objectClass"); objectClass.add("inetOrgPerson"); attributes.put(objectClass); Attribute sn=new BasicAttribute("sn"); Attribute cn=new BasicAttribute("cn"); sn.add("sahul"); cn.add("vetcha"); attributes.put(sn); attributes.put(cn); attributes.put("title","software engg") ctx.createSubcontext("uid=sahul,ou=some organization7,o=some company7,ou=system",attributes);
