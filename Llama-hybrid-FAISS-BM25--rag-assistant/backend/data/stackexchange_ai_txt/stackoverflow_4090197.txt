The first time you run your application, the file is not there, try this instead: using (IsolatedStorageFileStream stream = storage.OpenFile(STORIES_FILE, FileMode.OpenOrCreate)) { using (StreamReader reader = new StreamReader(stream)) { contents = reader.ReadToEnd(); } }
