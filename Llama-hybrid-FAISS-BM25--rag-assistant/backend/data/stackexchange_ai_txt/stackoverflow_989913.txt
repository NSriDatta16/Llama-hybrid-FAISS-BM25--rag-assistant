I'd try something like this: interface ISumT { /// /// the sum method adds rhs to this and returns the result /// /// /// use a interface (i.e. ISumT) /// with a method (i.e. sum) /// because "where" doesn't support operators /// T sum(T rhs); } public class AutoAverage where T : ISumT { public T NextAvg( T newValue ) { // can assume that T implements the sum method } }
