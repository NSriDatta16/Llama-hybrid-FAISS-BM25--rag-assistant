Since you have multiple draggable elements then things are easier when using a class instead of an id. $(function() { $.each($( ".draggable" ), function() { $(this).draggable(); }); $( "#droppable" ).droppable({ drop: function( event, ui ) { $("#team-name").val($("img", ui.draggable).attr('alt')); }, out: function( event, ui ) { $("#team-name").val(""); } }); });
