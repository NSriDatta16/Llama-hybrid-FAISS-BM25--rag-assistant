Use a CDATA node, like this: class Program { static void Main(string[] args) { XmlDocument d = new XmlDocument(); XmlNode root = d.CreateNode(XmlNodeType.Element, "root", null); d.AppendChild(root); XmlNode cdata = d.CreateNode(XmlNodeType.CDATA, "cdata", null); cdata.InnerText = "some bolded text"; root.AppendChild(cdata); PrintDocument(d); } private static void PrintDocument(XmlDocument d) { StringWriter sw = new StringWriter(); XmlTextWriter textWriter = new XmlTextWriter(sw); d.WriteTo(textWriter); Console.WriteLine(sw.GetStringBuilder().ToString()); } } This will print bolded text]]> The CDATA section looks ugly, but that's how you insert text without having to escape characters... Otherwise, you can use the InnerXml property of a node: static void Main(string[] args) { XmlDocument d = new XmlDocument(); XmlNode root = d.CreateNode(XmlNodeType.Element, "root", null); d.AppendChild(root); XmlNode cdata = d.CreateNode(XmlNodeType.Element, "cdata", null); cdata.InnerXml = "some bolded text"; root.AppendChild(cdata); PrintDocument(d); } This prints some bolded text But pay attention when you deserialize it, as the content of the "cdata" node is now actually three nodes.
