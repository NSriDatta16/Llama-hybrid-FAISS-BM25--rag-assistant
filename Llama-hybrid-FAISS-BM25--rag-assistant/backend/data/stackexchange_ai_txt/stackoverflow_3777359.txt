SELECT StudentID, TotalSubjectMark, MarksAverage, ROW_NUMBER() OVER (ORDER BY MarksAverage DESC) ClassPosition FROM ( SELECT StudentID, SUM(SubjectMark) TotalSubjectMark, AVG(SubjectMark) MarksAverage FROM StudentMarks sm GROUP BY StudentId )
