Color color1 = Color.FromArgb(UInt1); Color color2 = Color.FromArgb(UInt2); Color averageColor = Color.FromArgb(255,(color1.R + color2.R)/2,(color1.G + color2.G)/2,(color1.B + color2.B)/2); This is assuming that you need a fully opaque average color.
