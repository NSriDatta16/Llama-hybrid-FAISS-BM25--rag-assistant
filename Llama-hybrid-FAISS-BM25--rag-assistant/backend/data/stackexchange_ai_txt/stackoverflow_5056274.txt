Yet another way is in using blocks: Class cloadClass = NSClassFromString(@"Cloud"); NSArray *a = /* you array with clouds and dragons */; NSIndexSet *clouds = [a indexesOfObjectsPassingTest: ^(id obj, NSUInteger idx, BOOL *stop) { return [obj isKindOfClass:cloadClass]; }]; // now we can count clouds NSLog(@"%d", [clouds count]); // but also we now can return our clouds immediately and NSLog(@"%@", [a objectsAtIndexes:clouds]);
