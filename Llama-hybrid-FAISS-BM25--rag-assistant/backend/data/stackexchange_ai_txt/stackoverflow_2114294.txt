You almost had it: var q = from call in callLog where call.Incoming == true group call by call.Number into g select new contacts { contact.FirstName = g.FirstName, contact.LastName = g.LastName, Count = g.Count(), Avg = g.Where(c => c.Duration > 5).Average( c => c.Duration ), Total = g.Where(c => c.Duration >= 60).Sum( c => c.Duration ) };
