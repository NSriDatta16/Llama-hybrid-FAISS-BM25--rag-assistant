Response.Clear(), seems to be the problem here. You may try this out. public static void ForcedDownload(HttpResponse response, byte[] buffer, string fileName) { response.Clear(); response.Buffer=true; response.AddHeader("content-disposition", string.Format(@"attachment;filename={0}", fileName)); response.Charset=""; response.Cache.SetCacheability(HttpCacheability.NoCache); response.ContentType="application/octet-stream"; response.BinaryWrite(buffer); response.Flush(); response.End(); }
