Assuming that your target HTML is actually valid XML, you can use a recursive function to drag out the non-div bits. static function grabNonDivContents(xml:XML):XMLList { var out:XMLList = new XMLList(); var kids:XMLList = xml.children(); for each (var kid:XML in kids) { if (kid.name() && kid.name() == "div") { var grandkids:XMLList = grabNonDivContents(kid); for each (var grandkid:XML in grandkids) { out += grandKid; } } else { out += kid; } } return out; }
