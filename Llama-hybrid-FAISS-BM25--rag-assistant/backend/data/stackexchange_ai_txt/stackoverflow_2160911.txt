Try this: var EventIDs = from EvntID in Event select EvntID.EventID; var org = (from EventOrg in cntx.EventOrganizer from MstrOrg in cntx.Organizer where EventOrg.OrganizerID == MstrOrg.OrganizerID select new {E=EventOrg, M=MstrOrg} ).ToList(); org = org .Where(o => EventIDs.Contains(o.E.EventID) ) .Select(o => new HelprClass.Organizer { OrganizerName = o.M.OrganizerName } );
