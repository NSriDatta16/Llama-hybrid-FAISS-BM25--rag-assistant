Try using a pipeline to combine principle components analysis and linear regression: from sklearn.preprocessing import StandardScaler from sklearn.decomposition import PCA from sklearn.linear_model import LinearRegression from sklearn.pipeline import Pipeline # Principle components regression steps = [ ('scale', StandardScaler()), ('pca', PCA()), ('estimator', LinearRegression()) ] pipe = Pipeline(steps) pca = pipe.set_params(pca__n_components=3) pca.fit(X, y)
