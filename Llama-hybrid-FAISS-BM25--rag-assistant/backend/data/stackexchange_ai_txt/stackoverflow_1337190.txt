You could always escape the whitespace first: function stripNonPrintable($input) { $input = preg_replace('/ /','%%%%SPACE%%%%%%', $input); $input = preg_replace('/\t/','%%%%TAB%%%%%%', $input); $input = preg_replace('/\n/','%%%%NEWLINE%%%%%%', $input); $input = preg_replace('/[\x00\x08\x0B\x0C\x0E-\x1F]/', '', $input); $input = str_replace('%%%%SPACE%%%%%%', ' ', $input); $input = str_replace('%%%%TAB%%%%%%', "\t", $input); $input = str_replace('%%%%NEWLINE%%%%%%', "\n", $input); } Not elegant, but it works.
