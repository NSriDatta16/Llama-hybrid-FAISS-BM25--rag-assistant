You could base color retrieval on color lists , either using a custom or Apple one like "Web Safe Colors" : NSColorList* cl = [NSColorList colorListNamed:@"Web Safe Colors"]; NSLog(@"%@", cl); for (NSString *key in [cl allKeys]) { NSLog(@" * %@ - %@", key, [cl colorWithKey:key]); } If you need more and the order doesn't matter, you could use e.g. +colorWithColorSpace:components:count: : uint32_t c = 0; // increment c whenever a color is requested ... CGFloat components[4] = { ((c & 0xFF0000) >> 16) / 256.0, ((c & 0x00FF00) >> 8 ) / 256.0, ((c & 0x0000FF) >> 0 ) / 256.0, 0.0 }; NSColor *color = [NSColor colorWithColorSpace:[NSColorSpace genericRGBColorSpace] components:components count:4];
