public List GetByLocation(Location l) { using (Entities entities = new Entities()) { var results = from o in entities.OrganizationSet where o.Location.Id == l.Id select o; return results.ToList (); } } Since this query will be converted to SQL, you can't do a reference comparison of l . Compare by the PK, instead.
