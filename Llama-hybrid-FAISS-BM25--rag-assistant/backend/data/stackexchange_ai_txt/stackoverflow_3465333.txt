You could move width and height into one type and move the initialization code into an initialization helper function: // header: struct Size { int width, height; Size(int w, int h) : width(w), height(h) {} }; class Image { const Size size; // public data members are usually discouraged public: Image(const char *filename); }; // implementation: namespace { Size init_helper(const char* filename) { MetaData md((readDataFromFile(filename))); return Size(md.width(), md.height()); } } Image::Image(const char* filename) : size(init_helper(filename)) {}
