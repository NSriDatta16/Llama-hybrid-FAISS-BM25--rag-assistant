Not as elegant as using Except (which I never knew existed)... but this works: List listA = new List (); List listB = new List (); listA.Add("A"); listA.Add("B"); listA.Add("C"); listA.Add("D"); listB.Add("B"); listB.Add("D"); for (int i = listA.Count - 1; i >= 0; --i) { int matchingIndex = listB.LastIndexOf(listA[i]); if (matchingIndex != -1) listB.RemoveAt(matchingIndex); }
