For non-member functions: //implementation! #ifdef _WIN32 void EatWindow(Food &) {} void DrinkWindow(Beverage &) {} void SleepWindow(Bed &) {} #else void EatLinux(Food &) {} void DrinkLinux(Beverage &) {} void SleepLinux(Bed &) {} #endif //interface #ifdef _WIN32 #define Eat(food) EatWindow(food) #define Drink(beverage) DrinkWindow(beverage) #define Sleep(bed) SleepWindow(bed) #else #define Eat(food) EatLinux(food) #define Drink(beverage) DrinkLinux(beverage) #define Sleep(bed) SleepLinux(bed) #endif Use the interfaces Eat() , Drink() and Sleep() throughout your program, be it windows or linux. No conditional check anywhere else!
