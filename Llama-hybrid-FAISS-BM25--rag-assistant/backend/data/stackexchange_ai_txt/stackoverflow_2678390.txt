I'm not sure if this will help you, but I posted some script on animating an object after it was dragged in this paste bin . Essentially, you animate the clone after the element is dropped: $("#droppable").droppable({ drop: function(e, ui) { $(this).addClass('ui-state-highlight'); var x = ui.helper.clone(); x.appendTo('body') // move clone to original drag point .css({position: 'absolute',top: ui.draggable.position().top,left:ui.draggable.position().left}) .animate({ // animate clone to droppable target top: $(this).position().top, left: $(this).position().left}, // animation time 1500, // callback function - once animation is done function(){ x.find('.caption').text("I'm being munched on!"); ui.draggable.addClass('fade'); // remove clone after 500ms setTimeout(function(){ x.remove(); }, 500) } ); // remove the helper, so it doesn't animate backwards to the starting position (built into the draggable script) ui.helper.remove(); } });
