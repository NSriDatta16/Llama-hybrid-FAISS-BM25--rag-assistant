Just put your value to be shared in a list, and assign the list to both objects. class A(object): def __init__(self, strcontainer): self.strcontainer = strcontainer def upcase(self): self.strcontainer[0] = self.strcontainer[0].upper() def __str__(self): return self.strcontainer[0] # create a string, inside a shareable list shared = ['Hello, World!'] x = A(shared) y = A(shared) # both objects have the same list print id(x.strcontainer) print id(y.strcontainer) # change value in x x.upcase() # show how value is changed in both x and y print str(x) print str(y) Prints: 10534024 10534024 HELLO, WORLD!
