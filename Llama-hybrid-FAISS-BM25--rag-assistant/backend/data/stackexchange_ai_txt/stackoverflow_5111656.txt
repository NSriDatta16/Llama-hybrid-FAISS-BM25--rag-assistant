Just store last opened infoWindow in a global variable: var activeInfoWindow; function createMarker(posn, title, html) { var marker = new google.maps.Marker({ position: posn, title: title, draggable: false }); marker['infowindow'] = new google.maps.InfoWindow({ content: html }); infoWindows.push(marker['infowindow']); google.maps.event.addListener(marker, "click", function () { if ( activeInfoWindow == this['infowindow'] ) { return; } if ( activeInfoWindow ) { activeInfoWindow.close(); } this['infowindow'].open(map, this); activeInfoWindow = this['infowindow']; }); return marker; }
