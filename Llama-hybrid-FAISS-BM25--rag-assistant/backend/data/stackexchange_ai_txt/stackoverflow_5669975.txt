My final solution is to draw and rectangle in the midle of a view and implement onTouchEvent() in my activity to set new corners coordinates like this: @Override public boolean onTouchEvent(MotionEvent me) { if(SETTING_ROI == true){ if (me.getAction() == MotionEvent.ACTION_DOWN) { START_DRAGGING = true; myView.selectCorner((int) me.getRawX(), (int) me.getRawY()); // selecst nearest corner } if (me.getAction() == MotionEvent.ACTION_MOVE){ Log.d(TAG, "ACTION_MOVE"); myView.moveCorner((int) me.getRawX(), (int) me.getRawY()); // move selected corner continuously } if (me.getAction() == MotionEvent.ACTION_UP){ if (START_DRAGGING == true) { START_DRAGGING = false; myView.moveCorner((int) me.getRawX(), (int) me.getRawY()); // final selected corner move } } } return false; }
