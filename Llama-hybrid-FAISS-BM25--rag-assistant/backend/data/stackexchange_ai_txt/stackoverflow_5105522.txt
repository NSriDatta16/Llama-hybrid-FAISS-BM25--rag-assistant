I have to applaud your use of strict and would like to encourage you to also add warnings. :) And a solution that makes the semantics a bit more clear: use strict; use warnings; use File::Slurp 'read_file'; use Array::Split qw( split_by ); my $inputfile = 'file'; my @lines = read_file( "$inputfile" ); $_ =~ s/[\r\n]//g for @lines; # remove newlines my @grouped_lines = split_by( 2, @lines ); for my $group ( @grouped_lines ) { print join ',', @{$group}; print "\n"; } __END__ 144 542 123 54 234 233 becomes: 144,542 123,54 234,233
