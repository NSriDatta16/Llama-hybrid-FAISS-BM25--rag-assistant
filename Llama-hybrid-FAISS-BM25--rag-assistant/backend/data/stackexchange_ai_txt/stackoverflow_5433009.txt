Is it elegant enough ? function replaceVars(content, vars) { return content.replace(/\$(\w+)\$/g, function($0, $1) { return ($1 in vars ? vars[$1] : $0); }); } ProfileCode = replaceVars(ProfileCode, {"FullName" : "Friedrich Hayek"}); In case associative keys don't really matter, you can always opt to use: sprintf or vsprintf EDIT What about making the 2nd parameter optional ? function replaceVars(content, scope) { if (!scope || typeof scope != "object") { scope = this; } return content.replace(/\$(\w+)\$/g, function($0, $1) { return ($1 in scope ? scope[$1] : $0); }); } // example1 var FullName = "Friedrich Hayek"; ProfileCode = replaceVars(ProfileCode); // example2 var vars = {"FullName" : "Friedrich Hayek"}; ProfileCode = replaceVars(ProfileCode, vars);
