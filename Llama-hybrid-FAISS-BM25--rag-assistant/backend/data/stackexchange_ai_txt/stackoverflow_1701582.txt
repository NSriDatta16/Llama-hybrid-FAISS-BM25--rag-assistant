How to use NSCoder

I am developing iphone application. I use NSCoder. MyApplication.h #define ITEMS_KEY @"items" #define CATEGORIES_KEY @"categories" #import @interface MyApplicationData : NSObject { NSMutableArray* items; NSMutableArray* categories; } @property (nonatomic ,retain) NSMutableArray* items; @property (nonatomic, retain) NSMutableArray* categories; @end Myapplication.m #import "MyApplicationData.h" @implementation MyApplicationData @synthesize items; @synthesize categories; #pragma mark NSCoding - (void)encodeWithCoder:(NSCoder *)aCoder{ [aCoder encodeObject:items forKey:ITEMS_KEY]; [aCoder encodeObject:categories forKey:CATEGORIES_KEY]; } -(id)initWithCoder:(NSCoder *)aDecoder{ if(self = [super init]){ self.items = [aDecoder decodeObjectForKey:ITEMS_KEY]; self.categories = [aDecoder decodeObjectForKey:CATEGORIES_KEY]; } return self; } #pragma mark - #pragma mark NSCopying -(id)copyWithZone:(NSZone *)zone{ MyApplicationData* copy = [[[self class]allocWithZone:zone]init]; items = [self.items copy]; categories = [self.categories copy]; return copy; } @end But warnning.
