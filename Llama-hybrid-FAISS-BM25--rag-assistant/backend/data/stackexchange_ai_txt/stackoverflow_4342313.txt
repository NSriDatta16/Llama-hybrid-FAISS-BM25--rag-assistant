For whatever reason, D doesn't support struct constructors that don't require arguments, either use opCall or remove the default initializer on this() struct MyStr { int[] frontArr; static MyStr opCall() { MyStr s; s.frontArr = new int[10]; return s; } void push(int x) { frontArr[0] = x; } }
