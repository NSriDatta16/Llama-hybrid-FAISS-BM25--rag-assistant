I do it like this: internal static T LoadFile (string folderName, string filePrefix) where T : class { string fileStreamName = string.Format("{0}\\{1}.dat", folderName, filePrefix); return LoadFile (fileStreamName); } internal static T LoadFile (string fileStreamName) where T : class { T retval = default(T); if (IsoStore.FileExists(fileStreamName)) { using (IsolatedStorageFileStream stream = new IsolatedStorageFileStream(fileStreamName, FileMode.OpenOrCreate, IsoStore)) { if (stream.Length > 0) { DataContractSerializer dcs = new DataContractSerializer(typeof (T)); retval = dcs.ReadObject(stream) as T; } } } return retval; }
