Not very elegant, but it's the closest you can get without doing something like you have in concreteclass3 public class Concrete : AbstractClass { public new void DoSomething() { Console.WriteLine(Value); } } public abstract class AbstractClass { protected AbstractClass() { try { var value = Value; } catch (NotImplementedException) { throw new Exception("Value's getter must be overriden in base class"); } } public void DoSomething() { Console.WriteLine(Value); } /// /// Must be override in subclass /// public string Value { get { throw new NotImplementedException(); } } }
