The proper way (using DOM-Core interfaces including the splitText() method ) would be: var p = document.getElementById('myParagraph'); var text = p.childNodes[0]; var at = 5; // create new span node with content var span = document.createElement("span"); span.appendChild(document.createTextNode('ipsum')); // Split the text node into two and add new span p.insertBefore(span, text.splitText(at));
