Your example is assuming that a PrintDocument object has been dragged onto the form from the toolbox: you could though just as easily create the object yourself. private void Print_Click(object sender, EventArgs e) { PrintDocument printDocument = new PrintDocument(); printDocument.PrintPage += new PrintPageEventHandler(printDocument_PrintPage); PrintDialog printDialog = new PrintDialog { Document = printDocument }; DialogResult result = printDialog.ShowDialog(); if (result == DialogResult.OK) { printDocument.Print(); // Raises PrintPage event } } void printDocument_PrintPage(object sender, PrintPageEventArgs e) { e.Graphics.DrawString(...); }
