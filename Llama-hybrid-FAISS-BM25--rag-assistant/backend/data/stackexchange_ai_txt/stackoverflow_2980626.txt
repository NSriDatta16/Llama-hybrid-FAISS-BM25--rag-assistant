var search_form = document.getElementById('search-form'), query_people = document.getElementById('query-people'), query_intranet = document.getElementById('query-intranet'); search_form.onsubmit = function() { if (query_people.value) { people_search(); } else if (query_intranet.value) { intranet_search(); } return false; }; function people_search() { window.location = 'http://phonebook.corporate.com/?query='+ encodeURIComponent(query_people.value); } function intranet_search() { window.location = 'http://intranet.corporate.com/?query='+ encodeURIComponent(query_intranet.value); } Of course, there are other - more elegant - ways to do it...
