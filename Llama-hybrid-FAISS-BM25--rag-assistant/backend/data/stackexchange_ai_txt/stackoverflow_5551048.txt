Check this out : Content-Script : chrome.extension.sendRequest({ action : "getCode" }, function(response) { alert(response.getCode); }); Background Page : chrome.extension.onRequest.addListener(function(request, sender, sendResponse) { case "getCode": sendResponse({ getCode: localStorage['code'] }); break; }
