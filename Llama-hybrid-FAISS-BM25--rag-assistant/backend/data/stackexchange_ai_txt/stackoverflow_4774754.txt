I've found a solution : I had a timer on my controller which proceed the action if a scroll or a dragg happend : - (void)viewDidLoad { [super viewDidLoad]; self.mapTimer = [[NSTimer scheduledTimerWithTimeInterval:1.0f target:self selector:@selector(updateDisplay:) userInfo:nil repeats:YES] retain]; } - (void)updateDisplay:(NSTimer*)theTimer { if(mapHasMoved) { NSLog(@"updateDisplay"); [self updateProducts]; mapHasMoved = FALSE; } } - (void)endZoomingOrScrollingHandler{ mapHasMoved = TRUE; }
