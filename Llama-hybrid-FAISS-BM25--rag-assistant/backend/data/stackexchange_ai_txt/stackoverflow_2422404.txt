What I did for this on a project was to set a RadioGroup as the control with no items so it appeared as blank. private void viewTodoList_CustomRowCellEdit(object sender, DevExpress.XtraGrid.Views.Grid.CustomRowCellEditEventArgs e) { if (e.Column == CheckMarkColumn) { if (ConditionIsMet()) { e.RepositoryItem = new DevExpress.XtraEditors.Repository.RepositoryItemRadioGroup(); } } }
