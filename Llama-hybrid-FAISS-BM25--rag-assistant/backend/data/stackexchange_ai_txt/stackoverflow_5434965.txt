Add a debug function in your code to copy the required files from the protected folders onto your phone's SD card. You can then access them from your PC. public static void backupFile() throws IOException { String inFileName = "/data/data/your.package.name/........"; //TODO Use folder/filename File inFile = new File(inFileName); FileInputStream fis = new FileInputStream(inFile); String outFileName = Environment.getExternalStorageDirectory()+"/........"; //TODO Use output filename OutputStream output = new FileOutputStream(outFileName); byte[] buffer = new byte[1024]; int length; while ((length = fis.read(buffer))>0) output.write(buffer, 0, length); output.flush(); output.close(); fis.close(); }
