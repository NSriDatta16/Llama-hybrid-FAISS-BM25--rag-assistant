Declare this in your header file: - (void)processTouch:(UITouch *)touch; Then use this in the *.m file: - (void)touchesBegan:(NSSet *)touches withEvent:(UIEvent *)event { [self processTouch:[touches anyObject]]; } - (void)processTouch:(UITouch *)touch { NSLog(@"You just touch somewhere on the screen!"); } This code will run the "processTouch" method each time a user touches ANYWERE on the iPhone screen.
