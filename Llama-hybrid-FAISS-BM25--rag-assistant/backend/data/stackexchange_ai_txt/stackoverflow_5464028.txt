You have to build pcap as static module. Something like this include $(CLEAR_VARS) LOCAL_MODULE := pcap LOCAL_CFLAGS := declare_flags LOCAL_C_INCLUDES := declare_include LOCAL_SRC_FILES := src_files include $(BUILD_STATIC_LIBRARY) This build pcap as local module which you could link with LOCAL_STATIC_LIBRARIES just add LOCAL_STATIC_LIBRARIES := pcap before calling include $(BUILD_SHARED_LIBRARY)
