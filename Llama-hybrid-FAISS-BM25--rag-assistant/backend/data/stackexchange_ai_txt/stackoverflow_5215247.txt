On MouseDown set a boolean variable to indicate that other events should be ignored. private bool IgnoreEvents { set; get; } ... protected void Control_MouseDown(object sender, EventArgs e) { //Dragging or holding. this.IgnoreEvents = true; } protected void Control_MouseUp(object sender, EventArgs e) { //Finished dragging or holding. this.IgnoreEvents = false; } protected void OtherControl_SomeOtherEvent(object sender, EventArgs e) { if (this.IgnoreEvents) return; //Otherwise to normal stuff here. } You'll need to tweak this to work with your code of course but it should be a good start.
