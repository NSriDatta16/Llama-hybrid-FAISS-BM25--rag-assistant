Since Variables looks global you can use it inside the mySizer routine. If you don't want to have global variables you can create a closure : $('.draggable').resize(function() { if (Variables.ResizeTimer !== false) { clearTimeout(Variables.ResizeTimer); } var $this = $(this); Variables.ResizeTimer = setTimeout(function() { mySizer($this); }, 1000); });
