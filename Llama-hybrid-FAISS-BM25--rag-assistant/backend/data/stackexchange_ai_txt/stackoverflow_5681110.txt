select EffectiveDate AS [Month], count(MemberID) AS Agents, sum(ActualPremium) AS Premium from vwMemberPolicyInfo Where dateadd(year,0,EffectiveDate) = DATEADD(year, 0, GETDATE()) GROUP BY EffectiveDate dateadd(year,0,EffectiveDate) = DATEADD(year, -1, GETDATE()) dateadd(year,-1,EffectiveDate) = DATEADD(year, -1, GETDATE())
