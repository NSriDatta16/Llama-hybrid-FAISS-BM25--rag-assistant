Yes you can - something like CREATE VIEW [mySchema].vwItem AS WITH cteItem AS ( SELECT item_id AS 'ItemID', item_level AS 'ItemLevel', item_number_type AS 'ItemNumberType', CASE item_level WHEN 1 THEN item_id WHEN 2 THEN item_parent_id WHEN 3 THEN item_grandparent_id ELSE item_id END AS 'ItemParent', FROM [mySchema].[ITEM] ) SELECT child.*, parent.item_id AS 'ParentId', parent.desc AS 'ParentDescription' FROM cteItem child LEFT OUTER JOIN [mySchema].[ITEM] parent ON child.ItemParent = parent.item_id
