Use the following code: this.addEventListener(MouseEvent.MOUSE_DOWN, dragMovie); this.addEventListener(MouseEvent.MOUSE_UP, dropMovie); this.buttonMode = true; private function dragMovie(event:MouseEvent):void { this.startDrag(); } private function dropMovie(event:MouseEvent):void { this.stopDrag(); }
