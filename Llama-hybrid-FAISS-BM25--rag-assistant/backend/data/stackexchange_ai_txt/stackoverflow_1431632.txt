You should call that init function once the DOM is ready. function initDivMouseDown(id) { try { jQuery('#'+ id).draggable({ opacity: 0.5, revert: 'invalid', drag:function(ev,ui) { // ADDED DIV ID HERE alert($(this).attr('id') + " is being dragged"); } }); } catch (e) { alert ("exception in DIVmouse down: "+ e); } } jQuery(document).ready(function() { initDivMouseDown("my_div"); });
