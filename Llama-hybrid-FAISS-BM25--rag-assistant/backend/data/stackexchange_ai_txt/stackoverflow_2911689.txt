Answering my own, maybe i my question was wrong. I wanted list of default OSes returned either from Request.Browser.Platform or Request["HTTP_USER_AGENT"] so i can enter those into DB. I've got the list and wrote a function to check what kind of OS user uses. public static string getOsFromUserAgent(string userAgent) { string visitorOS = "other"; Dictionary osNamesAndRegexes = new Dictionary (); osNamesAndRegexes.Add("Windows 98", "(Windows 98)|(Win98)"); osNamesAndRegexes.Add("Windows XP", "(Windows NT 5.1)|(Windows XP)"); osNamesAndRegexes.Add("Windows Vista", "(Windows NT 6.0)"); osNamesAndRegexes.Add("Windows 7", "(Windows NT 7.0)"); osNamesAndRegexes.Add("Linux", "(Linux)|(X11)"); osNamesAndRegexes.Add("Mac OS", "(Mac_PowerPC)|(Macintosh)"); foreach (KeyValuePair kvpPair in osNamesAndRegexes) { if (Regex.IsMatch(userAgent, kvpPair.Value, System.Text.RegularExpressions.RegexOptions.IgnoreCase)) { visitorOS = kvpPair.Key; } } return visitorOS; }
