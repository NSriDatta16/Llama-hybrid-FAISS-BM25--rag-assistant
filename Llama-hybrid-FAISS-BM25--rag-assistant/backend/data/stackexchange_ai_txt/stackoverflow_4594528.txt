I wrote a project that does just this, it uses version 2 of Google Maps API. Here is the snippet of code to make a draggable marker that will show lat/long after you drag it to a new place on map: function createMarker(latlng, number, html) { var marker = new GMarker(latlng, {draggable: true}); marker.value = number; var myHtml = html; var center = marker.getLatLng(); map.openInfoWindowHtml(latlng, " " + myHtml + " " + center.toString() + " "); GEvent.addListener(marker, "dragstart", function() { map.closeInfoWindow(); }); GEvent.addListener(marker, "dragend", function() { var center = marker.getLatLng(); marker.openInfoWindowHtml(" " + myHtml + " " + center.toString() + " "); }); return marker; }
