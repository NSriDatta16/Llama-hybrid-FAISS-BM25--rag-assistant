something like this query should work (I've set up my own temp table with your data) create table #Temp (Name varchar(100), [Count] int) insert into #Temp (Name, [Count]) VALUES ('Kaganoff Benzion', 122) insert into #Temp (Name, [Count]) VALUES ('Van Gennep', 443) insert into #Temp (Name, [Count]) VALUES ('Michelen Luis', 656) insert into #Temp (Name, [Count]) VALUES ('kraig Beno', 333) insert into #Temp (Name, [Count]) VALUES ('Mogrobejo Endika', 555) select SUBSTRING(Name, 1, PATINDEX('% %', Name)) AS FirstName, SUBSTRING(Name, PATINDEX('% %', Name) + 1, LEN(Name) - PATINDEX('% %', Name)) AS SecondName, [Count] from #Temp ORDER BY SUBSTRING(Name, 1, 1), SUBSTRING(Name, PATINDEX('% %', Name) + 1, 1), [Count] DESC drop table #Temp
