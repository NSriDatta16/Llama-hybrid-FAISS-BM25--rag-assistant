You need to call end() on the request to signal that you are ready to send it. Also you should add a user-agent header to your request. Many web servers look for it. console.log('preparing request to ' + url) u = require('url').parse(url) var remote_client = http.createClient(80, u['host']); var request = remote_client.request("GET", u['pathname'], {"host": u['host'], "user-agent": "node.js"}); console.log("request made") request.addListener('response', function (response) { response.setEncoding('binary') var body = ''; response.addListener('data', function (chunk) { body += chunk; console.log('chunk received') }); }); request.end();
