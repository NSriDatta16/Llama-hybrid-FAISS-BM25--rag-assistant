I figured this problem out, I did have to add the extra LeaderId property to represent the foreign key, then mark this as the foreign key in the Associate attribute named parameter 'ThisKey' like so: Meet.cs [Table(Name = "dbo.Meets")] public class Meet { private EntityRef _leader; public Meet() { _leader = default(EntityRef ); } [Column(IsPrimaryKey = true, IsDbGenerated = true)] public int MeetId { get; set; } [Column] public string Location { get; set; } [Column] public DateTime Date { get; set; } [Column] public int LeaderId { get; set; } [Association(Name = "Member_Meet", Storage = "_leader", ThisKey = "LeaderId", OtherKey = "MemberId", IsForeignKey = true)] public Member Leader { get { return _leader.Entity; } set { _leader.Entity = value; } } }
