Reslove this problem. ^^ i think for drag pin, pin is already selected. so selected MKPinAnnotationView when it init. my new code. - (MKAnnotationView *)mapView:(MKMapView *)mapView viewForAnnotation:(id )annotation { MKPinAnnotationView *annotationView = [[MKPinAnnotationView alloc] initWithAnnotation:annotation reuseIdentifier:@"pin"]; [annotationView setDraggable:YES]; annotationView.pinColor = MKPinAnnotationColorPurple; [annotationView setSelected:YES animated:YES]; return [annotationView autorelease]; }
